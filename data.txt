				if(appInSchCargo.getSrc_app_ind()==null)
					appInSchCargo.setSrc_app_ind(AppConstants.SRC_APP_IND_RMC);
				
				//Added by Sanjeeva Nemani for date formattting BRGUS00304167- Start 
				if (appInSchCargo.getExpected_graduation_dt() != null
						&& appInSchCargo.getExpected_graduation_dt().trim().length() != 0
						&& !appInSchCargo.getExpected_graduation_dt().equals(
								AppConstants.HIGH_DATE)) {
					expectGradDt = appInSchCargo.getExpected_graduation_dt();
					if (expectGradDt.length() == 10
							&& expectGradDt.charAt(2) == '/') {
						expectGradDt = displayFormatter
								.getYYYYMMDDDate(expectGradDt);
						appInSchCargo.setExpected_graduation_dt(expectGradDt);
					}
				} else {
					appInSchCargo.setExpected_graduation_dt(AppConstants.HIGH_DATE);
				}
				// Added by Sanjeeva Nemani for date formattting BRGUS00304167- End
				
				
//				Setting the INDV_SEQ_NUM
				appInSchCargo.setIndv_seq_num(indvSeqNumber);
				//Checking the Before Collection is Null or Not
				if(appInSchBeforeColl != null){
					appInSchBeforeCargo = appInSchBeforeColl.getResult(0);
					//Now I am comparing the Cargos
					appInSchCargo = (APP_IN_SCHLE_Cargo)isChanged(appInSchBeforeCargo,appInSchCargo);
					if(appInSchCargo.isDirty()){
						appInSchCargo.setRowAction(FwConstants.ROWACTION_UPDATE);
						schoolEnrollmentBo.storeSchoolEnrollmentDetails(appInSchColl);
						List<String> textMesage = null;
						if(null!= session.get("updateSucess") ){
							 textMesage = (ArrayList<String>)session.get("updateSucess");
						}else {
							textMesage = new ArrayList<String>();
						}
						textMesage.add("32209");
						session.put("updateSucess",textMesage);
					}
				}
				else{
					//We need to insert new record to the data base
					appInSchCargo.setRowAction(FwConstants.ROWACTION_INSERT);	
					schoolEnrollmentBo.storeSchoolEnrollmentDetails(appInSchColl);
					List<String> textMesage = null;
					if(null!= session.get("updateSucess") ){
						 textMesage = (ArrayList<String>)session.get("updateSucess");
					}else {
						textMesage = new ArrayList<String>();
					}
					textMesage.add("32209");
					session.put("updateSucess",textMesage);
				}
			}
			
			boolean pageComplete = false;
			if(currentPageStatus == FwConstants.DRIVER_REQUIRED || currentPageStatus == FwConstants.DRIVER_VISIT_AGAIN){
				//Now checking for any persons need to be displayed
				int currentArrayIndex = Integer.parseInt((String)beforeColl.get(FwConstants.CURRENT_RECORD_INDEX));  
				if(currentArrayIndex + 1 == schoolEnrollList.size()){
					pageComplete = true;
				}
			}
			//If the driver status is add new and driver status is complete 					
			if( pageComplete || currentPageStatus == FwConstants.DRIVER_ADD_NEW || currentPageStatus == FwConstants.DRIVER_COMPLETE){
				driverManager.makeComplete(FwConstants.RMC_DRIVER,(String)request.get(FwConstants.CURRENT_PAGE_ID),driverArray,appNumber);
			}
			else{
				session.put(FwConstants.BEFORE_COLLECTION,beforeColl);
			}
			log(ILog.INFO, "RMCHouseHoldInfoEJBBean::storeHouseHoldInfoSchoolEnrollmentDetails:End");
		} catch(FwException fe) {
			ctx.setRollbackOnly();
			FwWrappedException we = new FwWrappedException();
			we.setCallingClassID(this.getClass().getName());
			we.setCallingMethodID("storeHouseHoldInfoSchoolEnrollmentDetails");
			we.setFwException(fe);
			throw we;
		} catch(Exception e) {
			ctx.setRollbackOnly();
			FwException fe = createFwException(this.getClass().getName(), "storeHouseHoldInfoSchoolEnrollmentDetails", e);
			throw fe;
		}
	}
