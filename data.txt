WITH DATES AS (
    SELECT
        SYSTEM_DATE,
        DATE(
            RTRIM(CHAR(PROCESS_CCYY)) || '-' ||
            CASE
                WHEN PROCESS_MM < 10 THEN '0' || RTRIM(CHAR(PROCESS_MM))
                ELSE RTRIM(CHAR(PROCESS_MM))
            END || '-01'
        ) AS CURR_MTH_BEG,
        MTH_END_DT AS CURR_MTH_END,
        CASE
            WHEN PROCESS_MM > 6
                THEN DATE(RTRIM(CHAR(PROCESS_CCYY)) || '-07-01')
            ELSE DATE(RTRIM(CHAR(PRV_PROCESS_CCYY)) || '-07-01')
        END AS FISCAL_START_DATE,
        CASE
            WHEN PROCESS_MM > 6
                THEN DATE(RTRIM(CHAR(PROCESS_CCYY + 1)) || '-06-30')
            ELSE DATE(RTRIM(CHAR(PROCESS_CCYY)) || '-06-30')
        END AS FISCAL_END_DATE,
        CASE
            WHEN (PROCESS_MM + 1) > 12
                THEN DATE(RTRIM(CHAR(PROCESS_CCYY - 3)) || '-01-01')
            ELSE DATE(
                RTRIM(CHAR(PROCESS_CCYY - 3)) || '-' ||
                CASE
                    WHEN (PROCESS_MM + 1) < 10 THEN '0' || RTRIM(CHAR(PROCESS_MM + 1))
                    ELSE RTRIM(CHAR(PROCESS_MM + 1))
                END || '-01'
            )
        END AS DT_36
    FROM {?SCHEMA}.TWNSYSDATE
),

CASE_OFFICE AS (
    SELECT DISTINCT
        FC.ICI,
        FC.OFC_CD,
        O.OFC_NM
    FROM {?SCHEMA}.TWNFAM_CASELD_CASE FC
    JOIN {?SCHEMA}.TWNOFFICE O
      ON FC.OFC_CD = O.OFC_CD
    JOIN {?SCHEMA}.TWNPRGM_CASE_HSTRY PH
      ON FC.ICI = PH.ICI
     AND PH.PRGM_CASE_TYPE = 'FS'
    WHERE FC.OFC_CD BETWEEN 'AA' AND 'ZZ'
),

/* =========================
   LINE 1 MONTH / FISCAL
   ========================= */
LINE1_BY_OFFICE AS (
    SELECT
        X.OFC_CD,
        X.OFC_NM,
        COUNT(DISTINCT CASE WHEN X.IN_FISCAL = 1 THEN X.UPI END) AS LINE1_FIS,
        COUNT(DISTINCT CASE WHEN X.IN_MTH = 1 THEN X.UPI END) AS LINE1_MTH
    FROM (
        SELECT
            CO.OFC_CD,
            CO.OFC_NM,
            P.UPI,
            CASE
                WHEN P.PRD_BEG_DT >= D.FISCAL_START_DATE
                 AND P.PRD_END_DT <= D.CURR_MTH_END
                 AND P.ABAWD_FG NOT IN ('X','N') THEN 1 ELSE 0
            END AS IN_FISCAL,
            CASE
                WHEN P.PRD_BEG_DT >= D.CURR_MTH_BEG
                 AND P.PRD_END_DT <= D.CURR_MTH_END
                 AND P.ABAWD_FG NOT IN ('X','N') THEN 1 ELSE 0
            END AS IN_MTH
        FROM {?SCHEMA}.TWNPERSON_ELIG_HST P
        JOIN {?SCHEMA}.TWNCASE_MEMBER_HST M
          ON P.UPI = M.UPI
         AND P.PRD_BEG_DT = M.PRD_BEG_DT
         AND P.PRD_END_DT = M.PRD_END_DT
        JOIN CASE_OFFICE CO
          ON M.ICI = CO.ICI
        CROSS JOIN DATES D
        WHERE P.PRD_BEG_DT >= D.DT_36
          AND P.PRD_END_DT <= D.CURR_MTH_END
    ) X
    GROUP BY X.OFC_CD, X.OFC_NM
),

/* =========================
   Y COUNTS
   ========================= */
Y_UPI_COUNTS AS (
    SELECT
        X.OFC_CD,
        X.OFC_NM,
        X.UPI,
        SUM(CASE WHEN X.IN_MTH = 1 THEN 1 ELSE 0 END) AS Y_MTH_CNT,
        SUM(CASE WHEN X.IN_FIS = 1 THEN 1 ELSE 0 END) AS Y_FIS_CNT,
        SUM(CASE WHEN X.IN_36 = 1 AND X.IN_MTH = 0 THEN 1 ELSE 0 END) AS Y_36_EXCL_MTH,
        SUM(CASE WHEN X.IN_36 = 1 AND X.IN_FIS = 0 THEN 1 ELSE 0 END) AS Y_36_EXCL_FIS
    FROM (
        SELECT
            CO.OFC_CD,
            CO.OFC_NM,
            P.UPI,
            CASE
                WHEN P.PRD_BEG_DT >= D.CURR_MTH_BEG
                 AND P.PRD_END_DT <= D.CURR_MTH_END THEN 1 ELSE 0
            END AS IN_MTH,
            CASE
                WHEN P.PRD_BEG_DT >= D.FISCAL_START_DATE
                 AND P.PRD_END_DT <= D.CURR_MTH_END THEN 1 ELSE 0
            END AS IN_FIS,
            CASE
                WHEN P.PRD_BEG_DT >= D.DT_36
                 AND P.PRD_END_DT <= D.CURR_MTH_END THEN 1 ELSE 0
            END AS IN_36
        FROM {?SCHEMA}.TWNPERSON_ELIG_HST P
        JOIN {?SCHEMA}.TWNCASE_MEMBER_HST M
          ON P.UPI = M.UPI
         AND P.PRD_BEG_DT = M.PRD_BEG_DT
         AND P.PRD_END_DT = M.PRD_END_DT
        JOIN CASE_OFFICE CO
          ON M.ICI = CO.ICI
        CROSS JOIN DATES D
        WHERE P.ABAWD_FG = 'Y'
          AND P.PRD_BEG_DT >= D.DT_36
          AND P.PRD_END_DT <= D.CURR_MTH_END
    ) X
    GROUP BY X.OFC_CD, X.OFC_NM, X.UPI
),

Y_BY_OFFICE AS (
    SELECT
        OFC_CD,
        OFC_NM,
        SUM(CASE
                WHEN CASE WHEN Y_MTH_CNT = 1 THEN Y_MTH_CNT + Y_36_EXCL_MTH ELSE Y_MTH_CNT END = 1
                THEN 1 ELSE 0
            END) AS WS_MTH_Y_CNT_1O,
        SUM(CASE
                WHEN CASE WHEN Y_MTH_CNT = 1 THEN Y_MTH_CNT + Y_36_EXCL_MTH ELSE Y_MTH_CNT END = 2
                THEN 1 ELSE 0
            END) AS WS_MTH_Y_CNT_20,
        SUM(CASE
                WHEN CASE WHEN Y_MTH_CNT = 1 THEN Y_MTH_CNT + Y_36_EXCL_MTH ELSE Y_MTH_CNT END = 3
                THEN 1 ELSE 0
            END) AS WS_MTH_Y_CNT_30,

        SUM(CASE
                WHEN CASE WHEN Y_FIS_CNT > 0 THEN Y_FIS_CNT + Y_36_EXCL_FIS ELSE Y_FIS_CNT END = 1
                THEN 1 ELSE 0
            END) AS WS_FYR_Y_CNT_1O,
        SUM(CASE
                WHEN CASE WHEN Y_FIS_CNT > 0 THEN Y_FIS_CNT + Y_36_EXCL_FIS ELSE Y_FIS_CNT END = 2
                THEN 1 ELSE 0
            END) AS WS_FYR_Y_CNT_20,
        SUM(CASE
                WHEN CASE WHEN Y_FIS_CNT > 0 THEN Y_FIS_CNT + Y_36_EXCL_FIS ELSE Y_FIS_CNT END = 3
                THEN 1 ELSE 0
            END) AS WS_FYR_Y_CNT_30
    FROM Y_UPI_COUNTS
    GROUP BY OFC_CD, OFC_NM
),

/* =========================
   R COUNTS
   ========================= */
R_UPI_COUNTS AS (
    SELECT
        X.OFC_CD,
        X.OFC_NM,
        X.UPI,
        SUM(CASE WHEN X.IN_MTH = 1 THEN 1 ELSE 0 END) AS R_MTH_CNT,
        SUM(CASE WHEN X.IN_PREV_MTH = 1 THEN 1 ELSE 0 END) AS R_PREV_MTH_CNT,
        SUM(CASE WHEN X.IN_FIS = 1 THEN 1 ELSE 0 END) AS R_FIS_CNT,
        SUM(CASE WHEN X.IN_36 = 1 AND X.IN_MTH = 0 THEN 1 ELSE 0 END) AS R_36_EXCL_MTH,
        SUM(CASE WHEN X.IN_36 = 1 AND X.IN_FIS = 0 THEN 1 ELSE 0 END) AS R_36_EXCL_FIS
    FROM (
        SELECT
            CO.OFC_CD,
            CO.OFC_NM,
            P.UPI,
            CASE
                WHEN P.PRD_BEG_DT >= D.CURR_MTH_BEG
                 AND P.PRD_END_DT <= D.CURR_MTH_END THEN 1 ELSE 0
            END AS IN_MTH,
            CASE
                WHEN P.PRD_BEG_DT >= D.PREV_MTH_BEG
                 AND P.PRD_END_DT <= D.PREV_MTH_END THEN 1 ELSE 0
            END AS IN_PREV_MTH,
            CASE
                WHEN P.PRD_BEG_DT >= D.FISCAL_START_DATE
                 AND P.PRD_END_DT <= D.CURR_MTH_END THEN 1 ELSE 0
            END AS IN_FIS,
            CASE
                WHEN P.PRD_BEG_DT >= D.DT_36
                 AND P.PD_END_DT <= D.CURR_MTH_END THEN 1 ELSE 0
            END AS IN_36
        FROM {?SCHEMA}.TWNPERSON_ELIG_HST P
        JOIN {?SCHEMA}.TWNCASE_MEMBER_HST M
          ON P.UPI = M.UPI
         AND P.PRD_BEG_DT = M.PRD_BEG_DT
         AND P.PRD_END_DT = M.PRD_END_DT
        JOIN CASE_OFFICE CO
          ON M.ICI = CO.ICI
        CROSS JOIN DATES D
        WHERE P.ABAWD_FG = 'R'
          AND P.PRD_BEG_DT >= D.DT_36
          AND P.PRD_END_DT <= D.CURR_MTH_END
    ) X
    GROUP BY X.OFC_CD, X.OFC_NM, X.UPI
),

R_BY_OFFICE AS (
    SELECT
        OFC_CD,
        OFC_NM,
        SUM(CASE
                WHEN CASE WHEN R_MTH_CNT = 1 THEN R_MTH_CNT + R_36_EXCL_MTH ELSE R_MTH_CNT END = 1
                THEN 1 ELSE 0
            END) AS WS_MTH_R_CNT_1O,
        SUM(CASE
                WHEN CASE WHEN R_MTH_CNT = 1 THEN R_MTH_CNT + R_36_EXCL_MTH ELSE R_MTH_CNT END = 2
                THEN 1 ELSE 0
            END) AS WS_MTH_R_CNT_20,
        SUM(CASE
                WHEN CASE WHEN R_MTH_CNT = 1 THEN R_MTH_CNT + R_36_EXCL_MTH ELSE R_MTH_CNT END = 3
                THEN 1 ELSE 0
            END) AS WS_MTH_R_CNT_30,

        COUNT(DISTINCT CASE
            WHEN R_PREV_MTH_CNT > 0
             AND CASE WHEN R_MTH_CNT = 1 THEN R_MTH_CNT + R_36_EXCL_MTH ELSE R_MTH_CNT END <> 1
            THEN UPI
        END) AS LINE6_MTH,

        SUM(CASE
                WHEN CASE WHEN R_FIS_CNT > 0 THEN R_FIS_CNT + R_36_EXCL_FIS ELSE R_FIS_CNT END = 1
                THEN 1 ELSE 0
            END) AS WS_FYR_R_CNT_1O,
        SUM(CASE
                WHEN CASE WHEN R_FIS_CNT > 0 THEN R_FIS_CNT + R_36_EXCL_FIS ELSE R_FIS_CNT END = 2
                THEN 1 ELSE 0
            END) AS WS_FYR_R_CNT_20,
        COUNT(DISTINCT CASE
                WHEN CASE WHEN R_FIS_CNT > 0 THEN R_FIS_CNT + R_36_EXCL_FIS ELSE R_FIS_CNT END = 3
                THEN UPI
            END) AS WS_FYR_R_CNT_30,
        COUNT(DISTINCT CASE WHEN R_FIS_CNT > 1 THEN UPI END) AS LINE6_FIS
    FROM R_UPI_COUNTS
    GROUP BY OFC_CD, OFC_NM
),

/* =========================
   Z / O
   ========================= */
ZO_BY_OFFICE AS (
    SELECT
        CO.OFC_CD,
        CO.OFC_NM,
        COUNT(DISTINCT CASE
            WHEN P.ABAWD_FG = 'Z'
             AND P.PRD_BEG_DT >= D.FISCAL_START_DATE
             AND P.PRD_END_DT <= D.FISCAL_END_DATE
            THEN P.UPI END) AS WS_3B_FIS,
        COUNT(DISTINCT CASE
            WHEN P.ABAWD_FG = 'O'
             AND P.PRD_BEG_DT >= D.FISCAL_START_DATE
             AND P.PRD_END_DT <= D.FISCAL_END_DATE
            THEN P.UPI END) AS WS_3C_FIS,
        COUNT(DISTINCT CASE
            WHEN P.ABAWD_FG = 'Z'
             AND P.PRD_BEG_DT >= D.CURR_MTH_BEG
             AND P.PRD_END_DT <= D.CURR_MTH_END
            THEN P.UPI END) AS WS_3B_MTH,
        COUNT(DISTINCT CASE
            WHEN P.ABAWD_FG = 'O'
             AND P.PRD_BEG_DT >= D.CURR_MTH_BEG
             AND P.PRD_END_DT <= D.CURR_MTH_END
            THEN P.UPI END) AS WS_3C_MTH
    FROM {?SCHEMA}.TWNPERSON_ELIG_HST P
    JOIN {?SCHEMA}.TWNCASE_MEMBER_HST M
      ON P.UPI = M.UPI
     AND P.PRD_BEG_DT = M.PRD_BEG_DT
     AND P.PRD_END_DT = M.PRD_END_DT
    JOIN CASE_OFFICE CO
      ON M.ICI = CO.ICI
    CROSS JOIN DATES D
    WHERE P.ABAWD_FG IN ('Z','O')
      AND P.PRD_BEG_DT >= D.DT_36
      AND P.PRD_END_DT <= D.CURR_MTH_END
    GROUP BY CO.OFC_CD, CO.OFC_NM
)

/* add W branches + line5 branches the same way */

SELECT
    COALESCE(L1.OFC_NM, Y.OFC_NM, R.OFC_NM, ZO.OFC_NM) AS OFC_NM,
    COALESCE(L1.OFC_CD, Y.OFC_CD, R.OFC_CD, ZO.OFC_CD) AS OFC_CD,

    COALESCE(L1.LINE1_FIS, 0) AS LINE1_FIS,
    COALESCE(L1.LINE1_MTH, 0) AS LINE1_MTH,

    COALESCE(Y.WS_MTH_Y_CNT_1O, 0) AS WS_MTH_Y_CNT_1O,
    COALESCE(Y.WS_MTH_Y_CNT_20, 0) AS WS_MTH_Y_CNT_20,
    COALESCE(Y.WS_MTH_Y_CNT_30, 0) AS WS_MTH_Y_CNT_30,
    COALESCE(Y.WS_FYR_Y_CNT_1O, 0) AS WS_FYR_Y_CNT_1O,
    COALESCE(Y.WS_FYR_Y_CNT_20, 0) AS WS_FYR_Y_CNT_20,
    COALESCE(Y.WS_FYR_Y_CNT_30, 0) AS WS_FYR_Y_CNT_30,

    COALESCE(R.WS_MTH_R_CNT_1O, 0) AS WS_MTH_R_CNT_1O,
    COALESCE(R.WS_MTH_R_CNT_20, 0) AS WS_MTH_R_CNT_20,
    COALESCE(R.WS_MTH_R_CNT_30, 0) AS WS_MTH_R_CNT_30,
    COALESCE(R.LINE6_MTH, 0) AS LINE6_MTH,
    COALESCE(R.WS_FYR_R_CNT_1O, 0) AS WS_FYR_R_CNT_1O,
    COALESCE(R.WS_FYR_R_CNT_20, 0) AS WS_FYR_R_CNT_20,
    COALESCE(R.WS_FYR_R_CNT_30, 0) AS WS_FYR_R_CNT_30,
    COALESCE(R.LINE6_FIS, 0) AS LINE6_FIS,

    COALESCE(ZO.WS_3B_FIS, 0) AS WS_3B_FIS,
    COALESCE(ZO.WS_3C_FIS, 0) AS WS_3C_FIS,
    COALESCE(ZO.WS_3B_MTH, 0) AS WS_3B_MTH,
    COALESCE(ZO.WS_3C_MTH, 0) AS WS_3C_MTH

FROM LINE1_BY_OFFICE L1
FULL OUTER JOIN Y_BY_OFFICE Y
  ON L1.OFC_CD = Y.OFC_CD
FULL OUTER JOIN R_BY_OFFICE R
  ON COALESCE(L1.OFC_CD, Y.OFC_CD) = R.OFC_CD
FULL OUTER JOIN ZO_BY_OFFICE ZO
  ON COALESCE(L1.OFC_CD, Y.OFC_CD, R.OFC_CD) = ZO.OFC_CD

FOR FETCH ONLY WITH UR
