private void setSuaFlags(SnapCharacteristicControlParam controlParam, MemberRecord memberRecord) {
    List<UtilityExpenseSorRecord> rows = this.dataService.getUtilityDetails(memberRecord.getUpi());
    for (UtilityExpenseSorRecord u : rows) {
        String ind = u.getStandardUtilityAllowanceIndicator();
        if ("Y".equalsIgnoreCase(ind) || "I".equalsIgnoreCase(ind) || "L".equalsIgnoreCase(ind)) {
            Integer n = u.getNumberSharingSUA();
            if (n != null && n > 1) {
                controlParam.setPsuaFlag(CHAR_Y);
            } else {
                controlParam.setSuaFlag(CHAR_Y);
            }
        }
    }
}
