			Map session = txnBean.getSession();
			Map request = txnBean.getRequest();
			Map pageCollection = txnBean.getPageCollection();
			
//			Get beforeMap
			Map beforeMap =(Map) session.get(FwConstants.BEFORE_COLLECTION);
			
			//Get driver info.
			int[] driverArray = (int[]) session.get(FwConstants.RMC_DRIVER);

			//Get appNumber from the session
			String appNum = (String)session.get(AppConstants.APP_NUMBER);
			
			//Get the pageCollection INDV_LIVE_CHG_Collection

			ARHCSValidator validator = new ARHCSValidator();
			String indvSeqNum = appInDablColl.getCargo(0).getIndv_seq_num();
//			Setting IndivSeqTypeBean
			IndivTypeSeqBean indivTypeSeqBean = new IndivTypeSeqBean();
			indivTypeSeqBean.setIndivSeqNum(indvSeqNum);
			indivTypeSeqBean.setType(AppConstants.DISABLED_BLIND);

			validator.validateErase(AppConstants.DISABLED_BLIND,name);
			if(validator.hasMessages()){
				String reqWarningMsgs =(String)request.get(FwConstants.WARNING_MSG_DETAILS);
				warningMsgFlag = this.checkForWarningMesgs(reqWarningMsgs,validator.getMessageList());
				if(!warningMsgFlag){
					request.put(FwConstants.MESSAGE_LIST,validator.getMessageList()); 
					txnBean.setPageCollection((Map) session.get(FwConstants.BEFORE_COLLECTION));
					beforeMap.put(AppConstants.INDV_TYPE_SEQ_BEAN,indivTypeSeqBean);
					return;
				}
				else
				{
					IndivTypeSeqBean postIndivSeqBean = (IndivTypeSeqBean)beforeMap.get(AppConstants.INDV_TYPE_SEQ_BEAN);
					boolean sameIndivSeqNum = (postIndivSeqBean.getIndivSeqNum().equals(indivTypeSeqBean.getIndivSeqNum()))?true:false;
					boolean sameType = (postIndivSeqBean.getType().equals(indivTypeSeqBean.getType()))?true:false;
					if((sameIndivSeqNum == false) || (sameType == false))
					{
						request.put(FwConstants.MESSAGE_LIST,validator.getMessageList());						
						session.put(FwConstants.BEFORE_COLLECTION,beforeMap);					
						txnBean.setPageCollection((Map) session.get(FwConstants.BEFORE_COLLECTION));

					}

			}
			APP_IN_DABL_Collection deleteAppInDablColl = new APP_IN_DABL_Collection();
			APP_IN_DABL_Cargo deleteAppInDablCargo = new APP_IN_DABL_Cargo();
			int appInDablCollSize = appInDablColl.size();
			for(int i=0;i < appInDablCollSize; i++)
			{

			}

			//make the page not required if, no other death row exists

            }

//				*********************
				//Not going to remove from RMC_RQST CARgo as of now
				//**********************

				//	RMC_RQST_Cargo rmcCargo = rmcRqstManager.loadRMCRequest(appNum);

	public void eraseAddPregnancyDetails(FwTransaction txnBean) throws javax.ejb.EJBException {

		try {

			Map session = txnBean.getSession();
			Map request = txnBean.getRequest();
			Map pageCollection = txnBean.getPageCollection();
			
//			Get beforeMap
			Map beforeMap =(Map) session.get(FwConstants.BEFORE_COLLECTION);
			
			//Get driver info.
			int[] driverArray = (int[]) session.get(FwConstants.RMC_DRIVER);

			//Get appNumber from the session
			String appNum = (String)session.get(AppConstants.APP_NUMBER);
			
			//Get the pageCollection INDV_LIVE_CHG_Collection
			APP_IN_PREG_Collection appInPregColl = (APP_IN_PREG_Collection) pageCollection.get("APP_IN_PREG_Collection");
			ARHCSValidator validator = new ARHCSValidator();
			String indvSeqNum = appInPregColl.getCargo(0).getIndv_seq_num();
//			Setting IndivSeqTypeBean
			IndivTypeSeqBean indivTypeSeqBean = new IndivTypeSeqBean();
			indivTypeSeqBean.setIndivSeqNum(indvSeqNum);
			indivTypeSeqBean.setType(AppConstants.BECAME_PREGNANT);
			
			PeopleHandler peopleHandler = (PeopleHandler) session.get(AppConstants.INDIV_MASTER_HANDLER);
			String name=peopleHandler.getFirstName(indvSeqNum);
			boolean warningMsgFlag = false;
			validator.validateErase(AppConstants.BECAME_PREGNANT,name);
			if(validator.hasMessages()){
				String reqWarningMsgs =(String)request.get(FwConstants.WARNING_MSG_DETAILS);
				warningMsgFlag = this.checkForWarningMesgs(reqWarningMsgs,validator.getMessageList());
				if(!warningMsgFlag){
					request.put(FwConstants.MESSAGE_LIST,validator.getMessageList()); 
					txnBean.setPageCollection((Map) session.get(FwConstants.BEFORE_COLLECTION));
					beforeMap.put(AppConstants.INDV_TYPE_SEQ_BEAN,indivTypeSeqBean);
					return;
				}
				else
				{
					IndivTypeSeqBean postIndivSeqBean = (IndivTypeSeqBean)beforeMap.get(AppConstants.INDV_TYPE_SEQ_BEAN);
					boolean sameIndivSeqNum = (postIndivSeqBean.getIndivSeqNum().equals(indivTypeSeqBean.getIndivSeqNum()))?true:false;
					boolean sameType = (postIndivSeqBean.getType().equals(indivTypeSeqBean.getType()))?true:false;
					if((sameIndivSeqNum == false) || (sameType == false))
					{
						request.put(FwConstants.MESSAGE_LIST,validator.getMessageList());						
						session.put(FwConstants.BEFORE_COLLECTION,beforeMap);					
						txnBean.setPageCollection((Map) session.get(FwConstants.BEFORE_COLLECTION));

						return;
					}	
				}
			}

			{

				deleteAppInPregCargo.setApp_num(appNum);
				deleteAppInPregCargo.setRowAction(FwConstants.ROWACTION_DELETE);
				deleteAppInPregCargo.setSrc_app_ind(AppConstants.RMC_NEW_RECORD_IND);
				deleteAppInPregColl.addCargo(deleteAppInPregCargo);
