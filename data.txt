WITH DATES AS (
    SELECT
        SYSTEM_DATE,
        MTH_BEG_DT,
        MTH_END_DT
    FROM W100DP1.TWNSYSDATE
    FETCH FIRST 1 ROW ONLY
),
OFFICE AS (
    SELECT
        OFC_CD,
        OFC_NM
    FROM W100DP1.TWNOFFICE
),
UPI AS (
    SELECT DISTINCT
        FC.APPL_UPI,
        CL.OFC_CD
    FROM W100DP1.TWNFAM_CASE_HSTRY FC
    JOIN W100DP1.TWNFAM_CASELD_CASE CL
      ON CL.ICI = FC.ICI
    JOIN DATES D
      ON 1 = 1
    WHERE CL.PRGM_OFC_TYPE = 'A'
      AND CL.OFC_CD NOT IN ('DC', 'DR', 'DL', 'DH')
      AND FC.PRD_BEG_DT <= D.MTH_BEG_DT
      AND FC.PRD_END_DT >= D.MTH_END_DT
      AND FC.CASE_STS IN ('O', 'P')
),
FAM_CASE AS (
    SELECT DISTINCT
        FC.APPL_UPI,
        FC.ICI
    FROM W100DP1.TWNFAM_CASE_HSTRY FC
    JOIN DATES D
      ON 1 = 1
    WHERE FC.PRD_BEG_DT <= D.MTH_BEG_DT
      AND FC.PRD_END_DT >= D.MTH_END_DT
),

      ON AC.PRGM = 'AF'

    JOIN DATES D ON 1 = 1

      ON AC.PRGM = 'SG'
     AND AC.AID_CD = CM.SG_AID_CD
    JOIN DATES D ON 1 = 1
    WHERE CM.PRD_BEG_DT <= D.MTH_BEG_DT
      AND CM.PRD_END_DT >= D.MTH_END_DT
      AND CM.SG_ELIG_CD NOT IN ('3','9')

    SELECT AC.AID_CD_TYPE, AC.AID_CD, 'TC' AS PRGM, CM.ICI

    UNION ALL
    SELECT AC.AID_CD_TYPE, AC.AID_CD, 'TL' AS PRGM, CM.ICI

    JOIN W100DP1.TWNCASE_MEMBER_HST CM
      ON AC.PRGM = 'TL'
     AND AC.AID_CD = CM.TL_AID_CD
    JOIN DATES D ON 1 = 1
    WHERE CM.PRD_BEG_DT <= D.MTH_BEG_DT
      AND CM.PRD_END_DT >= D.MTH_END_DT
      AND CM.TL_ELIG_CD NOT IN ('3','9')

    UNION ALL
    SELECT AC.AID_CD_TYPE, AC.AID_CD, 'TN' AS PRGM, CM.ICI
    FROM W100DP1.TWNAID_CODE_CATG AC

      ON AC.PRGM = 'TN'
     AND AC.AID_CD = CM.TN_AID_CD
    JOIN DATES D ON 1 = 1
    WHERE CM.PRD_BEG_DT <= D.MTH_BEG_DT
      AND CM.PRD_END_DT >= D.MTH_END_DT

    UNION ALL
    SELECT AC.AID_CD_TYPE, AC.AID_CD, 'TP' AS PRGM, CM.ICI

PRGM_CASE AS (
    SELECT
        PC.PRGM_CASE_STS,
        PC.RCRT_APL_DT,
        PC.ICI,
        PC.PRGM_CASE_TYPE,
        PC.AID_CD,
        PC.ELIG_CD
    FROM W100DP1.TWNPRGM_CASE_HSTRY PC
    JOIN DATES D
      ON 1 = 1
    WHERE PC.PRD_BEG_DT <= D.MTH_BEG_DT
      AND PC.PRD_END_DT >= D.MTH_END_DT
      AND PC.PRGM_CASE_STS IN ('P', 'O', 'D')
),
PRGM (AID_CD_TYPE, AID_CD, APPL_UPI, RCRT_APL_DT, PRGM_CASE_STS, IN_MONTH) AS (
    SELECT DISTINCT
        CM.AID_CD_TYPE,
        CM.AID_CD,
        FC.APPL_UPI,
        PC.RCRT_APL_DT,
        PC.PRGM_CASE_STS,
        CASE
            WHEN PC.RCRT_APL_DT BETWEEN D.MTH_BEG_DT AND D.MTH_END_DT THEN 1
            ELSE 0
        END AS IN_MONTH
    FROM CASE_MEMBER CM
    JOIN FAM_CASE FC
      ON FC.ICI = CM.ICI
    JOIN PRGM_CASE PC
      ON PC.ICI = CM.ICI
     AND PC.PRGM_CASE_TYPE = CM.PRGM
    JOIN DATES D
      ON 1 = 1
    WHERE PC.PRGM_CASE_STS IN ('P', 'O')
      AND (
            CM.PRGM <> 'CC'
         OR (CM.PRGM = 'CC' AND CM.AID_CD = PC.AID_CD AND PC.ELIG_CD NOT IN ('3','9'))
      )

    UNION ALL

    SELECT
        MIN(CM.AID_CD_TYPE) AS AID_CD_TYPE,
        MAX(CM.AID_CD)      AS AID_CD,
        FC.APPL_UPI,
        MAX(PC.RCRT_APL_DT) AS RCRT_APL_DT,
        MAX(PC.PRGM_CASE_STS) AS PRGM_CASE_STS,
        CASE
            WHEN MAX(PC.RCRT_APL_DT) BETWEEN D.MTH_BEG_DT AND D.MTH_END_DT THEN 1
            ELSE 0
        END AS IN_MONTH
    FROM CASE_MEMBER CM
    JOIN FAM_CASE FC
      ON FC.ICI = CM.ICI
    JOIN PRGM_CASE PC
      ON PC.ICI = CM.ICI
     AND PC.PRGM_CASE_TYPE = CM.PRGM
    JOIN DATES D
      ON 1 = 1
    WHERE PC.PRGM_CASE_STS = 'D'
      AND CM.PRGM <> 'MC'
      AND (
            CM.PRGM <> 'CC'
         OR (CM.PRGM = 'CC' AND CM.AID_CD = PC.AID_CD AND PC.ELIG_CD NOT IN ('3','9'))
      )
    GROUP BY FC.APPL_UPI, D.MTH_BEG_DT, D.MTH_END_DT
),
