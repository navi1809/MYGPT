public void calculateGagaIncome(H2Person person) {
    // Reset state
    VdbControl.reset();
    GagaIncomeRecord currentRecord = new GagaIncomeRecord();
    currentRecord.setUpi(person.getUpi());

    // Fetch initial data
    List<GagaIncomeRecord> gagaRecords = gagaIncomeService.fetchGagaIncomeByUpi(currentRecord.getUpi());

    for (GagaIncomeRecord record : gagaRecords) {
        double monthlyAmount = convertToMonthly(record.getIncomeAmount(), record.getFrequency());

        if (monthlyAmount > 0) {
            ame48W1.setGagaIncomeFlag("Y");

            ame48W1.incrementGagaCount();
            ame48W.incrementGagaCount();

            ame48W1.addToGagaTotal(monthlyAmount);
            ame48W.addToGagaTotal(monthlyAmount);
        }
    }
}
