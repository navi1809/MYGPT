WITH DATES AS (
SELECT 
    SYSTEM_DATE,
    MTH_BEG_DT,
    MTH_END_DT,
    QTR_BEG_DT,
  QTR_END_DT,
    FEDRFIS_BEG_DT,
    FEDRFIS_END_DT
FROM {?SCHEMA}.TWNSYSDATE
),
OFFICE AS (
SELECT 
 OFC_CD,
 OFC_NM
FROM 
 {?SCHEMA}.TWNOFFICE T1 JOIN 
 {?SCHEMA}.TWNCODE_TBL T2 ON 
  T1.OFC_CD=T2.CODE_TBL_CD AND
  T2.CODE_TBL_TYP='DISTCOD'
WHERE OFC_CD NOT IN ('TP')
ORDER BY OFC_CD
),
MTHLY AS (
SELECT DISTINCT
 T2.AFDC_DUPL_ASST_IND AS DUPL_ASST_FG,
 T2.ICI,
 T2.PRD_BEG_DT,
 T2.PRD_END_DT
FROM 
 {?SCHEMA}.TWNCASE_MEMBER_HST T2
WHERE 
 T2.AFDC_DUPL_ASST_IND IN ('C','D','E','S') AND 
 T2.FS_ELIG_CD = '1'
),
DATA AS (
SELECT 
 ICI,
 OFC_CD
FROM 
 {?SCHEMA}.TWNFAM_CASELD_CASE T1 JOIN 
 {?SCHEMA}.TWNCASE_MEMBER_HST T2 ON T1.ICI=T2.ICI
WHERE 
 T2.AFDC_DUPL_ASST_IND IN ('C','D','E','S') AND 
 T2.FS_ELIG_CD = '1'
)
-- MTHLY_TOT AS (
SELECT 
 SUM(SUM(CASE WHEN DUPL_ASST_FG='C' THEN 1 ELSE 0 END)) OVER(PARTITION BY OD.OFC_CD) AS DUPL_ASST_MTH_C,
 SUM(SUM(CASE WHEN DUPL_ASST_FG='D' THEN 1 ELSE 0 END)) OVER(PARTITION BY OD.OFC_CD) AS DUPL_ASST_MTH_D,
 SUM(SUM(CASE WHEN DUPL_ASST_FG='E' THEN 1 ELSE 0 END)) OVER(PARTITION BY OD.OFC_CD) AS DUPL_ASST_MTH_E,
 SUM(SUM(CASE WHEN DUPL_ASST_FG='S' THEN 1 ELSE 0 END)) OVER(PARTITION BY OD.OFC_CD) AS DUPL_ASST_MTH_S,
 OD.OFC_CD,
 OFC_NM
FROM 
 MTHLY M JOIN 
 DATA T1 ON 
  T1.ICI=M.ICI JOIN 
 OFFICE OD ON 
  OD.OFC_CD=T1.OFC_CD JOIN 
 DATES CD ON 
  M.PRD_BEG_DT <= MTH_BEG_DT AND 
  M.PRD_END_DT >= MTH_END_DT
GROUP BY OD.OFC_CD,OFC_NM
FOR FETCH ONLY WITH UR;

	-- OFFICE OD ON 
		-- T1.OFC_CD = OD.OFC_CD JOIN 
	DATES CD ON

		T2.PRD_END_DT >= MTH_END_DT AND 
		T3.PRD_BEG_DT <= MTH_BEG_DT AND

	T3.ABAWD_FG,
	COUNT(T2.UPI) AS WS_COUNT,
	PROCESS_CCYY AS RPT_YR,
    PROCESS_MM AS RPT_MTH
FROM 
	-- {?SCHEMA}.TWNFAM_CASELD_CASE T1 JOIN 
	{?SCHEMA}.TWNCASE_MEMBER_HST T2 JOIN

	-- T1.OFC_CD,

	T3.ABAWD_FG,

	T2.FS_ELIG_CD = '1'
GROUP BY 
	-- T1.OFC_CD, OD.OFC_NM, 
	T2.ICI, T3.ABAWD_FG

MTHLY_ABWD_TOT AS (
SELECT 
	SUM(SUM(CASE WHEN ABAWD_FG='R' THEN WS_COUNT ELSE 0 END)) OVER(PARTITION BY OD.OFC_CD) AS ABAWD_MTH_R,
	SUM(SUM(CASE WHEN ABAWD_FG='W' THEN WS_COUNT ELSE 0 END)) OVER(PARTITION BY OD.OFC_CD) AS ABAWD_MTH_W,
	SUM(SUM(CASE WHEN ABAWD_FG='Y' THEN WS_COUNT ELSE 0 END)) OVER(PARTITION BY OD.OFC_CD) AS ABAWD_MTH_Y,
	SUM(SUM(CASE WHEN ABAWD_FG='Z' THEN WS_COUNT ELSE 0 END)) OVER(PARTITION BY OD.OFC_CD) AS ABAWD_MTH_Z,
	OD.OFC_CD,
	OFC_NM
FROM 
	MTHLY_ABWD MA JOIN 
	{?SCHEMA}.TWNFAM_CASELD_CASE T1 ON 
		T1.ICI= MA.ICI JOIN
	OFFICE OD ON 
		T1.OFC_CD=OD.OFC_CD
GROUP BY OD.OFC_CD,OFC_NM
),
QTR_ABWD_TOT AS (
SELECT 
	SUM(SUM(CASE WHEN ABAWD_FG='R' THEN WS_COUNT ELSE 0 END)) OVER(PARTITION BY OD.OFC_CD) AS ABAWD_QTR_R,
	SUM(SUM(CASE WHEN ABAWD_FG='W' THEN WS_COUNT ELSE 0 END)) OVER(PARTITION BY OD.OFC_CD) AS ABAWD_QTR_W,
	SUM(SUM(CASE WHEN ABAWD_FG='Y' THEN WS_COUNT ELSE 0 END)) OVER(PARTITION BY OD.OFC_CD) AS ABAWD_QTR_Y,

	OFC_NM

	OFFICE OD ON

SELECT 
	'ST' AS OFC_CD,
	'NEVADA STATEWIDE' AS OFC_NM,
	SUM(ABAWD_MTH_R) AS ABAWD_MTH_R,
	SUM(ABAWD_MTH_W) AS ABAWD_MTH_W,
	SUM(ABAWD_MTH_Y) AS ABAWD_MTH_Y,
	SUM(ABAWD_MTH_Z) AS ABAWD_MTH_Z,
	SUM(ABAWD_QTR_R) AS ABAWD_QTR_R,
	SUM(ABAWD_QTR_W) AS ABAWD_QTR_W,
	SUM(ABAWD_QTR_Y) AS ABAWD_QTR_Y,
	SUM(ABAWD_QTR_Z) AS ABAWD_QTR_Z,
	SUM(ABAWD_FSC_R) AS ABAWD_FSC_R,
	SUM(ABAWD_FSC_W) AS ABAWD_FSC_W,
	SUM(ABAWD_FSC_Y) AS ABAWD_FSC_Y,
	SUM(ABAWD_FSC_Z) AS ABAWD_FSC_Z,
	0 AS SORT
FROM 
	MTHLY_ABWD_TOT M FULL OUTER JOIN
	QTR_ABWD_TOT Q ON M.OFC_CD=Q.OFC_CD AND M.OFC_NM=Q.OFC_NM FULL OUTER JOIN
	FISCAL_ABWD_TOT F ON M.OFC_CD=F.OFC_CD AND M.OFC_NM=F.OFC_NM

	COALESCE(M.OFC_CD,Q.OFC_CD,F.OFC_CD) AS OFC_CD,
	COALESCE(M.OFC_NM,Q.OFC_NM,F.OFC_NM) AS OFC_NM,

	SUM(ABAWD_MTH_W) AS ABAWD_MTH_W,
	SUM(ABAWD_MTH_Y) AS ABAWD_MTH_Y,
	SUM(ABAWD_MTH_Z) AS ABAWD_MTH_Z,
	SUM(ABAWD_QTR_R) AS ABAWD_QTR_R,
	SUM(ABAWD_QTR_W) AS ABAWD_QTR_W,
