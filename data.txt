function displaySelectedInterfaceDetails(selectedUpi, selectedScreen) {
	var jsonVar = displayConstructMap[selectedScreen];
	var params = "selectedUpi=" + selectedUpi;
	var isdDate = $("#isdSystemDate").val();
	var ibdxDate = $("#ibdxSystemDate").val();
	var options = {
		type : 'POST',
		url : jsonVar.endpoint,
		data : params,
		success : function(htmlText, status, xhr) {
			$("#selectedInterfaceDetails_ISD").empty();
			updateScreenForJSONResponse(htmlText, false, xhr, jsonVar.divID,
					'', false);
			$("#selectedValue").val("");
			$("#interfaceSection").val("");
			if('IBDXSELECTED' === selectedScreen){
				//refreshIBDXSummary(selectedUpi, 'viewBendexHistory');
				var year = $('#processYearBNDX').val();
				var selectedRadio = $('input:radio[name="selectedRecord"][value="'
						+ selectedUpi + '"]');
				if (selectedRadio) {
					selectedRadio.attr('checked', true);
				}
				$('#processYearBNDX').val(year);
				document.getElementById("ibdxLink"+selectedUpi).innerHTML = ibdxDate;
				
				refreshIBDXCommonInfo(selectedUpi, selectedScreen);
			} else if(selectedScreen.indexOf("ISD") >= 0) {
				//refreshISDSummary(selectedUpi, selectedScreen);
				var year = $('#processYearISD').val();
				$('#processYearISD').val(year);
				if(selectedScreen.indexOf("ISD1") >= 0){
					document.getElementById("isd1Link"+selectedUpi).innerHTML = isdDate;
				}
				if(selectedScreen.indexOf("ISD2") >= 0){
					document.getElementById("isd2Link"+selectedUpi).innerHTML = isdDate;
				}
				if(selectedScreen.indexOf("ISD3") >= 0){
					document.getElementById("isd3Link"+selectedUpi).innerHTML = isdDate;
				}
				if(selectedScreen.indexOf("ISD4") >= 0){
					document.getElementById("isd4Link"+selectedUpi).innerHTML = isdDate;
				}
				if(selectedScreen.indexOf("ISD5") >= 0){
					document.getElementById("isd5Link"+selectedUpi).innerHTML = isdDate;
				}
				refreshISDCommonInfo(selectedUpi, selectedScreen);
			}
			
			$.unblockUI();
		},
		error : function(errordata) {
			$.unblockUI();
			console.log(errordata);
		}
	};
	console.log(options);
	showWait();
	$.ajax(options);
}

function checkForSubmitButtonEnabling(elem) {
	var progChanged = "";
	if (elem.id == 'snap') {
		updateFlag(elem.value, 'snapFalg', $("#oldSnapValue").val());
		progChanged = progChanged+"##snap";
	}
	if (elem.id == 'maabd') {
		updateFlag(elem.value, 'maabdFalg', $("#oldMaabdValue").val());
		progChanged = progChanged+"##maabd";
	}
	if (elem.id == 'fmc') {
		updateFlag(elem.value, 'fmcFalg', $("#oldFmcValue").val());
		progChanged = progChanged+"##fmc";
	}
	if (elem.id == 'tchd') {
		updateFlag(elem.value, 'tchdFalg', $("#oldTchdValue").val());
		progChanged = progChanged+"##tchd";
	}
	if (elem.id == 'temp') {
		updateFlag(elem.value, 'tempFalg', $("#oldTempValue").val());
		progChanged = progChanged+"##temp";
	}
	if (elem.id == 'neon') {
		updateFlag(elem.value, 'neonFalg', $("#oldNeonValue").val());
		progChanged = progChanged+"##neon";
	}
	if (elem.id == 'loan') {
		updateFlag(elem.value, 'loanFalg', $("#oldLoanValue").val());
		progChanged = progChanged+"##loan";
	}
	if (elem.id == 'ssg') {
		updateFlag(elem.value, 'ssgFalg', $("#oldSsgValue").val());
		progChanged = progChanged+"##ssg";
	}
	if (elem.id == 'ao') {
		updateFlag(elem.value, 'aoFalg', $("#oldAoValue").val());
		progChanged = progChanged+"##ao";
	}
	if (elem.id == 'mcb') {
		updateFlag(elem.value, 'mcbFalg', $("#oldMcbValue").val());
		progChanged = progChanged+"##mcb";
	}
	if (elem.id == 'ccpa') {
		updateFlag(elem.value, 'ccpaFalg', $("#oldCcpaValue").val());
		progChanged = progChanged+"##ccpa";
	}
	if (elem.id == 'dcfs') {
		updateFlag(elem.value, 'dcfsFalg', $("#oldDcfsValue").val());
		progChanged = progChanged+"##dcfs";
	}
	if (isFlag('snapFalg') || isFlag('maabdFalg') || isFlag('fmcFalg')
			|| isFlag('tchdFalg') || isFlag('tempFalg') || isFlag('neonFalg')
			|| isFlag('loanFalg') || isFlag('ssgFalg') || isFlag('aoFalg')
			|| isFlag('mcbFalg') || isFlag('ccpaFalg') || isFlag('dcfsFalg')) {
		$('#statusUpdateButton').show();
	} else {
		$('#statusUpdateButton').hide();
	}
	$("#progChanged").val(progChanged);
}

function isFlag(flag) {
	if ($('#' + flag).val() == 'true') {
		return true;
	} else {
		return false;
	}
}

function updateFlag(changedValue, flag, existingValue) {
	if (changedValue != existingValue) {
		$("#" + flag).val(true);
	} else {
		$("#" + flag).val(false);
	}
}

function submitProgramChangeStatus() {
	showWait();
	var progChanged = $("#progChanged").val();
	var jsonVar = editorConstructMap['updateProgramStatus'];
	$.ajax({
		type : "POST",
		data : $("#caseStatusSummary").serializeArray(),
		url : jsonVar.endpoint,
		success : function(htmlText, status, xhr) {
			updateScreenForJSONResponse(htmlText, false, xhr, jsonVar.divID,
					'', false);
			var progs = progChanged.split("##");
			for(i=0;i<progs.length;i++){
				$("#" + progs[i] + "Falg").val(false);
				var progVal = $("#" + progs[i]).val();
				var progOldVal = $("old" + progs[i] + "Value");
				progOldVal.val(progVal);
				$('#statusUpdateButton').hide();
			}
			$.unblockUI();
		}
	});
}

function getMessageTemplate(selectedType,targetElementId1,targetElementId2) {
	if (!checkAllRequiredFields()) {
		return true;
	}
	targetElementId1 = '#' + targetElementId1;
	targetElementId2 = '#' + targetElementId2;
	var appointmentUrl = getPathosUrl("getMessageTemplate");
	showWait();
	var sendData = {};
	sendData['messageTypeCode'] = $('#messageTypeCode').val();
	sendData['messageCategory'] = selectedType;
	$.ajax({
		type : "POST",
		url : appointmentUrl,
		data : sendData,
		success : function(data) {
			console.log(data);
			$(targetElementId1).empty();
			$(targetElementId2).empty();
			$(data).each(function() {
				$(targetElementId1).val(this.key);
				$(targetElementId2).val(this.value);
			});
			$.unblockUI();
		},
		error : function(xhr, ajaxOptions, thrownError) {
			console.log(xhr.status + thrownError);
			$.unblockUI();
		}
	});
}

function saveCourtOrderRecord(button, selectedType, actionUpdate) {
	$("#messageCaseClosed").hide();
	$("#messageseciondiv").hide();

	var jsonVar = editorConstructMap[selectedType];

	$('input[name="selectedRecord"]').each(function(checked) {
		if ($(this).attr('checked') == "checked") {
			isSorSelected = true;
		}
	});
	$('input[name="selectedStagingRecord"]').each(function(checked) {
		if ($(this).attr('checked') == "checked") {
			isStgSelected = true;
		}
	});

	if (!(isSorSelected && isStgSelected) && !bapHome.isStgAddAction()) {
		if (!checkIfDataChangedForCord()) {
			showSimpleDialog("None of the screen details were modified.");
			return true;
		}
	}

	if (jsonVar.preValidateCallback && jsonVar.preValidateCallback()) {
		return true;
	}

	if (!jsonVar.skipMandatoryFieldCheck && !checkAllRequiredFieldsForCORD()) {
		return true;
	}

	if (jsonVar.validateEditorCallback && jsonVar.validateEditorCallback()) {
		return true;
	}

	var isSorSelected = false;
	var isStgSelected = false;

	var URL = '';
	if (actionUpdate == 'true') {
		console.log("Updating ..." + jsonVar.updateEndpoint);
		URL = jsonVar.updateEndpoint;
	} else {
		console.log("Inserting ..." + jsonVar.insertEndpoint);
		URL = jsonVar.insertEndpoint;
	}

	showWait();

	var saveRecordPromise = $.ajax({
		type : "POST",
		url : URL,
		data : $(".cordBapHome").serializeArray()
	});

	saveRecordPromise
			.done(
					function(data, status, xhr) {
						console.dir(data);
						if (data.response.status) {
							$(document.body).data('clean_form_data_cord', '');
							$("#showHideButton_spousalResourceAssessment").trigger("click");
							if (data.response.data
									&& data.response.data.length >= 1
									&& data.response.data[0].defaultSorRecord !== '') {
								var valueToSelected = getValueForSorRecordObj(
										data.response.data, "defaultSorRecord");
								$("#cordSelectedRecord").val(valueToSelected);
								$("#showHideButton_spousalResourceAssessment").trigger("click");
							}
						}
					}).always(function(data, status, xhr) {
				checkMessageContainer(data,jsonVar.messageDivId);
				updWarningFlag(data.response.data);
			});
}

function checkMessageContainer(data, message) {
	  if($('#bapSummaryTable').is(':visible')){ 
	   displayResults(data, message, '', false);
	  } else {
	    setTimeout( function() { checkMessageContainer(data, message)}, 50); //wait 50 ms, then try again
	  }
	}

var checkIfDataChangedForCord = function() {
	var clean_form_data_cord = $(document.body).data('clean_form_data_cord');

	if (typeof clean_form_data_cord === 'undefined' || clean_form_data_cord == '') {
		$(document.body).data('courtorder_changed', 'N');
		return false;
	}

	var dirty_form_data_cord = $(".cordBapHome").find(
			"select,input[type!=hidden],textarea").serialize();

	if (dirty_form_data_cord != clean_form_data_cord) {
		return true;
	} else {
		$(document.body).data('courtorder_changed', 'N');
		return false;
	}
}

function openSaveResponseDetails(indvUpi) {
	showWait();
	var jsonVar = displayConstructMap['SAVERESPONSE'];
	var sendData = {};
	sendData['selectedRowUpi'] = indvUpi;
	var options = {
		type : "POST",
		data : sendData,
		url : jsonVar.endpoint,
		success : function(result, status, xhr) {
			var ct = xhr.getResponseHeader("content-type") || "";
			if (ct.indexOf('text/html') > -1) {
				$("#saveResponseDetailsDiv").html(result);
				$("#saveResponseDetailsDiv").dialog("open");
			} else {
				updateScreenForJSONResponse(result, false, xhr, 'errorTrace',
						'', false);
			}
			$.unblockUI();
		},
		error : function(xhr, ajaxOptions, thrownError) {
			$("#saveResponseDetailsDiv").html(xhr.status + thrownError);
			$("#saveResponseDetailsDiv").dialog("open");
			$.unblockUI();
		}
	};
	$.ajax(options);
	return false;
}

	var upi = $('#toVerify').val();

	var params = "upi=" + upi;
	$.ajax({
		type : "POST",
		data : params,
		url : jsonVar.endpoint,
		success : function(result, status, xhr) {
			var messageList = result.response.messages;
			for (var i = 0; i < messageList.length; i++) {
				if (isincarCheckBox && messageList[i].indexOf("P") >= 0) {

					$('#cancelIncarceration').hide();
					$("#cancelForPris").val(false);
				}
				if (isqocCheckBox && messageList[i].indexOf("Q") >= 0) {
					$('#cancelQtrs').show();

				} else {
					$('#cancelQtrs').hide();
					$("#cancelForQoc").val(false);
				}
			}
			console.log(result);
			$.unblockUI();
		},
		error : function(result, xhr, thrownError) {
			console.log(xhr.status + thrownError);
			updateScreenForJSONResponse(result, false, xhr, 'errorTrace', '',
					false);
			$.unblockUI();
		}
	});

}

function displayClientCommunication(selectedType) {
	if ($("#messageCategory").val()==undefined && !checkAllRequiredFields()) {
		return true;
	}
	if ($(selectedType).val() == "") {
		$('.details').hide();
		$('#selectActionClose').show();
		return;
	}
	showWait();
	var message = false;
	var successMessage = '';
	 var recepient = $('#selectRecipient').children("option:selected").val();
	if ($(selectedType).val()=="SENDMSG"){
		var appointmentUrl = getPathosUrl("sendMessage");
		$('#selectRecipientLabel').hide();
		$('#selectRecipient').hide();
		$('#accountNameLabel').hide();
		$('#acountId').hide();
	} else {
		var appointmentUrl = getPathosUrl("changePreferences");
		$('#selectRecipientLabel').hide();
		$('#selectRecipient').hide();
		$('#accountNameLabel').show();
		$('#acountId').show();
		$('#acountId').html(recepient.split("##")[1]);
	}
	if(arguments[1] && arguments[1] == "showMessage"){
		message = true;
		successMessage = arguments[2];
	}
	var options = {
		type : 'POST',
		url : appointmentUrl,
		data : 'selectedMember=' + recepient.split("##")[0],
		success : function(htmlText, status, xhr) {
			updateScreenForJSONResponse(htmlText, false, xhr, "communicationDiv",
					'', false);
			$('#communicationDiv').show();
			$('#selectActionClose').hide();
			highlightMandatoryFieldsOnLoad();
			if(message){
				displayCommSettingsSuccessMessage(successMessage);
			}
		},
		error : function(errordata) {
			$.unblockUI();
			console.log(errordata);
		}
	};
	console.log(options);
	showWait();
	$.ajax(options);
}

function displaySelectedRecordForCord(selectedRecord, screentypecode) {
	var existingRecordValue = document.getElementById("table_radio_"
			+ bapHome.getExistingRecordValue());

	if ((bapHome.isDifferentRecord(selectedRecord.value) || (bapHome
			.isAddAction()))
			&& checkIfDataChangedForCord()) {
		var dialog = $(
				'<p>Are you sure you want to continue? Any unsaved information will be lost.</p>')
				.dialog(
						{
							draggable : false,
							resizable : false,
							position : 'center',
							dialogClass : "hideCloseIcon",
							modal : true,
							buttons : {
								"Ok" : function() {
									displaySor(selectedRecord, screentypecode);
									dialog.dialog('close');
								},
								"Cancel" : function() {
									dialog.dialog('close');
									if (!bapHome.isAddAction()) {
										var newRecordValue = document
												.getElementById("table_radio_"
														+ selectedRecord.value);
										existingRecordValue.checked = true;
										newRecordValue.checked = false;
										bapHome
												.setExistingRecordValue(existingRecordValue.value);
									} else {
										$('input[name="selectedRecord"]')
												.each(
														function(checked) {
															if ($(this).attr(
																	'checked') == "checked") {
																this.checked = false;
															}
														});
									}
								}
							}
						});
	} else {
		displaySor(selectedRecord, screentypecode);
	}
}

function updateCommentsAndReIssue(reasonCode){
	var actionType = $('#breqActionType').val();
	var bselType = $('#bselType').val();
	if((reasonCode=='Z' && actionType=='PR')){
		$('#resissueAmount').removeAttr("disabled");
		$('#comments').removeAttr("disabled");
	}else if((bselType=='F' && reasonCode=='Z' && actionType=='VR')){
		$('#resissueAmount').prop("disabled", true);
		$('#comments').removeAttr("disabled");
	}
}

function initiateManualRD(){
	$("#messagesectionForInitiation").hide();
	var isMaabdChecked = $('#maabdProgramTypeCode').is(':checked');
	var isFmcChecked = $('#fmcProgramTypeCode').is(':checked');
	var isTchdChecked = $('#tchdProgramTypeCode').is(':checked');
	var isNeonChecked = $('#neonProgramTypeCode').is(':checked');
	var isLoanChecked = $('#loanProgramTypeCode').is(':checked');
	
	if(isMaabdChecked){
		$("#isMaabdChecked").val('Y');
	}
	if(isFmcChecked){
		$("#isFmcChecked").val('Y');
	}
	if(isTchdChecked){
		$("#isTchdChecked").val('Y');
	}
	if(isNeonChecked){
		$("#isNeonChecked").val('Y');
	}
	if(isLoanChecked){
		$("#isLoanChecked").val('Y');
	}
	if(isFmcChecked  && isMaabdChecked){
		$("#isBothChecked").val('Y');
	}
	showWait();
	$.ajax({
		type : "POST",
		url : "initiateredetermination", 
		data : $(".initiateRDForm").serializeArray(),
		success : function(data, status, xhr) {
			updateScreenForJSONResponse(data, status, xhr, 'messagesectionForInitiation', '', false);
			if(data.response.status){
				if(isMaabdChecked){
					$('#maabdProgramTypeCode').attr("disabled",true);
					$('#maabdProgramTypeCode').attr("checked",false);
				} 
				if(isFmcChecked){
					$('#fmcProgramTypeCode').attr("disabled",true);
					$('#fmcProgramTypeCode').attr("checked",false);
				}
				if(isTchdChecked){
					$('#tchdProgramTypeCode').attr("disabled",true);
					$('#tchdProgramTypeCode').attr("checked",false);
				}
				if(isNeonChecked){
					$('#neonProgramTypeCode').attr("disabled",true);
					$('#neonProgramTypeCode').attr("checked",false);
				}
				if(isLoanChecked){
					$('#loanProgramTypeCode').attr("disabled",true);
					$('#loanProgramTypeCode').attr("checked",false);
				}
			}
			
			$.unblockUI();
		},
		error : function(xhr, ajaxOptions, thrownError) {
			var msg = ["Technical Error : "+xhr.status+'-'+ xhr.statusText];
			var errors = displayeMessage(msg,'red');
			$('#messagesectionForInitiation', findContainerForm(currentElement)).html(errors);
			$.unblockUI();
		}
	});
	return false;
}

function saveCsepRespPersonRecord(screenType) {
	var jsonVar = editorConstructMap[screenType];
	var data = '';
	var formData = $("#PARENTSUMMARYHOME").find("input[type='checkbox']").serializeArray().filter(function( t ) {
	    return t.name.includes("selectCheckbox");
	}); 
	if (formData.length == 0) {
		showSimpleDialog("No Responsible Person were selected.");
		return true;
	} else if (formData.length > 1) {
		showSimpleDialog("ONLY 1 NEW SELECTIONS ALLOWED");
		return true;
	} else {
		data = formData[0].value;
	}
	
	showWait();
	var saveRecordPromise = $.ajax({
		type : "POST",
		url : jsonVar.insertEndpoint,
		data : "selectedRecord="+data
	});
	
	saveRecordPromise
	.done(
			function(data, status, xhr) {
				console.dir(data);
				if (data.response.status) {
					$(document.body).data('clean_form_data', '');
					$("#current").trigger("click");
					$("#current").focus();
				}
			}).always(function(data, status, xhr) {
		console.log("in saveRecordPromise always function");	    
		displayResults(data, 'messageContainer', '', false);
		updWarningFlag(data.response.data);
		$.unblockUI();
	});
}


function updateCsepRespPersonRecord(screenType) {
	
	var formData = $("#CSEPRESPPERSONUPDATE").find("input[type='checkbox']").serializeArray().filter(function( t ) {
	    return t.name.includes("selectCsepRespUpdateCheckbox");
	}); 
	if (formData.length == 0) {
			showSimpleDialog("No Responsible Person were selected.");
			return true;
	}
	
	if ($("#csepRespPersonEffectivetStartDate").val().length == 0){
		showSimpleDialog("You must enter Effective Start Date");
		return true;
	}
	if (checkNoOfExistingCsepResp() == 0) {
		if (formData.length == 1) {
			showSimpleDialog("ADD CSEP RESP PERSON.");
			return true;
		}
    } else if (checkNoOfExistingCsepResp() == 1) {
		if (formData.length == 1 && formData[0].value.includes("T")) {
			showSimpleDialog("IF EFF DATES ENTERED, RESP PERSON MUST BE CHANGED.");
			return true;
		} else if (formData.length == 2) {
			showSimpleDialog("ADD 2ND CSEP RESP PERSON.");
			return true;
		} 
	} else if (checkNoOfExistingCsepResp() > 1 && checkNoOfExistingCsepResp() == formData.length) {
		showSimpleDialog("IF EFF DATES ENTERED, RESP PERSON MUST BE CHANGED.");
		return true;
	}
	

	
	var jsonVar = editorConstructMap[screenType];
	
	showWait();
	var saveRecordPromise = $.ajax({
		type : "POST",
		url : jsonVar.updateEndpoint,
		data : $("#CSEPRESPPERSONUPDATE").serializeArray()
	});
	
	saveRecordPromise
	.done(
			function(data, status, xhr) {
				console.dir(data);
				if (data.response.status) {
					$(document.body).data('clean_form_data', '');
					$("#current").trigger("click");
					$("#current").focus();
				}
			}).always(function(data, status, xhr) {
		console.log("in saveRecordPromise always function");	    
		displayResults(data, 'messageContainer', '', false);
		updWarningFlag(data.response.data);
		$.unblockUI();
	});
	
}

function checkNoOfExistingCsepResp(){
	var count = 0;
	var csepRespPersonsArray = $("#PARENTSUMMARYHOME").find("input[type='checkbox']");
	 if (csepRespPersonsArray.length > 0) {
		 for (var i=0;i < csepRespPersonsArray.length ; i++){
			 if (csepRespPersonsArray[i].name.includes("selectCheckbox") && csepRespPersonsArray[i].value.includes("T")) {
			 		count++;
			 }
		 }
	 }
	 return count;
}
