CREATE TABLE program_payment_details (
    id SERIAL PRIMARY KEY,
    report_month DATE,
    program_name VARCHAR(50),        -- e.g., TANF CHILD ONLY
    district_office VARCHAR(100),    -- e.g., BL BELROSE DISTRICT OFFICE
    record_type VARCHAR(50),         -- 'INDIVIDUAL', 'INADVERTENT ERROR', etc.
    debt_number VARCHAR(20),
    client_name VARCHAR(100),
    ssn VARCHAR(15),
    grant_reduction NUMERIC(10,2),
    direct_payment NUMERIC(10,2),
    treasury_offset NUMERIC(10,2),
    fs_ebt_payment NUMERIC(10,2),
    expungement NUMERIC(10,2),
    adjustment NUMERIC(10,2)
);
