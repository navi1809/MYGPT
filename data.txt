function taxDependentChange(el) {
    var value = el.options[el.selectedIndex].value;

    // 1. Find the TR that contains this <select>
    var row = el;
    while (row && row.tagName !== "TR") {
        row = row.parentNode;
    }
    if (!row) return;

    // 2. Find the *next* TR whose id starts with "taxRelationBlock"
    var relRow = row.nextSibling;
    while (relRow && (relRow.nodeType !== 1 || relRow.tagName !== "TR" ||
                      !/^taxRelationBlock/.test(relRow.id))) {
        relRow = relRow.nextSibling;
    }
    if (!relRow) return;

    // 3. Show / hide that row
    if (value === "Y") {
        relRow.style.display = "table-row";   // or "block" if you prefer
    } else {
        relRow.style.display = "none";

        // 4. Reset the fields inside the tax row (no seq needed)
        var firstNameElement = relRow.querySelector("select[name^='firstName_']");
        var relationElement  = relRow.querySelector("input[name^='relation_']");

        if (firstNameElement) {
            firstNameElement.value = "SEL";   // default "select"
        }
        if (relationElement) {
            relationElement.value = "";       // clear relationship
        }
    }
}
