// GETTING THE PROCESS DATE
Function AIF21P9-FETCHBDTS()
  /* -------------------------------------------------------------*/
  /* SCAN & GETTING PROCESS DATE*/
  /* --------------------------------------------------------------*/
  VGVar.handleHardIOErrors = 1;
  try
    get next BDATEREC ;
  end

  if (BDATEREC is ioError || 
  BDATEREC is endOfFile)

    VDBCONTROL.UPROC-NM = "AIF21P9-FETCHBDTS";
    VDBCONTROL.UPROC-OBJ = "BDATEREC";
    VDBCONTROL.UPROC-OPT = "SCAN";
    VMESSAGE.UMSGINS[2] = "AIF21A";
    VDBCONTROL.UMSQLCODE = 0;
    AIF21W.WS-ERR-CD = sysVar.errorCode;
    VMESSAGE.UMSGINS[1] = AIF21W.WS-ERR-LAST-FOUR;
    VDBCONTROL.UEZESQRRM = "INPUT PARAMETER FILE - I/O ERROR";

  AIF21W.WS-BDATE = BDATEREC.BDATE-PROCESS-DATE;
  AIF21W.WS-DATE-CCYY = AIF21W.WS-BDATE-CCYY;
  AIF21W.WS-DATE-MM = AIF21W.WS-BDATE-MM;
  AIF21W.WS-DATE-DD = AIF21W.WS-BDATE-DD;
  AIF21W.WS-DATE-DASH = "-";

// MAIN PROCESS
Function AIF21P9-MAIN()
  /* * PERFORM PROCESS TO INITIALIZE ALL ADDITIONAL RECORDS ***/
  AIF21P9-SET-EMPTY();

  /* * SCAN BDATE ***/
  AIF21P9-FETCHBDTS();

  /* * SCAN THE FIRST RECORD FROM FLAT FILE ***/
  AIF21P9-SCAN-INFLE();

  while (SR-AIF21-RECIN not endOfFile)

    if (SR-AIF21-RECIN.INPUT-SSN != AIF21W.WS-PREV-SSN)

      if (SR-AIF21-RECIN.INPUT-SSN not blanks)

        /* * CHECK TO SEE IF SSN EXIST*/
        /* * IN TWNPERSON.*/
        AIF21Q1-INQ-PERSON();

        if (sysVar.sqlData.sqlcode == 0)
            /* *********/
            /* * UPI ***/
          AIF21W.WS-UPI-NUM = H2-PERSON.UPI;
          AIF21W.WS-UPI-BIN = AIF21W.WS-UPI-NUM;
          H2-WTPY-40QTRS.UPI = AIF21W.WS-UPI-BIN;

            /* * POPULATE H2-WTPY-TYPE4 ***/
          AIF21P9-POP-H240QT();

            /* * UPDATE TWNWTPY_TYPE4 ***/
          AIF21Q2-UPDT-40QTR();

          if (sysVar.sqlData.sqlcode == 100)

              /* * INSERT TO THE TABLE TWNWTPY-TYPE4 ***/
            AIF21P9-INST-40QTR();

          end /* (IF EZESQCOD = 100;)*/

        end /* (IF EZESQCOD = 0;)*/


      end /* (IF SR-AIF21-RECIN.VERIFIED-SSN ^= '*/
                                   /* ';*/

    end /* (IF SR-AIF21-RECIN.VERIFIED-SSN ^=*/
                                   /* AIF21W.WS-PREV-SSN;)*/

    /* * MOVE WORKING STORAGE FOR THE FLAG,*/
    /* * MAKING SURE NOT GETTING THE SAME SSN;*/
    AIF21W.WS-PREV-SSN = SR-AIF21-RECIN.INPUT-SSN;

    /* * EMPTY OR INIT SR-AIF20 ***/
    set SR-AIF21-RECIN empty;

    /* * SCAN NEXT RECORD FROM FLAT FILE ***/
    AIF21P9-SCAN-INFLE();

  end /* *(WHILE SR-AIF21-RECIN NOT EOF)*/

  /* * PROCESS TO CLOSE THE APPLICATION*/
  AIF21P9-CLOS-PARA();
end // end AIF21P9-MAIN

    VMESSAGE.UMSGINS[1] = AIF21W.WS-ERR-LAST-FOUR;
  /* PERFORM WCM00P2-TO-CEP01A;*/
  end
end // end AIF21P9-SCAN-INFLE
