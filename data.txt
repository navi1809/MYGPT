// GETTING THE PROCESS DATE
Function AIF20P9-FETCHBDTS()
  /* -------------------------------------------------------------*/
  /* SCAN & GETTING PROCESS DATE*/
  /* --------------------------------------------------------------*/
  VGVar.handleHardIOErrors = 1;
  try
    get next BDATEREC ;
  end

  if (BDATEREC is ioError || 
  BDATEREC is endOfFile)

    VDBCONTROL.UPROC-NM = "AIF20P9-FETCHBDTS";
    VDBCONTROL.UPROC-OBJ = "BDATEREC";
    VDBCONTROL.UPROC-OPT = "SCAN";
    VMESSAGE.UMSGINS[2] = "AIF20A";
    VDBCONTROL.UMSQLCODE = 0;
    AIF20W.WS-ERR-CD = sysVar.errorCode;
    VMESSAGE.UMSGINS[1] = AIF20W.WS-ERR-LAST-FOUR;
    VDBCONTROL.UEZESQRRM = "INPUT PARAMETER FILE - I/O ERROR";
    WCM00P2-TO-CEP01A();

  end
  /* --------------------------------------------------------------*/
  AIF20W.WS-BDATE = BDATEREC.BDATE-PROCESS-DATE;
  AIF20W.WS-DATE-CCYY = AIF20W.WS-BDATE-CCYY;
  AIF20W.WS-DATE-MM = AIF20W.WS-BDATE-MM;
  AIF20W.WS-DATE-DD = AIF20W.WS-BDATE-DD;
  AIF20W.WS-DATE-DASH = "-";
  AIF20W.WS-DATE-DASH1 = "-";

// MAIN PROCESS
Function AIF20P9-MAIN()
  /* * PERFORM PROCESS TO INITIALIZE ALL ADDITIONAL RECORDS ***/
  AIF20P9-SET-EMPTY();

  /* * SCAN BDATE ***/
  AIF20P9-FETCHBDTS();

  /* * SCAN THE FIRST RECORD FROM FLAT FILE ***/
  AIF20P9-SCAN-INFLE();

  while (SR-AIF20-RECIN not endOfFile)

    if (SR-AIF20-RECIN.RECORD-TYPE != "1")

      /* * RECORD TYPE 2 POP H2-WTPY-TYPE4 ***/
      if (SR-AIF20-RECIN.RECORD-TYPE == "2")
        AIF20P9-POP-H2WTP2();
      end

      /* * RECORD TYPE 3 POP H2-WTPY-TYPE4 ***/
      if (SR-AIF20-RECIN.RECORD-TYPE == "3")
        AIF20P9-POP-H2WTP3();
      end

      /* * RECORD TYPE 4 POP H2-WTPY-TYPE4 ***/
      if (SR-AIF20-RECIN.RECORD-TYPE == "4")
        AIF20P9-POP-H2WTP4();
      end

      if (SR-AIF20-RECIN.INPUT-SSN != AIF20W.WS-PREV-SSN || 
      SR-AIF20-RECIN.CAN != AIF20W.WS-PREV-CAN)

        if (SR-AIF20-RECIN.INPUT-SSN not blanks)

          /* * CHECK TO SEE IF SSN EXIST*/
          /* * IN TWNPERSON.*/
          AIF20Q1-INQ-PERSON();

          if (sysVar.sqlData.sqlcode == 0)
            /* *********/
            /* * UPI ***/
            AIF20W.WS-UPI-NUM = H2-PERSON.UPI;
            AIF20W.WS-UPI-BIN = AIF20W.WS-UPI-NUM;
            H2-WTPY-TYPE4.UPI = AIF20W.WS-UPI-BIN;

            /* * POPULATE H2-WTPY-TYPE4 ***/
            /* PERFORM AIF20P9-POP-H2WTPY;*/

            /* * UPDATE TWNWTPY_TYPE4 ***/
            AIF20Q3-UPDAT-WTPY();

            if (sysVar.sqlData.sqlcode == 100)

              /* * INSERT TO THE TABLE TWNWTPY-TYPE4 ***/
              AIF20P9-INSRT-WTPY();

            end

          end

        else
          if (SR-AIF20-RECIN.CAN not blanks)
            /* * MOVE CLAIM ACCOUNT NUMBER(CAN) TO*/
            /* * WORKING STORAGE.*/
            AIF20W.WS-CAN-CHAR = SR-AIF20-RECIN.CAN;

            /* * CHECK TO SEE IF CLAIM ACCOUNT NUMBER*/
            /* * EXIST IN TWNMEDICARE_INFO.*/
            AIF20Q2-INQ-MEDICA();

            if (sysVar.sqlData.sqlcode == 0)
              /* *********/
              /* * UPI ***/
              AIF20W.WS-UPI-NUM = H2-MEDICARE-INFO.UPI;
              AIF20W.WS-UPI-BIN = AIF20W.WS-UPI-NUM;
              H2-WTPY-TYPE4.UPI = AIF20W.WS-UPI-BIN;

              /* * POPULATE H2-WTPY-TYPE4 ***/
              /* PERFORM AIF20P9-POP-H2WTPY;*/

              /* * UPDATE TWNWTPY_TYPE4 ***/
              AIF20Q3-UPDAT-WTPY();

              if (sysVar.sqlData.sqlcode == 100)

                /* * INSERT TO THE TABLE TWNWTPY-TYPE4 ***/
                AIF20P9-INSRT-WTPY();

              end /* (IF EZESQCOD = 100;)*/

            end /* (IF EZESQCOD = 0;)*/


          end /* (IF SR-AIF20-RECIN.CAN ^= '*/
                                   /* ';)*/

        end /* (IF SR-AIF20-RECIN.INPUT-SSN ^= '*/
                                   /* ';)*/
      end
    end /* (IF SR-AIF20-RECIN.RECORD-TYPE = '4';)*/

    /* * MOVE WORKING STORAGE FOR THE FLAG,*/
    AIF20W.WS-PREV-SSN = SR-AIF20-RECIN.INPUT-SSN;
    AIF20W.WS-PREV-CAN = SR-AIF20-RECIN.CAN;

    /* * EMPTY OR INIT SR-AIF20 ***/
    set SR-AIF20-RECIN empty;

    /* * SCAN NEXT RECORD FROM FLAT FILE ***/
    AIF20P9-SCAN-INFLE();

  end /* *(WHILE SR-AIF20-RECIN NOT EOF)*/

  /* * PROCESS TO CLOSE THE APPLICATION*/
  AIF20P9-CLOS-PARA();
end // end AIF20P9-MAIN
