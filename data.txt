// Control Retrieve of Data
Function DMI01P9-PROC()
  /* GET EMPLOYER DATA*/
  set H2-EMPLOYER empty;
  H2-EMPLOYER.EMPLR-ID = DMI01W.WS-EMPLR-ID;
  DMI01P9-S01();
  /* GET NCP DATA*/
  set H2-PERSON empty;
  H2-PERSON.UPI = DMI01W.WS-NCP-UPI-NUM;
  DMI01P9-S02();
  /* GET CASE DATA*/
  set H2-IVD-CASE empty;
  H2-IVD-CASE.NCP-UPI = DMI01W.WS-NCP-UPI-NUM;
  H2-IVD-CASE.CASE-SUFX = DMI01W.WS-CASE-SUFX;
  DMI01P9-S03();
  /* GET CASEWORKER DATA*/
  set H2-4-DCCNMU-SPTURE empty;
  H2-4-DCCNMU-SPTURE.UPI = DMI01W.WS-NCP-UPI-NUM;
  H2-4-DCCNMU-SPTURE.CASE-SUFX = DMI01W.WS-CASE-SUFX;
  DMI01P9-S04();
  set H2-OFFICE empty;
  H2-OFFICE.OFC-CD = H2-4-DCCNMU-SPTURE.OFC-CD;
  DMI01P9-S05(); /* OFFICE*/
  /*  */
  /* --- Correspondence address from IRG:*/
  /*  */
  set H2-INTERST-RFGUIDE empty;
  /*  */
  H2-INTERST-RFGUIDE.FIPS-CD = H2-OFFICE.FIPS-CD;
  H2-INTERST-RFGUIDE.FIPS-CD-SUFX = H2-OFFICE.FIPS-CD-SUFX;
  DMI01P9-S06();
  /*  */
  /* Second attempt:*/
  if (sysVar.sqlData.sqlcode == 100)
    set H2-INTERST-RFGUIDE empty;
    H2-INTERST-RFGUIDE.FIPS-CD = H2-OFFICE.FIPS-CD;
    H2-INTERST-RFGUIDE.FIPS-CD-SUFX = H2-OFFICE.FIPS-CD-SUFX;
    DMI01P9-S07();
  end
  /*  */
  /* Third attempt:*/
  if (sysVar.sqlData.sqlcode == 100)
    set H2-INTERST-RFGUIDE empty;
    H2-INTERST-RFGUIDE.FIPS-CD = H2-OFFICE.FIPS-CD;
    H2-INTERST-RFGUIDE.FIPS-CD-SUFX = H2-OFFICE.FIPS-CD-SUFX;
    DMI01P9-S08();
  end
  /*  */
  DMI01W1.WS-WRKR-FAX = H2-INTERST-RFGUIDE.FAX-NMB;
  /*  */
  if (H2-INTERST-RFGUIDE.PHN-EXT > " ")
    VSTRREC4.USTRING = H2-INTERST-RFGUIDE.PHN-NMB;
    VSTRREC4.UCONCAT = "X";
    VSTRREC4.UCONCAT2 = H2-INTERST-RFGUIDE.PHN-EXT;
    VSTRREC4.USTRLEN = 10;
    VSTRREC4.UCONLEN = 1;
    VSTRREC4.UCONLEN2 = 5;
    VSTRREC4.UCONLEN3 = 0;
    call "WST50A" (VSTRREC4) {isNoRefresh = yes};
    DMI01W1.WS-WRKR-PHONE = VSTRREC4.USTRING;
  else
    DMI01W1.WS-WRKR-PHONE = H2-INTERST-RFGUIDE.PHN-NMB;
  end
  /*  */
  /* *****************************************************************/


  /* LOAD DATA IN WORKING-STORAGE*/
  DMI01W1.WS-REC-TYPE = "EV  ";
  DMI01W1.WS-EMPLR-ID = DMI01W.WS-EMPLR-ID;
  /*  */
  DMI01W1.WS-FORMATTED-FEIN = DMI01W.WS-EMPLR-ID;
  if (DMI01W1.WS-FORMATTED-FEIN-FILL1 == "-")
    DMI01W1.WS-FEIN = DMI01W.WS-EMPLR-ID;
  else
    DMI01W1.WS-FEIN = H2-EMPLOYER.FEIN;
  end
  DMI01W1.WS-EMPLR-NM = H2-EMPLOYER.EMPLR-NM;
  DMI01W1.WS-EMPLR-ADR-LN1 = H2-EMPLOYER.ADR-LN1;
  DMI01W1.WS-EMPLR-ADR-LN2 = H2-EMPLOYER.ADR-LN2;
  DMI01W1.WS-EMPLR-CITY = H2-EMPLOYER.CITY;
  DMI01W1.WS-EMPLR-ST = H2-EMPLOYER.ST-CD;
  DMI01W1.WS-EMPLR-ZIP = H2-EMPLOYER.ZIP-CD;
  DMI01W1.WS-EMPLR-CNTRY = H2-EMPLOYER.CNTRY;
  DMI01W1.WS-NCP-FRST-NM = H2-PERSON.FRST-NM;
  DMI01W1.WS-NCP-MID-NM = H2-PERSON.MID-NM;
  DMI01W1.WS-NCP-LST-NM = H2-PERSON.LST-NM;
  DMI01W1.WS-NCP-SSN = H2-PERSON.SSN;
  DMI01W1.WS-NCP-UPI = DMI01W.WS-NCP-UPI;
  DMI01W1.WS-CASE-ID = DMI01W.WS-CASE-ID;
  DMI01W1.WS-CNTY-CD = DMI01W.WS-CNTY-CD;
  DMI01W1.WS-FUNC-AREA-CD = H2-IVD-CASE.FUNC-AREA-CD;
  /*  */
  DFC03W1.PIO-STR-1 = H2-4-DCCNMU-SPTURE.FRST-NM;
  DFC03W1.PIO-STR-2 = H2-4-DCCNMU-SPTURE.MID-NM;
  DFC03W1.PIO-STR-3 = H2-4-DCCNMU-SPTURE.LST-NM;
  DFC03W1.PIO-STR-4 = H2-4-DCCNMU-SPTURE.MODFR;
  DFC03W1.PIO-STR-5 = " ";
  call "DFC03A" (DFC03W1) {isNoRefresh = yes};
  DMI01W1.WS-WRKR-NM = DFC03W1.PIO-STR-CHAR;
  /*  */
  DMI01W1.WS-WRKR-TITLE = "CASE MANAGER";
  DMI01W1.WS-NOMADS-USRID = H2-4-DCCNMU-SPTURE.NOMADS-USRID;
  DMI01W1.WS-WRKR-EMAIL = "http://dwss.nv.gov/employers.htm";
  /* WRITE SEQUENTIAL RECORD*/
  DMI01P9-WRITE();
end // end DMI01P9-PROC


// Get employer data
Function DMI01P9-S01()
  WDB00PI();
  VDBCONTROL.UACCTYP = "R";
  VDBCOMMON.UAPPLNAM = "DMI01A";

  VDBCOMMON.USQLREC = "H2-EMPLOYER";
  VDBCONTROL.UPROC-NM = "DMI01P9-S01";
  VDBCONTROL.UPROC-OPT = "INQUIRY";
  VDBCONTROL.UPROC-OBJ = "H2-EMPLOYER";
  try
    get H2-EMPLOYER singleRow
      with #sql{
        select EMPLR_NM,
          ZIP_CD, ST_CD, CITY,
          ADR_LN1, ADR_LN2,
          CNTRY,
          EMPLR_ID, FEIN
        from TWNEMPLOYER T1
         --** INSERT WHERE, GROUP BY AND HAVING CLAUSES HERE **
        WHERE EMPLR_ID = :EMPLR-ID
         --** INSERT ORDER BY CLAUSE HERE **
      }
      into EMPLR-NM,
           ZIP-CD, ST-CD, CITY,
           ADR-LN1, ADR-LN2,
           CNTRY,
           EMPLR-ID, FEIN ;
  end

  if (sysVar.sqlData.sqlcode == 0)
    VDBCONTROL.UROWR = VDBCONTROL.UROWR + 1;
  end
  WDB00PE-SQLERR();
  if (VDBCONTROL.URC > 104)
    exit program;
  end
end // end DMI01P9-S01

// get wrkr name,title
Function DMI01P9-S04()
  /* Get the worker name and title*/
  VDBCOMMON.USQLREC = "H2-4-DCCNMU-SPTURE";
  VDBCONTROL.UPROC-OBJ = "H2-4-DCCNMU-SPTURE";
  VDBCONTROL.UPROC-NM = "DMI01P9-S04";
  VDBCONTROL.UPROC-OPT = "INQUIRY";

  call "DATEAPP" (GDW, VDBCONTROL, VMESSAGE) {isNoRefresh = yes};
  try
    get H2-4-DCCNMU-SPTURE singleRow
      with #sql{
        select T2.FRST_NM,T2.LST_NM,T2.MID_NM,T2.MODFR,T4.ROLE_DESC,T2.OFC_CD,
          T2.NOMADS_USRID
        from TWNIVD_CASELD_CASE T1,
             TWNNOmADS_USER T2,
             TWNSUP_UNT_POS_HST T3,
             TWNUSR_ROLE T4
        WHERE
                     T1.UPI=:UPI AND
                     T1.CASE_SUFX=:CASE-SUFX AND
                     T1.CASELD_IND = 'N' AND
                     T1.OFC_CD = T3.OFC_CD AND
                     T1.PRGM_OFC_TYPE = T3.PRGM_OFC_TYPE AND
                     T1.SPRVS_UNIT = T3.SPRVS_UNIT AND
                     T1.PSN_NMB = T3.PSN_NMB AND
                     (T3.ASMNT_BEG_DT BETWEEN '0001-01-01' AND :GDW.DTCD) AND
                     (T3.ASMNT_END_DT BETWEEN :GDW.DTCD AND '9999-12-31') AND
                      T2.NOMADS_USRID = T3.NOMADS_USRID AND
                      T2.ROLE_ID = T4.ROLE_ID
         --** INSERT ORDER BY CLAUSE HERE **
      }
      into FRST-NM,LST-NM,MID-NM,MODFR,ROLE-DESC,OFC-CD,
           NOMADS-USRID ;
  end

  WDB00PE-SQLERR();
  if (VDBCONTROL.URC > 104)
    exit program;
  end
end // end DMI01P9-S04

// Select IRG based on ADR_TYPE2
Function DMI01P9-S06()
  VDBCOMMON.USQLREC = "H2-INTERST-RFGUIDE";
  VDBCONTROL.UPROC-OBJ = "H2-INTERST-RFGUIDE";
  VDBCONTROL.UPROC-NM = "DMI01P9-S06";
  VDBCONTROL.UPROC-OPT = "INQUIRY";
  VDBCOMMON.UNRF = "1";
  VDBCOMMON.UDUP = "N";
  /*  */
  try
    get H2-INTERST-RFGUIDE
      with #sql{
        select AGENCY_NM, ST_CD_CNTRY_NM, ADLN1, ADLN2, CITY,
          ZIP_CD, PHN_NMB, PHN_EXT, FAX_NMB
        from TWNINTERST_RFGUIDE T1
        WHERE ADR_TYPE2    =  :ADR-TYPE2
          AND FIPS_CD      =  :FIPS-CD
          AND FIPS_CD_SUFX =  :FIPS-CD-SUFX
          AND STS_CD       <> 'D'
        ORDER BY LST_UPD_DT DESC
        FOR FETCH ONLY
         --** INSERT ORDER BY CLAUSE HERE **
      }
      into AGENCY-NM, ST-CD-CNTRY-NM, ADLN1, ADLN2, CITY,
           ZIP-CD, PHN-NMB, PHN-EXT, FAX-NMB ;
  end
  /*  */
  WDB00PE-SQLERR();
  if (VDBCONTROL.URC > 104)
    exit program;
  end
  /*  */
end // end DMI01P9-S06

// Add the record
Function DMI01P9-WRITE()
  /* Description :*/
  /* This process writes a record to SR-DMI01-RECOUT.*/
  /*  */
  /* Change History :*/
  /* Date       Usrid  WI     Description*/
  /* ---------  -----  -----  -----------*/
  /* 07Jun2011  WNMAV  1180   New application which is a file extract*/
  /* for EWS EV form.*/
  /*  */
  SR-DMI01-RECOUT.FILE-DATA = DMI01W1.WS-OUT-DATA;
  try
    add SR-DMI01-RECOUT ;
  end

  if (SR-DMI01-RECOUT is ioError)
    VDBCONTROL.UMSQLCODE = 0;
    VDBCONTROL.UPROC-OBJ = "DMI01T1";
    VDBCONTROL.UPROC-NM = "DMI01P9-WRITE";
    VDBCONTROL.UPROC-OPT = "ADD";
    DMI01W1.WS-ERR-CD = sysVar.errorCode;
    VMESSAGE.UMSGINS[1] = DMI01W1.WS-ERR-LAST-FOUR;
    VMESSAGE.UMSGINS[2] = "DMI01A";
    VMESSAGE.UMSGCODE = " ";
    exit program;
  end

  /*  */
end // end DMI01P9-WRITE
