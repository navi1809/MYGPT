WITH
PARAMS AS (
    SELECT
        SYSTEM_DATE AS REPORTDATE,
        DATE(PROCESS_CCYY || '-' || RIGHT('00' || VARCHAR(PROCESS_MM), 2) || '-01') AS MONTH_BEGIN,
        MTH_END_DT AS MONTH_END,
        CASE
            WHEN PROCESS_MM > 6 THEN DATE(PROCESS_CCYY || '-07-01')
            ELSE DATE(PRV_PROCESS_CCYY || '-07-01')
        END AS FISCAL_BEGIN,
        CASE
            WHEN PROCESS_MM > 6 THEN DATE((PROCESS_CCYY + 1) || '-06-30')
            ELSE DATE(PROCESS_CCYY || '-06-30')
        END AS FISCAL_END,
        CASE
            WHEN (PROCESS_MM + 1) > 12
                THEN DATE((PROCESS_CCYY - 3) || '-01-01')
            ELSE DATE((PROCESS_CCYY - 3) || '-' || RIGHT('00' || VARCHAR(PROCESS_MM + 1), 2) || '-01')
        END AS DT_36_BEGIN,
        YEAR(SYSTEM_DATE) AS REPORT_YEAR,
        MONTH(SYSTEM_DATE) AS REPORT_MONTH
    FROM {?SCHEMA}.TWNSYSDATE
),

/* Small driving population */
BASE_POP AS (
    SELECT DISTINCT
        CM.UPI,
        CM.ICI,
        CL.OFC_CD,
        O.OFC_NM
    FROM {?SCHEMA}.TWNFAM_CASELD_CASE CL
    JOIN {?SCHEMA}.TWNOFFICE O
      ON O.OFC_CD = CL.OFC_CD
    JOIN {?SCHEMA}.TWNPRGM_CASE_HSTRY PH
      ON PH.ICI = CL.ICI
     AND PH.PRGM_CASE_TYPE = 'FS'
    JOIN {?SCHEMA}.TWNCASE_MEMBER_HST CM
      ON CM.ICI = CL.ICI
    WHERE CL.OFC_CD BETWEEN 'AA' AND 'ZZ'
),

/* One filtered pass over eligibility history */
ELIG_BASE AS (
    SELECT
        BP.UPI,
        BP.ICI,
        BP.OFC_CD,
        BP.OFC_NM,
        E.PRD_BEG_DT,
        E.PRD_END_DT,
        E.ABAWD_FG
    FROM {?SCHEMA}.TWNPERSON_ELIG_HST E
    JOIN BASE_POP BP
      ON BP.UPI = E.UPI
    CROSS JOIN PARAMS P
    WHERE E.PRD_BEG_DT >= P.DT_36_BEGIN
      AND E.PRD_END_DT <= P.MONTH_END
),

/* Eligibility counts in a single aggregation */
ELIG_AGG AS (
    SELECT
        UPI,
        OFC_CD,
        OFC_NM,

        SUM(CASE
                WHEN ABAWD_FG NOT IN ('X','N')
                 AND PRD_BEG_DT >= (SELECT FISCAL_BEGIN FROM PARAMS)
                THEN 1 ELSE 0
            END) AS LINE1_COUNT_FISCAL,

        SUM(CASE
                WHEN ABAWD_FG NOT IN ('X','N')
                 AND PRD_BEG_DT >= (SELECT MONTH_BEGIN FROM PARAMS)
                THEN 1 ELSE 0
            END) AS LINE1_COUNT_MONTH,

        SUM(CASE
                WHEN ABAWD_FG = 'Y'
                 AND PRD_BEG_DT >= (SELECT MONTH_BEGIN FROM PARAMS)
                THEN 1 ELSE 0
            END) AS Y_MONTH_COUNT,

        SUM(CASE
                WHEN ABAWD_FG = 'Y'
                 AND PRD_BEG_DT >= (SELECT FISCAL_BEGIN FROM PARAMS)
                THEN 1 ELSE 0
            END) AS Y_FISCAL_COUNT,

        SUM(CASE
                WHEN ABAWD_FG = 'Y'
                THEN 1 ELSE 0
            END) AS Y_36M_COUNT,

        SUM(CASE
                WHEN ABAWD_FG = 'R'
                 AND PRD_BEG_DT >= (SELECT MONTH_BEGIN FROM PARAMS)
                THEN 1 ELSE 0
            END) AS R_MONTH_COUNT,

        SUM(CASE
                WHEN ABAWD_FG = 'R'
                 AND PRD_BEG_DT >= (SELECT FISCAL_BEGIN FROM PARAMS)
                THEN 1 ELSE 0
            END) AS R_FISCAL_COUNT,

        SUM(CASE
                WHEN ABAWD_FG = 'R'
                THEN 1 ELSE 0
            END) AS R_36M_COUNT,

        SUM(CASE
                WHEN ABAWD_FG = 'R'
                 AND PRD_BEG_DT >= ADD_MONTHS((SELECT MONTH_BEGIN FROM PARAMS), -1)
                 AND PRD_END_DT <= ADD_MONTHS((SELECT MONTH_END FROM PARAMS), -1)
                THEN 1 ELSE 0
            END) AS R_PREV_MONTH_COUNT,

        SUM(CASE
                WHEN ABAWD_FG = 'R'
                 AND PRD_BEG_DT >= (SELECT FISCAL_BEGIN FROM PARAMS)
                THEN 1 ELSE 0
            END) AS WS_ABWD_CNTR,

        SUM(CASE
                WHEN ABAWD_FG = 'Z'
                 AND PRD_BEG_DT >= (SELECT FISCAL_BEGIN FROM PARAMS)
                THEN 1 ELSE 0
            END) AS Z_FIS_COUNT,

        SUM(CASE
                WHEN ABAWD_FG = 'O'
                 AND PRD_BEG_DT >= (SELECT FISCAL_BEGIN FROM PARAMS)
                THEN 1 ELSE 0
            END) AS O_FIS_COUNT,

        SUM(CASE
                WHEN ABAWD_FG = 'Z'
                 AND PRD_BEG_DT >= (SELECT MONTH_BEGIN FROM PARAMS)
                THEN 1 ELSE 0
            END) AS Z_MTH_COUNT,

        SUM(CASE
                WHEN ABAWD_FG = 'O'
                 AND PRD_BEG_DT >= (SELECT MONTH_BEGIN FROM PARAMS)
                THEN 1 ELSE 0
            END) AS O_MTH_COUNT
    FROM ELIG_BASE
    GROUP BY UPI, OFC_CD, OFC_NM
),

/* Member history only for rows that can matter */
CASE_MEMBER_BASE AS (
    SELECT
        BP.UPI,
        BP.ICI,
        BP.OFC_CD,
        BP.OFC_NM,
        CM.PRD_BEG_DT,
        CM.PRD_END_DT,
        CM.ET_EXMPT_RSN_CD,
        CM.ET_PRTCP_CD,
        CM.FS_ASST_IND
    FROM {?SCHEMA}.TWNCASE_MEMBER_HST CM
    JOIN BASE_POP BP
      ON BP.UPI = CM.UPI
     AND BP.ICI = CM.ICI
    CROSS JOIN PARAMS P
    WHERE CM.PRD_BEG_DT >= P.DT_36_BEGIN
      AND CM.PRD_END_DT <= P.MONTH_END
),

/* Join member history to eligibility only once */
CASE_MEMBER_ELIG AS (
    SELECT
        CMB.UPI,
        CMB.ICI,
        CMB.OFC_CD,
        CMB.OFC_NM,
        CMB.PRD_BEG_DT,
        CMB.PRD_END_DT,
        CMB.ET_EXMPT_RSN_CD,
        CMB.ET_PRTCP_CD,
        CMB.FS_ASST_IND,
        EB.ABAWD_FG
    FROM CASE_MEMBER_BASE CMB
    JOIN ELIG_BASE EB
      ON EB.UPI = CMB.UPI
     AND EB.ICI = CMB.ICI
     AND EB.PRD_BEG_DT = CMB.PRD_BEG_DT
     AND EB.PRD_END_DT = CMB.PRD_END_DT
),

CASE_MEMBER_AGG AS (
    SELECT
        UPI,
        OFC_CD,
        OFC_NM,

        SUM(CASE
                WHEN ABAWD_FG = 'W'
                 AND ET_EXMPT_RSN_CD NOT IN ('27','28')
                 AND FS_ASST_IND = 'Y'
                 AND PRD_BEG_DT >= (SELECT FISCAL_BEGIN FROM PARAMS)
                THEN 1 ELSE 0
            END) AS LINE3A1_FYR_COUNTS,

        SUM(CASE
                WHEN ABAWD_FG = 'W'
                 AND ET_EXMPT_RSN_CD NOT IN ('27','28')
                 AND FS_ASST_IND = 'Y'
                 AND PRD_BEG_DT >= (SELECT MONTH_BEGIN FROM PARAMS)
                THEN 1 ELSE 0
            END) AS LINE3A1_MTH_COUNTS,

        SUM(CASE
                WHEN ABAWD_FG = 'W'
                 AND ET_EXMPT_RSN_CD = '27'
                 AND FS_ASST_IND = 'Y'
                 AND PRD_BEG_DT >= (SELECT FISCAL_BEGIN FROM PARAMS)
                THEN 1 ELSE 0
            END) AS WS_3A2_FIS_CNT,

        SUM(CASE
                WHEN ABAWD_FG = 'W'
                 AND ET_EXMPT_RSN_CD = '28'
                 AND FS_ASST_IND = 'Y'
                 AND PRD_BEG_DT >= (SELECT FISCAL_BEGIN FROM PARAMS)
                THEN 1 ELSE 0
            END) AS WS_3A3_FIS_CNT,

        SUM(CASE
                WHEN ABAWD_FG = 'W'
                 AND ET_EXMPT_RSN_CD = '27'
                 AND FS_ASST_IND = 'Y'
                 AND PRD_BEG_DT >= (SELECT MONTH_BEGIN FROM PARAMS)
                THEN 1 ELSE 0
            END) AS WS_3A2_MTH_CNT,

        SUM(CASE
                WHEN ABAWD_FG = 'W'
                 AND ET_EXMPT_RSN_CD = '28'
                 AND FS_ASST_IND = 'Y'
                 AND PRD_BEG_DT >= (SELECT MONTH_BEGIN FROM PARAMS)
                THEN 1 ELSE 0
            END) AS WS_3A3_MTH_CNT,

        SUM(CASE
                WHEN ABAWD_FG = 'W'
                 AND ET_PRTCP_CD = 'E'
                 AND ET_EXMPT_RSN_CD = '10'
                 AND FS_ASST_IND = 'Y'
                 AND PRD_BEG_DT >= (SELECT FISCAL_BEGIN FROM PARAMS)
                THEN 1 ELSE 0
            END) AS WS_4A_FIS_CNT,

        SUM(CASE
                WHEN ABAWD_FG = 'W'
                 AND ET_PRTCP_CD = 'E'
                 AND ET_EXMPT_RSN_CD IN ('25','26')
                 AND FS_ASST_IND = 'Y'
                 AND PRD_BEG_DT >= (SELECT FISCAL_BEGIN FROM PARAMS)
                THEN 1 ELSE 0
            END) AS WS_4B_FIS_CNT,

        SUM(CASE
                WHEN ABAWD_FG = 'W'
                 AND ET_PRTCP_CD = 'E'
                 AND ET_EXMPT_RSN_CD = '06'
                 AND FS_ASST_IND = 'Y'
                 AND PRD_BEG_DT >= (SELECT FISCAL_BEGIN FROM PARAMS)
                THEN 1 ELSE 0
            END) AS WS_4C_FIS_CNT,

        SUM(CASE
                WHEN ABAWD_FG = 'W'
                 AND ET_PRTCP_CD = 'E'
                 AND ET_EXMPT_RSN_CD = '18'
                 AND FS_ASST_IND = 'Y'
                 AND PRD_BEG_DT >= (SELECT FISCAL_BEGIN FROM PARAMS)
                THEN 1 ELSE 0
            END) AS WS_4D_FIS_CNT,

        SUM(CASE
                WHEN ABAWD_FG = 'W'
                 AND ET_PRTCP_CD = 'E'
                 AND ET_EXMPT_RSN_CD = '10'
                 AND FS_ASST_IND = 'Y'
                 AND PRD_BEG_DT >= (SELECT MONTH_BEGIN FROM PARAMS)
                THEN 1 ELSE 0
            END) AS WS_4A_MTH_CNT,

        SUM(CASE
                WHEN ABAWD_FG = 'W'
                 AND ET_PRTCP_CD = 'E'
                 AND ET_EXMPT_RSN_CD IN ('25','26')
                 AND FS_ASST_IND = 'Y'
                 AND PRD_BEG_DT >= (SELECT MONTH_BEGIN FROM PARAMS)
                THEN 1 ELSE 0
            END) AS WS_4B_MTH_CNT,

        SUM(CASE
                WHEN ABAWD_FG = 'W'
                 AND ET_PRTCP_CD = 'E'
                 AND ET_EXMPT_RSN_CD = '06'
                 AND FS_ASST_IND = 'Y'
                 AND PRD_BEG_DT >= (SELECT MONTH_BEGIN FROM PARAMS)
                THEN 1 ELSE 0
            END) AS WS_4C_MTH_CNT,

        SUM(CASE
                WHEN ABAWD_FG = 'W'
                 AND ET_PRTCP_CD = 'E'
                 AND ET_EXMPT_RSN_CD = '18'
                 AND FS_ASST_IND = 'Y'
                 AND PRD_BEG_DT >= (SELECT MONTH_BEGIN FROM PARAMS)
                THEN 1 ELSE 0
            END) AS WS_4D_MTH_CNT,

        SUM(CASE
                WHEN ABAWD_FG = 'W'
                 AND ET_PRTCP_CD = 'E'
                 AND ET_EXMPT_RSN_CD NOT IN ('06','10','18','25','26')
                 AND FS_ASST_IND = 'Y'
                 AND PRD_BEG_DT >= (SELECT FISCAL_BEGIN FROM PARAMS)
                THEN 1 ELSE 0
            END) AS WS_4E_FIS_CNT,

        SUM(CASE
                WHEN ABAWD_FG = 'W'
                 AND ET_PRTCP_CD = 'E'
                 AND ET_EXMPT_RSN_CD NOT IN ('06','10','18','25','26')
                 AND FS_ASST_IND = 'Y'
                 AND PRD_BEG_DT >= (SELECT MONTH_BEGIN FROM PARAMS)
                THEN 1 ELSE 0
            END) AS WS_4E_MTH_CNT
    FROM CASE_MEMBER_ELIG
    GROUP BY UPI, OFC_CD, OFC_NM
),

LATEST_VER AS (
    SELECT
        ICI,
        PRGM_CASE_TYPE,
        BNFT_YR,
        BNFT_MTH,
        MAX(VER_NMB) AS VER_NMB
    FROM {?SCHEMA}.TWNELIG_VERSION
    WHERE PRGM_CASE_TYPE = 'FS'
      AND LST_POSTD_VER_IND = 'Y'
    GROUP BY ICI, PRGM_CASE_TYPE, BNFT_YR, BNFT_MTH
),

/* Result rows for line 5 logic, filtered once */
TSRSLT_BASE AS (
    SELECT
        T.UPI,
        T.ICI,
        BP.OFC_CD,
        BP.OFC_NM,
        T.BNFT_YR,
        T.BNFT_MTH,
        T.MBR_RSLT_CD,
        T.RSRCE_TST_RSCD,
        T.SSN_TEST_RSLT_CD,
        T.SFU_TST_RSCD,
        T.SSN_OVRD_RSLT_CD,
        T.GRP_LVG_TST_RSCD,
        T.STDNT_TST_RSCD,
        T.FS_ATTND_TST_RSCD,
        T.BRDR_TST_RSCD,
        T.TEMP_ABSN_TST_RSCD,
        T.STRK_TST_RSCD,
        T.TEMP_ABSN_OVRS_CD,
        T.LMPSM_INC_TSTRS_CD,
        T.ALN_RES_TST_RSCD,
        T.CHD_AGE_TST_RSCD,
        T.HM_PRSNT_TST_RSCD,
        T.DUPL_AST_TST_RSCD,
        T.IPV_TEST_RSCD,
        T.WORK_PRTP_TST_RSCD,
        T.CTZNP_TST_RSCD,
        T.FELON_TST_RSCD,
        T.DRGCNVT_TST_RSCD,
        T.ABAWD_TST_RSCD
    FROM {?SCHEMA}.TWNELIG_MBR_TSRSLT T
    JOIN BASE_POP BP
      ON BP.UPI = T.UPI
     AND BP.ICI = T.ICI
    JOIN LATEST_VER LV
      ON LV.ICI = T.ICI
     AND LV.PRGM_CASE_TYPE = 'FS'
     AND LV.BNFT_YR = T.BNFT_YR
     AND LV.BNFT_MTH = T.BNFT_MTH
     AND LV.VER_NMB = T.VER_NMB
    WHERE T.PRGM_CASE_TYPE = 'FS'
      AND T.MBR_RSLT_CD = 'I'
      AND T.RSRCE_TST_RSCD <> 'F'
      AND T.SSN_TEST_RSLT_CD <> 'F'
      AND T.SFU_TST_RSCD <> 'F'
      AND T.SSN_OVRD_RSLT_CD <> 'F'
      AND T.GRP_LVG_TST_RSCD <> 'F'
      AND T.STDNT_TST_RSCD <> 'F'
      AND T.FS_ATTND_TST_RSCD <> 'F'
      AND T.BRDR_TST_RSCD <> 'F'
      AND T.TEMP_ABSN_TST_RSCD <> 'F'
      AND T.STRK_TST_RSCD <> 'F'
      AND T.TEMP_ABSN_OVRS_CD <> 'F'
      AND T.LMPSM_INC_TSTRS_CD <> 'F'
      AND T.ALN_RES_TST_RSCD <> 'F'
      AND T.CHD_AGE_TST_RSCD <> 'F'
      AND T.HM_PRSNT_TST_RSCD <> 'F'
      AND T.DUPL_AST_TST_RSCD <> 'F'
      AND T.IPV_TEST_RSCD <> 'F'
      AND T.WORK_PRTP_TST_RSCD <> 'F'
      AND T.CTZNP_TST_RSCD <> 'F'
      AND T.FELON_TST_RSCD <> 'F'
      AND T.DRGCNVT_TST_RSCD <> 'F'
      AND T.ABAWD_TST_RSCD = 'F'
),

LINE5_AGG AS (
    SELECT
        T.UPI,
        T.OFC_CD,
        T.OFC_NM,
        SUM(CASE
                WHEN (
                    (T.BNFT_YR = YEAR((SELECT FISCAL_BEGIN FROM PARAMS)) AND T.BNFT_MTH BETWEEN '07' AND '12')
                    OR
                    (T.BNFT_YR = YEAR((SELECT FISCAL_END FROM PARAMS)) AND T.BNFT_MTH BETWEEN '01' AND '06')
                )
                 AND COALESCE(EA.Y_FISCAL_COUNT, 0) >= 3
                THEN 1 ELSE 0
            END) AS LINE5_FIS_CNT,
        SUM(CASE
                WHEN T.BNFT_YR = YEAR((SELECT REPORTDATE FROM PARAMS))
                 AND T.BNFT_MTH = RIGHT('00' || VARCHAR((SELECT REPORT_MONTH FROM PARAMS)), 2)
                THEN 1 ELSE 0
            END) AS LINE5_MTH_CNT
    FROM TSRSLT_BASE T
    LEFT JOIN ELIG_AGG EA
      ON EA.UPI = T.UPI
     AND EA.OFC_CD = T.OFC_CD
    GROUP BY T.UPI, T.OFC_CD, T.OFC_NM
),

/* Person-level metrics, one row per UPI/office */
PERSON_METRICS AS (
    SELECT
        BP.UPI,
        BP.OFC_CD,
        BP.OFC_NM,

        COALESCE(EA.LINE1_COUNT_FISCAL, 0) AS LINE1_COUNT_FISCAL,
        COALESCE(EA.LINE1_COUNT_MONTH, 0) AS LINE1_COUNT_MONTH,

        COALESCE(EA.Y_MONTH_COUNT, 0) AS Y_MONTH_COUNT,
        COALESCE(EA.Y_FISCAL_COUNT, 0) AS Y_FISCAL_COUNT,
        COALESCE(EA.Y_36M_COUNT, 0) AS Y_36M_COUNT,

        COALESCE(L5.LINE5_FIS_CNT, 0) AS LINE5_FIS_CNT,
        COALESCE(L5.LINE5_MTH_CNT, 0) AS LINE5_MTH_CNT,

        COALESCE(EA.R_MONTH_COUNT, 0) AS R_MONTH_COUNT,
        COALESCE(EA.R_FISCAL_COUNT, 0) AS R_FISCAL_COUNT,
        COALESCE(EA.R_36M_COUNT, 0) AS R_36M_COUNT,
        COALESCE(EA.R_PREV_MONTH_COUNT, 0) AS R_PREV_MONTH_COUNT,
        COALESCE(EA.WS_ABWD_CNTR, 0) AS WS_ABWD_CNTR,

        COALESCE(CMA.LINE3A1_FYR_COUNTS, 0) AS LINE3A1_FYR_COUNTS,
        COALESCE(CMA.LINE3A1_MTH_COUNTS, 0) AS LINE3A1_MTH_COUNTS,
        COALESCE(CMA.WS_3A2_FIS_CNT, 0) AS WS_3A2_FIS_CNT,
        COALESCE(CMA.WS_3A3_FIS_CNT, 0) AS WS_3A3_FIS_CNT,
        COALESCE(CMA.WS_3A2_MTH_CNT, 0) AS WS_3A2_MTH_CNT,
        COALESCE(CMA.WS_3A3_MTH_CNT, 0) AS WS_3A3_MTH_CNT,
        COALESCE(EA.Z_FIS_COUNT, 0) AS Z_FIS_COUNT,
        COALESCE(EA.O_FIS_COUNT, 0) AS O_FIS_COUNT,
        COALESCE(EA.Z_MTH_COUNT, 0) AS Z_MTH_COUNT,
        COALESCE(EA.O_MTH_COUNT, 0) AS O_MTH_COUNT,
        COALESCE(CMA.WS_4A_FIS_CNT, 0) AS WS_4A_FIS_CNT,
        COALESCE(CMA.WS_4B_FIS_CNT, 0) AS WS_4B_FIS_CNT,
        COALESCE(CMA.WS_4C_FIS_CNT, 0) AS WS_4C_FIS_CNT,
        COALESCE(CMA.WS_4D_FIS_CNT, 0) AS WS_4D_FIS_CNT,
        COALESCE(CMA.WS_4A_MTH_CNT, 0) AS WS_4A_MTH_CNT,
        COALESCE(CMA.WS_4B_MTH_CNT, 0) AS WS_4B_MTH_CNT,
        COALESCE(CMA.WS_4C_MTH_CNT, 0) AS WS_4C_MTH_CNT,
        COALESCE(CMA.WS_4D_MTH_CNT, 0) AS WS_4D_MTH_CNT,
        COALESCE(CMA.WS_4E_FIS_CNT, 0) AS WS_4E_FIS_CNT,
        COALESCE(CMA.WS_4E_MTH_CNT, 0) AS WS_4E_MTH_CNT
    FROM BASE_POP BP
    LEFT JOIN ELIG_AGG EA
      ON EA.UPI = BP.UPI
     AND EA.OFC_CD = BP.OFC_CD
    LEFT JOIN CASE_MEMBER_AGG CMA
      ON CMA.UPI = BP.UPI
     AND CMA.OFC_CD = BP.OFC_CD
    LEFT JOIN LINE5_AGG L5
      ON L5.UPI = BP.UPI
     AND L5.OFC_CD = BP.OFC_CD
),

OFFICE_SUMMARY AS (
    SELECT
        OFC_NM,
        OFC_CD,

        SUM(CASE WHEN LINE1_COUNT_FISCAL = 1 THEN 1 ELSE 0 END) AS LINE1_FIS,
        SUM(CASE WHEN LINE1_COUNT_MONTH = 1 THEN 1 ELSE 0 END) AS LINE1_MTH,

        SUM(CASE WHEN Y_36M_COUNT = 1 AND Y_MONTH_COUNT = 1 THEN 1 ELSE 0 END) AS WS_MTH_Y_CNT_1O,
        SUM(CASE WHEN Y_36M_COUNT = 2 AND Y_MONTH_COUNT = 1 THEN 1 ELSE 0 END) AS WS_MTH_Y_CNT_20,
        SUM(CASE WHEN Y_36M_COUNT >= 3 AND Y_MONTH_COUNT = 1 THEN 1 ELSE 0 END) AS WS_MTH_Y_CNT_30,

        SUM(CASE WHEN Y_FISCAL_COUNT = 1 THEN 1 ELSE 0 END) AS WS_FYR_Y_CNT_1O,
        SUM(CASE WHEN Y_FISCAL_COUNT = 2 THEN 1 ELSE 0 END) AS WS_FYR_Y_CNT_20,
        SUM(CASE WHEN Y_FISCAL_COUNT >= 3 THEN 1 ELSE 0 END) AS WS_FYR_Y_CNT_30,

        SUM(CASE WHEN LINE5_FIS_CNT >= 1 THEN 1 ELSE 0 END) AS LINE5_FIS,
        SUM(CASE WHEN LINE5_MTH_CNT = 1 THEN 1 ELSE 0 END) AS LINE5_MTH,

        SUM(CASE WHEN R_36M_COUNT = 1 AND R_MONTH_COUNT = 1 THEN 1 ELSE 0 END) AS WS_MTH_R_CNT_1O,
        SUM(CASE WHEN R_36M_COUNT = 2 AND R_MONTH_COUNT = 1 THEN 1 ELSE 0 END) AS WS_MTH_R_CNT_20,
        SUM(CASE WHEN R_36M_COUNT >= 3 AND R_MONTH_COUNT = 1 THEN 1 ELSE 0 END) AS WS_MTH_R_CNT_30,

        COUNT(DISTINCT CASE
            WHEN R_MONTH_COUNT >= 1 AND R_PREV_MONTH_COUNT >= 1 AND R_36M_COUNT <> 1
            THEN UPI END) AS LINE6_MTH,

        SUM(CASE WHEN R_FISCAL_COUNT = 1 THEN 1 ELSE 0 END) AS WS_FYR_R_CNT_1O,
        SUM(CASE WHEN R_FISCAL_COUNT = 2 THEN 1 ELSE 0 END) AS WS_FYR_R_CNT_20,
        COUNT(DISTINCT CASE WHEN R_FISCAL_COUNT >= 3 THEN UPI END) AS WS_FYR_R_CNT_30,

        COUNT(DISTINCT CASE WHEN WS_ABWD_CNTR > 1 THEN UPI END) AS LINE6_FIS,

        SUM(CASE WHEN LINE3A1_FYR_COUNTS > 0 THEN 1 ELSE 0 END) AS WS_3A1_FIS,
        SUM(CASE WHEN LINE3A1_MTH_COUNTS = 1 THEN 1 ELSE 0 END) AS WS_3A1_MTH,

        SUM(CASE WHEN WS_3A2_FIS_CNT > 0 THEN 1 ELSE 0 END) AS WS_3A2_FIS,
        SUM(CASE WHEN WS_3A3_FIS_CNT > 0 THEN 1 ELSE 0 END) AS WS_3A3_FIS,
        SUM(CASE WHEN WS_3A2_MTH_CNT = 1 THEN 1 ELSE 0 END) AS WS_3A2_MTH,
        SUM(CASE WHEN WS_3A3_MTH_CNT = 1 THEN 1 ELSE 0 END) AS WS_3A3_MTH,

        SUM(CASE WHEN Z_FIS_COUNT > 0 THEN 1 ELSE 0 END) AS WS_3B_FIS,
        SUM(CASE WHEN O_FIS_COUNT > 0 THEN 1 ELSE 0 END) AS WS_3C_FIS,
        SUM(CASE WHEN Z_MTH_COUNT = 1 THEN 1 ELSE 0 END) AS WS_3B_MTH,
        SUM(CASE WHEN O_MTH_COUNT = 1 THEN 1 ELSE 0 END) AS WS_3C_MTH,

        SUM(CASE WHEN WS_4A_FIS_CNT > 0 THEN 1 ELSE 0 END) AS WS_4A_FIS,
        SUM(CASE WHEN WS_4B_FIS_CNT > 0 THEN 1 ELSE 0 END) AS WS_4B_FIS,
        SUM(CASE WHEN WS_4C_FIS_CNT > 0 THEN 1 ELSE 0 END) AS WS_4C_FIS,
        SUM(CASE WHEN WS_4D_FIS_CNT > 0 THEN 1 ELSE 0 END) AS WS_4D_FIS,

        SUM(CASE WHEN WS_4A_MTH_CNT = 1 THEN 1 ELSE 0 END) AS WS_4A_MTH,
        SUM(CASE WHEN WS_4B_MTH_CNT = 1 THEN 1 ELSE 0 END) AS WS_4B_MTH,
        SUM(CASE WHEN WS_4C_MTH_CNT = 1 THEN 1 ELSE 0 END) AS WS_4C_MTH,
        SUM(CASE WHEN WS_4D_MTH_CNT = 1 THEN 1 ELSE 0 END) AS WS_4D_MTH,

        SUM(CASE WHEN WS_4E_FIS_CNT > 0 THEN 1 ELSE 0 END) AS WS_4E_FIS,
        SUM(CASE WHEN WS_4E_MTH_CNT = 1 THEN 1 ELSE 0 END) AS WS_4E_MTH,

        0 AS SORT
    FROM PERSON_METRICS
    GROUP BY OFC_NM, OFC_CD
),

FINAL_QUERY AS (
    SELECT * FROM OFFICE_SUMMARY
    UNION ALL
    SELECT
        'NEVADA STATEWIDE' AS OFC_NM,
        'NS' AS OFC_CD,

        SUM(LINE1_FIS) AS LINE1_FIS,
