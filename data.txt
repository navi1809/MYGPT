// REPORTS MAIN PARA
Function AIN23P9-MAIN()
  /* cbsi start*/
  /* ************************************************************************/
  /* This program has main control over this application*/
  /* initailises all the report variables*/
  /* prints ths report*/
  /* closes the application*/
  /* ************************************************************************/
  /* PERFORM PROCESS TO INITIALISE VARIABLES*/

  AIN23P9-INIT-PARA();

  /* PERFORM PROCESS TO PRINT REPORT*/

  AIN23P9-PRINT-PARA();

  /* call the ain55a application and closes*/

  /* move the application name*/
  AIN55W.WS-PRG-NAME = "AIN23A";

  /* call ain55a with vdbcontrol,vmessage,ain55W*/
  call "AIN55A" (VDBCONTROL, VMESSAGE, AIN55W) {isNoRefresh = yes};

  /* check for error processing*/
  if (VDBCONTROL.URC >= 104)
    WCM00P2-TO-CEP01A();
    exit program;
  end
  /* close the application*/

  exit program;


  /* cbsi end*/
end // end AIN23P9-MAIN


// POPULATING REPORTS DETAILS
Function AIN23P9-POP-DET()
  /* cbsi start*/
  /* ***********************************************************************/
  /* **/
  /* This process is to format and move the retrieved values to the report*/
  /* variables*/
  /* concatenates some variables using wst50a*/
  /* uses a common reports process format-proc*/
  /* sets all the flags to format the report variable*/
  /* ***********************************************************************/
  /* **/
  /* write the detail screen layout*/
  AIN23W-DET.S-DIST-CD = SR-AIN23-RECIN.AIN22A-01-OFC-CD;
  AIN23W-DET.S-CASE-LD = SR-AIN23-RECIN.AIN22A-01-CASLD-ID;
  AIN23W-DET.S-WRKR-NMB = SR-AIN23-RECIN.AIN22A-01-NOMADS-USRID;
  /* PTR 5508 ADD NEW-SSN TO REPORT ------- 11/08/96 -------*/
  AIN23W-DET.S-NEW-SSN = SR-AIN23-RECIN.AIN22A-01-NEW-SSN;

  /* concatenate the name in ws*/
  VSTRREC4.UCONLEN = 19;
  VSTRREC4.UCONLEN2 = 10;
  VSTRREC4.UCONLEN3 = 1;
  VSTRREC4.UCONCAT = SR-AIN23-RECIN.AIN22A-01-LAST-NAME;
  VSTRREC4.UCONCAT2 = SR-AIN23-RECIN.AIN22A-01-FIRST-NAME;
  VSTRREC4.UCONCAT3 = SR-AIN23-RECIN.AIN22A-01-MIDDLE-INIT;

  call "WST50A" (VSTRREC4) {isNoRefresh = yes};

  AIN23W.WS-NAME = VSTRREC4.USTRING;
  AIN23W-DET.S-NAME = AIN23W.WS-NAME-1;
  set VSTRREC4 empty;

  AIN23W-DET.S-CASE-NUMBER = SR-AIN23-RECIN.AIN22A-01-CASE-NUMBER;


  if (SR-AIN23-RECIN.AIN22A-01-NOMADS-AMT != 0)
    set FORMATW empty;
    FORMATW.INPUT-NUMBER = SR-AIN23-RECIN.AIN22A-01-NOMADS-AMT;
    FORMATW.LENGTH = 8;
    FORMATW.DEC-COUNT = 2;

    FORMATW.FORMAT-FLAG = "N";
    FORMATW.DOLLAR-FLAG = "N";
    REPP9-FORMAT-PROC();
    AIN23W-DET.S-SSI-NOMADS = FORMATW.OUTPUT-CHAR;
  else
    AIN23W-DET.S-SSI-NOMADS = "    0.00";
  end


  if (SR-AIN23-RECIN.AIN22A-01-INTERFACE-AMT != 0)
    set FORMATW empty;
    FORMATW.INPUT-NUMBER = SR-AIN23-RECIN.AIN22A-01-INTERFACE-AMT;
    FORMATW.LENGTH = 8;
    FORMATW.DEC-COUNT = 2;
    FORMATW.FORMAT-FLAG = "N";
    FORMATW.DOLLAR-FLAG = "N";
    REPP9-FORMAT-PROC();
    AIN23W-DET.S-SSI-AMOUNT = FORMATW.OUTPUT-CHAR;
  else
    AIN23W-DET.S-SSI-AMOUNT = "    0.00";
  end


  if (SR-AIN23-RECIN.AIN22A-01-REPORT-DIFF != 0)
    AIN23W.WS-DIFF-SIGN-FLAG = "N";
    if (SR-AIN23-RECIN.AIN22A-01-REPORT-DIFF < 0)
      AIN23W.WS-TEMP-DIFF-AMT = -1 * SR-AIN23-RECIN.AIN22A-01-REPORT-DIFF;
      AIN23W.WS-DIFF-SIGN-FLAG = "Y";
    else
      AIN23W.WS-TEMP-DIFF-AMT = SR-AIN23-RECIN.AIN22A-01-REPORT-DIFF;
    end
    set FORMATW empty;
    FORMATW.INPUT-NUMBER = AIN23W.WS-TEMP-DIFF-AMT;
    FORMATW.DEC-COUNT = 2;
    FORMATW.LENGTH = 8;
    FORMATW.FORMAT-FLAG = "N";
    FORMATW.DOLLAR-FLAG = "N";
    REPP9-FORMAT-PROC();
    AIN23W-DET.S-DIFF-AMT = FORMATW.OUTPUT-CHAR;
    if (AIN23W.WS-DIFF-SIGN-FLAG == "N")
      AIN23W-DET.S-DIFF-AMT-SIGN = " ";
    else
      AIN23W-DET.S-DIFF-AMT-SIGN = "-";
    end
  else
    AIN23W-DET.S-DIFF-AMT = "    0.00";
    AIN23W-DET.S-DIFF-AMT-SIGN = " ";
  end

  AIN23W-DET.S-ACTION = SR-AIN23-RECIN.AIN22A-01-ACTION;

    AIN23W.WS-03-PRO-CASE = AIN23W.WS-03-PRO-CASE + 1;

// PROCESS TO PRINT THE REPORT
Function AIN23P9-PRINT-PARA()
  /* cbsi start*/
  /* ***********************************************************************/
  /* This process is to print the report*/
  /* 1.prints the no data to report if there is no data*/
  /* 2.performs the additional processing process*/
  /* ***********************************************************************/
  /* ERROR PROCESSING FOR SERIAL RECORD*/

  if ((SR-AIN23-RECIN is endOfFile))
    AIN23P9-PRN-HEAD();
    REP132P9-NRF-PROC();
    return;
  end

  /* PRINT ALL THE DETAILS RETRIVED FROM SEQUENTIAL FILE*/
  AIN23W.WS-OFC-CD-OLD = SR-AIN23-RECIN.AIN22A-01-OFC-CD;


  while ((SR-AIN23-RECIN not endOfFile))

    /* PERFORM APPLICATION SPECIFIC PROCESSING*/

    /* PRINT HEADER AND SUB HEADER AT THE START OF A NEW PAGE*/

    if ((REP-COMMON.WS-NEW-PAGE == "Y"))

       /* PRINT THE HEADER DETAILS*/

      AIN23P9-PRN-HEAD();

       /* PRINT THE SUBHEADER*/

      AIN23P9-PRN-SUBH();

       /* RESET NEW PAGE INDICATOR*/

      REP-COMMON.WS-NEW-PAGE = "N";
    end

    AIN23P9-POP-DET();

       /* PERFORM PROCESS TO PRINT THE DETAILS*/

    AIN23P9-PRN-DETAIL();

    if ((REP-COMMON.WS-LINE-NO > REP-COMMON.WS-MAX-NO-LINES))
      SR-132REP-RECOUT.PRNT-CC = "1";
      REP-COMMON.WS-PAGE-NO = REP-COMMON.WS-PAGE-NO + 1;
      REP-COMMON.WS-NEW-PAGE = "Y";
    end
    /* PERFORM PROCESS TO SCAN THE CURSOR*/
    AIN23P9-SCAN();
    AIN23W.WS-OFC-CD-NEW = SR-AIN23-RECIN.AIN22A-01-OFC-CD;

    if (AIN23W.WS-OFC-CD-OLD != AIN23W.WS-OFC-CD-NEW)
      AIN23W.WS-OFC-CD-OLD = AIN23W.WS-OFC-CD-NEW;
          /* PRINT THE HEADING AND THE SUB HEADING IF END OF PAGE*/
          /* IS ENCOUNTERED*/

      if (AIN23W.WS-02-DIS-CHA != 0)
        set FORMATW empty;
        FORMATW.INPUT-NUMBER = AIN23W.WS-02-DIS-CHA;
        FORMATW.LENGTH = 7;
        FORMATW.FORMAT-FLAG = "N";
        FORMATW.DOLLAR-FLAG = "N";
        REPP9-FORMAT-PROC();
        AIN23W-SUBFOT.S-DIS-TOT-CHA-CASE = FORMATW.OUTPUT-CHAR;
      else
        AIN23W-SUBFOT.S-DIS-TOT-CHA-CASE = "      0";
      end

      if (AIN23W.WS-02-DIS-PRO != 0)
        set FORMATW empty;
        FORMATW.INPUT-NUMBER = AIN23W.WS-02-DIS-PRO;
        FORMATW.LENGTH = 7;
        FORMATW.FORMAT-FLAG = "N";
        FORMATW.DOLLAR-FLAG = "N";
        REPP9-FORMAT-PROC();
        AIN23W-SUBFOT.S-DIS-TOT-PRO-CASE = FORMATW.OUTPUT-CHAR;
      else
