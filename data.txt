WITH DATES AS (
    SELECT
        SYSTEM_DATE,
        MTH_BEG_DT,
        MTH_END_DT,
        QTR_BEG_DT,
        QTR_END_DT,
        FEDRFIS_BEG_DT,
        FEDRFIS_END_DT,
        PROCESS_MM,
        PROCESS_CCYY
    FROM {?SCHEMA}.TWNSYSDATE
),

OFFICE AS (
    SELECT
        T1.OFC_CD,
        T1.OFC_NM
    FROM {?SCHEMA}.TWNOFFICE T1
    JOIN {?SCHEMA}.TWNCODE_TBL T2
      ON T1.OFC_CD = T2.CODE_TBL_CD
     AND T2.CODE_TBL_TYP = 'DISTCOD'
    WHERE T1.OFC_CD NOT IN ('TP')
),

CASE_OFFICE AS (
    SELECT DISTINCT
        C.ICI,
        C.OFC_CD,
        O.OFC_NM
    FROM {?SCHEMA}.TWNFAM_CASELD_CASE C
    JOIN OFFICE O
      ON C.OFC_CD = O.OFC_CD
),

/* Read TWNCASE_MEMBER_HST once for the full federal fiscal window */
MEMBER_YEAR AS (
    SELECT
        M.ICI,
        M.UPI,
        M.AFDC_DUPL_ASST_IND,
        CASE
            WHEN M.PRD_BEG_DT <= D.MTH_BEG_DT
             AND M.PRD_END_DT >= D.MTH_END_DT THEN 1
            ELSE 0
        END AS IN_MTH,
        CASE
            WHEN M.PRD_BEG_DT <= D.QTR_END_DT
             AND M.PRD_END_DT >= D.QTR_BEG_DT THEN 1
            ELSE 0
        END AS IN_QTR,
        CASE
            WHEN M.PRD_BEG_DT <= D.FEDRFIS_END_DT
             AND M.PRD_END_DT >= D.FEDRFIS_BEG_DT THEN 1
            ELSE 0
        END AS IN_FSC
    FROM {?SCHEMA}.TWNCASE_MEMBER_HST M
    CROSS JOIN DATES D
    WHERE M.FS_ELIG_CD = '1'
      AND M.PRD_BEG_DT <= D.FEDRFIS_END_DT
      AND M.PRD_END_DT >= D.FEDRFIS_BEG_DT
),

/* Read TWNPERSON_ELIG_HST once for the full federal fiscal window */
PERSON_YEAR AS (
    SELECT
        P.UPI,
        P.ABAWD_FG,
        CASE
            WHEN P.PRD_BEG_DT <= D.MTH_BEG_DT
             AND P.PRD_END_DT >= D.MTH_END_DT THEN 1
            ELSE 0
        END AS IN_MTH,
        CASE
            WHEN P.PRD_BEG_DT <= D.QTR_END_DT
             AND P.PRD_END_DT >= D.QTR_BEG_DT THEN 1
            ELSE 0
        END AS IN_QTR,
        CASE
            WHEN P.PRD_BEG_DT <= D.FEDRFIS_END_DT
             AND P.PRD_END_DT >= D.FEDRFIS_BEG_DT THEN 1
            ELSE 0
        END AS IN_FSC
    FROM {?SCHEMA}.TWNPERSON_ELIG_HST P
    CROSS JOIN DATES D
    WHERE P.ABAWD_FG IN ('R','W','Y','Z')
      AND P.PRD_BEG_DT <= D.FEDRFIS_END_DT
      AND P.PRD_END_DT >= D.FEDRFIS_BEG_DT
),

/* Duplicate assistance:
   original logic counts distinct ICI + flag by period */
DUPL_CASE_FLAG AS (
    SELECT
        M.ICI,
        M.AFDC_DUPL_ASST_IND AS DUPL_ASST_FG,
        MAX(M.IN_MTH) AS IN_MTH,
        MAX(M.IN_QTR) AS IN_QTR,
        MAX(M.IN_FSC) AS IN_FSC
    FROM MEMBER_YEAR M
    WHERE M.AFDC_DUPL_ASST_IND IN ('C','D','E','S')
    GROUP BY
        M.ICI,
        M.AFDC_DUPL_ASST_IND
),

DUPL_OFFICE AS (
    SELECT
        CO.OFC_CD,
        CO.OFC_NM,
        SUM(CASE WHEN DCF.DUPL_ASST_FG = 'C' AND DCF.IN_MTH = 1 THEN 1 ELSE 0 END) AS DUPL_ASST_MTH_C,
        SUM(CASE WHEN DCF.DUPL_ASST_FG = 'D' AND DCF.IN_MTH = 1 THEN 1 ELSE 0 END) AS DUPL_ASST_MTH_D,
        SUM(CASE WHEN DCF.DUPL_ASST_FG = 'E' AND DCF.IN_MTH = 1 THEN 1 ELSE 0 END) AS DUPL_ASST_MTH_E,
        SUM(CASE WHEN DCF.DUPL_ASST_FG = 'S' AND DCF.IN_MTH = 1 THEN 1 ELSE 0 END) AS DUPL_ASST_MTH_S,

        SUM(CASE WHEN DCF.DUPL_ASST_FG = 'C' AND DCF.IN_QTR = 1 THEN 1 ELSE 0 END) AS DUPL_ASST_QTR_C,
        SUM(CASE WHEN DCF.DUPL_ASST_FG = 'D' AND DCF.IN_QTR = 1 THEN 1 ELSE 0 END) AS DUPL_ASST_QTR_D,
        SUM(CASE WHEN DCF.DUPL_ASST_FG = 'E' AND DCF.IN_QTR = 1 THEN 1 ELSE 0 END) AS DUPL_ASST_QTR_E,
        SUM(CASE WHEN DCF.DUPL_ASST_FG = 'S' AND DCF.IN_QTR = 1 THEN 1 ELSE 0 END) AS DUPL_ASST_QTR_S,

        SUM(CASE WHEN DCF.DUPL_ASST_FG = 'C' AND DCF.IN_FSC = 1 THEN 1 ELSE 0 END) AS DUPL_ASST_FSC_C,
        SUM(CASE WHEN DCF.DUPL_ASST_FG = 'D' AND DCF.IN_FSC = 1 THEN 1 ELSE 0 END) AS DUPL_ASST_FSC_D,
        SUM(CASE WHEN DCF.DUPL_ASST_FG = 'E' AND DCF.IN_FSC = 1 THEN 1 ELSE 0 END) AS DUPL_ASST_FSC_E,
        SUM(CASE WHEN DCF.DUPL_ASST_FG = 'S' AND DCF.IN_FSC = 1 THEN 1 ELSE 0 END) AS DUPL_ASST_FSC_S
    FROM DUPL_CASE_FLAG DCF
    JOIN CASE_OFFICE CO
      ON CO.ICI = DCF.ICI
    GROUP BY
        CO.OFC_CD,
        CO.OFC_NM
),

/* ABAWD:
   preserve original behavior by joining member rows to elig rows once,
   then conditionally accumulating month / quarter / fiscal counts */
ABAWD_CASE_FLAG AS (
    SELECT
        M.ICI,
        P.ABAWD_FG,
        SUM(CASE WHEN M.IN_MTH = 1 AND P.IN_MTH = 1 THEN 1 ELSE 0 END) AS ABAWD_MTH_CNT,
        SUM(CASE WHEN M.IN_QTR = 1 AND P.IN_QTR = 1 THEN 1 ELSE 0 END) AS ABAWD_QTR_CNT,
        SUM(CASE WHEN M.IN_FSC = 1 AND P.IN_FSC = 1 THEN 1 ELSE 0 END) AS ABAWD_FSC_CNT
    FROM MEMBER_YEAR M
    JOIN PERSON_YEAR P
      ON M.UPI = P.UPI
    GROUP BY
        M.ICI,
        P.ABAWD_FG
),

ABAWD_OFFICE AS (
    SELECT
        CO.OFC_CD,
        CO.OFC_NM,
        SUM(CASE WHEN ACF.ABAWD_FG = 'R' THEN ACF.ABAWD_MTH_CNT ELSE 0 END) AS ABAWD_MTH_R,
        SUM(CASE WHEN ACF.ABAWD_FG = 'W' THEN ACF.ABAWD_MTH_CNT ELSE 0 END) AS ABAWD_MTH_W,
        SUM(CASE WHEN ACF.ABAWD_FG = 'Y' THEN ACF.ABAWD_MTH_CNT ELSE 0 END) AS ABAWD_MTH_Y,
        SUM(CASE WHEN ACF.ABAWD_FG = 'Z' THEN ACF.ABAWD_MTH_CNT ELSE 0 END) AS ABAWD_MTH_Z,

        SUM(CASE WHEN ACF.ABAWD_FG = 'R' THEN ACF.ABAWD_QTR_CNT ELSE 0 END) AS ABAWD_QTR_R,
        SUM(CASE WHEN ACF.ABAWD_FG = 'W' THEN ACF.ABAWD_QTR_CNT ELSE 0 END) AS ABAWD_QTR_W,
        SUM(CASE WHEN ACF.ABAWD_FG = 'Y' THEN ACF.ABAWD_QTR_CNT ELSE 0 END) AS ABAWD_QTR_Y,
        SUM(CASE WHEN ACF.ABAWD_FG = 'Z' THEN ACF.ABAWD_QTR_CNT ELSE 0 END) AS ABAWD_QTR_Z,

        SUM(CASE WHEN ACF.ABAWD_FG = 'R' THEN ACF.ABAWD_FSC_CNT ELSE 0 END) AS ABAWD_FSC_R,
        SUM(CASE WHEN ACF.ABAWD_FG = 'W' THEN ACF.ABAWD_FSC_CNT ELSE 0 END) AS ABAWD_FSC_W,
        SUM(CASE WHEN ACF.ABAWD_FG = 'Y' THEN ACF.ABAWD_FSC_CNT ELSE 0 END) AS ABAWD_FSC_Y,
        SUM(CASE WHEN ACF.ABAWD_FG = 'Z' THEN ACF.ABAWD_FSC_CNT ELSE 0 END) AS ABAWD_FSC_Z
    FROM ABAWD_CASE_FLAG ACF
    JOIN CASE_OFFICE CO
      ON CO.ICI = ACF.ICI
    GROUP BY
        CO.OFC_CD,
        CO.OFC_NM
),

DETAIL AS (
    SELECT
        COALESCE(DO.OFC_CD, AO.OFC_CD) AS OFC_CD,
        COALESCE(DO.OFC_NM, AO.OFC_NM) AS OFC_NM,

        COALESCE(AO.ABAWD_MTH_R, 0) AS ABAWD_MTH_R,
        COALESCE(AO.ABAWD_MTH_W, 0) AS ABAWD_MTH_W,
        COALESCE(AO.ABAWD_MTH_Y, 0) AS ABAWD_MTH_Y,
        COALESCE(AO.ABAWD_MTH_Z, 0) AS ABAWD_MTH_Z,

        COALESCE(AO.ABAWD_QTR_R, 0) AS ABAWD_QTR_R,
        COALESCE(AO.ABAWD_QTR_W, 0) AS ABAWD_QTR_W,
        COALESCE(AO.ABAWD_QTR_Y, 0) AS ABAWD_QTR_Y,
        COALESCE(AO.ABAWD_QTR_Z, 0) AS ABAWD_QTR_Z,

        COALESCE(AO.ABAWD_FSC_R, 0) AS ABAWD_FSC_R,
        COALESCE(AO.ABAWD_FSC_W, 0) AS ABAWD_FSC_W,
        COALESCE(AO.ABAWD_FSC_Y, 0) AS ABAWD_FSC_Y,
        COALESCE(AO.ABAWD_FSC_Z, 0) AS ABAWD_FSC_Z,

        COALESCE(DO.DUPL_ASST_MTH_C, 0) AS DUPL_ASST_MTH_C,
        COALESCE(DO.DUPL_ASST_MTH_D, 0) AS DUPL_ASST_MTH_D,
        COALESCE(DO.DUPL_ASST_MTH_E, 0) AS DUPL_ASST_MTH_E,
        COALESCE(DO.DUPL_ASST_MTH_S, 0) AS DUPL_ASST_MTH_S,

        COALESCE(DO.DUPL_ASST_QTR_C, 0) AS DUPL_ASST_QTR_C,
        COALESCE(DO.DUPL_ASST_QTR_D, 0) AS DUPL_ASST_QTR_D,
        COALESCE(DO.DUPL_ASST_QTR_E, 0) AS DUPL_ASST_QTR_E,
        COALESCE(DO.DUPL_ASST_QTR_S, 0) AS DUPL_ASST_QTR_S,

        COALESCE(DO.DUPL_ASST_FSC_C, 0) AS DUPL_ASST_FSC_C,
        COALESCE(DO.DUPL_ASST_FSC_D, 0) AS DUPL_ASST_FSC_D,
        COALESCE(DO.DUPL_ASST_FSC_E, 0) AS DUPL_ASST_FSC_E,
        COALESCE(DO.DUPL_ASST_FSC_S, 0) AS DUPL_ASST_FSC_S,

        1 AS SORT
    FROM DUPL_OFFICE DO
    FULL OUTER JOIN ABAWD_OFFICE AO
      ON DO.OFC_CD = AO.OFC_CD
     AND DO.OFC_NM = AO.OFC_NM
),

FINAL_ROWS AS (
    SELECT
        'ST' AS OFC_CD,
        'NEVADA STATEWIDE' AS OFC_NM,
        SUM(ABAWD_MTH_R) AS ABAWD_MTH_R,
        SUM(ABAWD_MTH_W) AS ABAWD_MTH_W,
        SUM(ABAWD_MTH_Y) AS ABAWD_MTH_Y,
        SUM(ABAWD_MTH_Z) AS ABAWD_MTH_Z,
        SUM(ABAWD_QTR_R) AS ABAWD_QTR_R,
        SUM(ABAWD_QTR_W) AS ABAWD_QTR_W,
        SUM(ABAWD_QTR_Y) AS ABAWD_QTR_Y,
        SUM(ABAWD_QTR_Z) AS ABAWD_QTR_Z,
        SUM(ABAWD_FSC_R) AS ABAWD_FSC_R,
        SUM(ABAWD_FSC_W) AS ABAWD_FSC_W,
        SUM(ABAWD_FSC_Y) AS ABAWD_FSC_Y,
        SUM(ABAWD_FSC_Z) AS ABAWD_FSC_Z,

        SUM(DUPL_ASST_MTH_C) AS DUPL_ASST_MTH_C,
        SUM(DUPL_ASST_MTH_D) AS DUPL_ASST_MTH_D,
        SUM(DUPL_ASST_MTH_E) AS DUPL_ASST_MTH_E,
        SUM(DUPL_ASST_MTH_S) AS DUPL_ASST_MTH_S,
        SUM(DUPL_ASST_QTR_C) AS DUPL_ASST_QTR_C,
        SUM(DUPL_ASST_QTR_D) AS DUPL_ASST_QTR_D,
        SUM(DUPL_ASST_QTR_E) AS DUPL_ASST_QTR_E,
        SUM(DUPL_ASST_QTR_S) AS DUPL_ASST_QTR_S,
        SUM(DUPL_ASST_FSC_C) AS DUPL_ASST_FSC_C,
        SUM(DUPL_ASST_FSC_D) AS DUPL_ASST_FSC_D,
        SUM(DUPL_ASST_FSC_E) AS DUPL_ASST_FSC_E,
        SUM(DUPL_ASST_FSC_S) AS DUPL_ASST_FSC_S,
        0 AS SORT
    FROM DETAIL

    UNION ALL

    SELECT
        OFC_CD,
        OFC_NM,
        ABAWD_MTH_R,
        ABAWD_MTH_W,
        ABAWD_MTH_Y,
        ABAWD_MTH_Z,
        ABAWD_QTR_R,
        ABAWD_QTR_W,
        ABAWD_QTR_Y,
        ABAWD_QTR_Z,
        ABAWD_FSC_R,
        ABAWD_FSC_W,
        ABAWD_FSC_Y,
        ABAWD_FSC_Z,
        DUPL_ASST_MTH_C,
        DUPL_ASST_MTH_D,
        DUPL_ASST_MTH_E,
        DUPL_ASST_MTH_S,
        DUPL_ASST_QTR_C,
        DUPL_ASST_QTR_D,
        DUPL_ASST_QTR_E,
        DUPL_ASST_QTR_S,
        DUPL_ASST_FSC_C,
        DUPL_ASST_FSC_D,
        DUPL_ASST_FSC_E,
        DUPL_ASST_FSC_S,
        SORT
    FROM DETAIL
)

SELECT
    F.OFC_CD,
    F.OFC_NM,
    F.ABAWD_MTH_R,
    F.ABAWD_MTH_W,
    F.ABAWD_MTH_Y,
    F.ABAWD_MTH_Z,
    F.ABAWD_QTR_R,
    F.ABAWD_QTR_W,
    F.ABAWD_QTR_Y,
    F.ABAWD_QTR_Z,
    F.ABAWD_FSC_R,
    F.ABAWD_FSC_W,
    F.ABAWD_FSC_Y,
    F.ABAWD_FSC_Z,
    F.DUPL_ASST_MTH_C,
    F.DUPL_ASST_MTH_D,
    F.DUPL_ASST_MTH_E,
    F.DUPL_ASST_MTH_S,
    F.DUPL_ASST_QTR_C,
    F.DUPL_ASST_QTR_D,
    F.DUPL_ASST_QTR_E,
    F.DUPL_ASST_QTR_S,
    F.DUPL_ASST_FSC_C,
    F.DUPL_ASST_FSC_D,
    F.DUPL_ASST_FSC_E,
    F.DUPL_ASST_FSC_S,
    F.SORT,
    D.SYSTEM_DATE AS SYS_DATE
FROM FINAL_ROWS F
CROSS JOIN DATES D
ORDER BY
    F.SORT,
    F.OFC_CD
FOR FETCH ONLY WITH UR;
