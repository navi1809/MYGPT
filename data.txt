SELECT 
    T1.ICI, 
    T1.PRGM_CASE_STS, 
    T1.PRGM_CASE_TYPE, 
    T1.AID_CD, 
    T2.BNFT_YR, 
    T2.BNFT_MTH, 
    T2.VER_NMB, 
    T2.WTHDR_TST_RSLT_CD, 
    T2.APL_DTH_TSTRSL_CD, 
    T2.RSDN_TST_RSLT_CD, 
    T2.AGDDIS_ST_TSTRS_CD, 
    T2.VRF_TST_RSLT_CD, 
    T2.SSN_TEST_RSLT_CD, 
    T2.COOP_TEST_RSLT_CD, 
    T2.MRF_COMP_TSTRSL_CD, 
    T3.OFC_CD AS PCE_OFC_CD, 
    T3.CASLD_ID AS PCE_CASLD_ID, 
    T4.OFC_NM AS PCE_OFC_NM, 
    T3.CASLD_EFF_DT, 
    T5.LST_BU_CNSD AS BU_NMB, 
    T6.RSRCS_TST_RSLT_CD, 
    T6.GRS_INC_TST_RSLTCD 
FROM 
    TWNPRGM_CASE_HSTRY T1, 
    TWNELIG_VERSION T2, 
    TWNFAM_CASELD_CASE T3, 
    TWNOFFICE T4, 
    TWNELG_VER_MBR T5, 
    TWNELG_BDGT_UNIT T6 
WHERE 
    T1.ICI = T2.ICI 
    AND T2.LST_POSTD_VER_IND = 'Y' 
    AND T1.ICI = T3.ICI 
    AND T3.OFC_CD = T4.OFC_CD 
    AND T1.ICI = T5.ICI 
    AND T5.PRGM_CASE_TYPE = T1.PRGM_CASE_TYPE 
    AND T5.BNFT_YR = T2.BNFT_YR 
    AND T5.BNFT_MTH = T2.BNFT_MTH 
    AND T5.VER_NMB = T2.VER_NMB 
    AND T5.ICI = T6.ICI 
    AND T6.PRGM_CASE_TYPE = T1.PRGM_CASE_TYPE 
    AND T6.BNFT_YR = T2.BNFT_YR 
    AND T6.BNFT_MTH = T2.BNFT_MTH 
    AND T6.VER_NMB = T2.VER_NMB 
    AND T6.BU_NMB = T5.LST_BU_CNSD 
    AND T1.PRGM_CASE_TYPE = 'MA' 
    AND T1.PRGM_CASE_STS IN ('T', 'D') 
    AND T1.PRD_BEG_DT <= :PRD-BEG-DT 
    AND T1.PRD_END_DT >= :PRD-END-DT 
    AND T2.BNFT_MTH = :BNFT-MTH 
    AND T2.BNFT_YR = :BNFT-YR;
