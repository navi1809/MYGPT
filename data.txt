// PROCESS TO READ INPUT FILE
Function AIF20P9-SCAN-INFLE()
  VDBCONTROL.UPROC-NM = "AIF20P9-SCAN-INFLE";
  VDBCONTROL.UPROC-OPT = "SCAN";
  VDBCONTROL.UPROC-OBJ = "SR-AIF20-RECIN";
  VMESSAGE.UMSGINS[2] = "AIF20A";
  try
    get next SR-AIF20-RECIN ;
  end

  /* check if any i/o error has occured*/
  if (SR-AIF20-RECIN is ioError && SR-AIF20-RECIN not endOfFile)
    VMESSAGE.UMSGCODE-JCL-ERR = 0;
    AIF20W.WS-ERR-CD = sysVar.errorCode;
    VMESSAGE.UMSGINS[1] = AIF20W.WS-ERR-LAST-FOUR;
    /* return error code to indicate abnormal ending*/
    WCM00P2-TO-CEP01A();
  end
  if (SR-AIF20-RECIN is endOfFile)
    VMESSAGE.UMSGCODE-JCL-ERR = 0;
    AIF20W.WS-ERR-CD = sysVar.errorCode;
    VMESSAGE.UMSGINS[1] = AIF20W.WS-ERR-LAST-FOUR;
  /* PERFORM WCM00P2-TO-CEP01A;*/
  end
end // end AIF20P9-SCAN-INFLE
