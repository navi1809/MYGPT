Function AIN90P9-BLD-MBR-A()
  /* -----------------------------------------------------------------------*/
  /* This process populates the AIN90W1 Working storage record with*/
  /* information about each member in the case.*/
  /* -----------------------------------------------------------------------*/
  /*  */
  set H2-MEMBER-REL empty;

  H2-MEMBER-REL.PRD-BEG-DT = AIN90W.WS-PREV-BEG-DT;
  H2-MEMBER-REL.PRD-END-DT = AIN90W.WS-PREV-END-DT;
  H2-MEMBER-REL.ICI = H2-2-EVM-EVR.ICI;

  AIN90P9-MRR-Q1(); /* Find HOH UPI*/

  set H2-2-CMH-PER empty;

  H2-2-CMH-PER.PRD-BEG-DT = AIN90W.WS-PREV-BEG-DT;
  H2-2-CMH-PER.PRD-END-DT = AIN90W.WS-PREV-END-DT;
  H2-2-CMH-PER.ICI = H2-2-EVM-EVR.ICI;

  AIN90P9-CMH-PER-Q1(); /* Get all members*/

  AIN90W.WS-MBR-INDEX = 1;

  while (H2-2-CMH-PER not ioError && AIN90W.WS-MBR-INDEX <= 25)
    AIN90P9-CMH-PER-S1(); /* Read the member*/
    if (H2-2-CMH-PER not ioError)

      set H3-MEMBER-REL empty;

      H3-MEMBER-REL.PRD-BEG-DT = AIN90W.WS-PREV-BEG-DT;
      H3-MEMBER-REL.PRD-END-DT = AIN90W.WS-PREV-END-DT;
      H3-MEMBER-REL.ICI = H2-2-EVM-EVR.ICI;
      H3-MEMBER-REL.FROM-UPI = H2-MEMBER-REL.FROM-UPI;
      H3-MEMBER-REL.TO-UPI = H2-2-CMH-PER.UPI;

      AIN90P9-MRR-Q2(); /* Get relationship code*/

      AIN90W1.WS-TANF-CRTKR-IND[WS-MBR-INDEX] = H2-2-CMH-PER.TANF-CRTKR-IND;
      AIN90W1.WS-HH-LEFT-DT[WS-MBR-INDEX] = H2-2-CMH-PER.HH-LEFT-DT;
      AIN90W1.WS-PCN-CD[WS-MBR-INDEX] = H2-2-CMH-PER.PCN-CD;
      AIN90W1.WS-JOIN-HH-DT[WS-MBR-INDEX] = H2-2-CMH-PER.JOIN-HH-DT;
      AIN90W1.WS-FRST-NM[WS-MBR-INDEX] = H2-2-CMH-PER.FRST-NM;
      AIN90W1.WS-LST-NM[WS-MBR-INDEX] = H2-2-CMH-PER.LST-NM;
      AIN90W1.WS-MID-NM[WS-MBR-INDEX] = H2-2-CMH-PER.MID-NM;
      AIN90W1.WS-SEX-CD[WS-MBR-INDEX] = H2-2-CMH-PER.SEX-CD;
      AIN90W1.WS-ETHNC-CD[WS-MBR-INDEX] = H2-2-CMH-PER.ETHNC-CD;
      AIN90W1.WS-SSN[WS-MBR-INDEX] = H2-2-CMH-PER.SSN;
      AIN90W1.WS-DOB[WS-MBR-INDEX] = H2-2-CMH-PER.DOB;
      AIN90W1.WS-UPI[WS-MBR-INDEX] = H2-2-CMH-PER.UPI;
      AIN90W1.WS-AGE-OF-MEMBER[WS-MBR-INDEX] = AIN90W.WS-AGE-OF-MEMBER;
      AIN90W1.WS-RELSH-CD[WS-MBR-INDEX] = H3-MEMBER-REL.RELSH-CD;
      AIN90W1.WS-HM-PHN-NMB[WS-MBR-INDEX] = H2-2-CMH-PER.HM-PHN-NMB;

  /* WI23901 - Add Medicaid Billing Number to file*/
      AIN90W1.WS-MEDIC-BLNG-SQ-NMB[WS-MBR-INDEX] = H2-2-CMH-PER.MEDIC-BLNG-SQ-NMB;

      AIN90W.WS-MBR-INDEX = AIN90W.WS-MBR-INDEX + 1;

    end /* End not err*/
  end /* End not err*/

Function AIN90P9-CHILD()
  /* -----------------------------------------------------------------------*/
  /* This process retrieves information about the Child that is needed*/
  /* to populate the Child working storage record.*/
  /* -----------------------------------------------------------------------*/
  /*  */

  AIN90W.WS-CHILD-INDEX = AIN90W.WS-CHILD-INDEX + 1;

  /* ***** Get Nevada Check Up Flag *******/

  set H3-2-EVM-EVR empty;

  H3-2-EVM-EVR.BNFT-MTH = AIN90W.WS-BNFT-MTH;
  H3-2-EVM-EVR.BNFT-YR = AIN90W.WS-BNFT-YR;
  H3-2-EVM-EVR.UPI = AIN90W1.WS-UPI[WS-MBR-INDEX];

  AIN90P9-EVM-EVR-Q2();

  /* ***** Get Child's Citizenship *******/

  set H2-PERSON-HSTRY empty;

  H2-PERSON-HSTRY.PRD-BEG-DT = AIN90W.WS-PREV-BEG-DT;
  H2-PERSON-HSTRY.PRD-END-DT = AIN90W.WS-PREV-END-DT;
  H2-PERSON-HSTRY.UPI = AIN90W1.WS-UPI[WS-MBR-INDEX];

  AIN90P9-PEH-Q1();

  /* ***** Get Insurance Policy Information *******/

  set H2-2-MIP-PCO empty;

  H2-2-MIP-PCO.PRSNS-COVRD-UPI = AIN90W1.WS-UPI[WS-MBR-INDEX];

  AIN90P9-MIP-PCO-Q1(); /* Get all insurance policies*/

  AIN90W3.WS-CH-INS[WS-CHILD-INDEX] = "N";
  AIN90W3.WS-MORE-INSURANCE[WS-CHILD-INDEX] = "N";

  while (H2-2-MIP-PCO not ioError)
    AIN90P9-MIP-PCO-S1(); /* Read the case*/
    if (H2-2-MIP-PCO not ioError)

      if (VDBCONTROL.UROWR == 1) /* Child has insurance flag*/
        AIN90W3.WS-CH-INS[WS-CHILD-INDEX] = "Y";

        /* ** Get Insurance Company Information *******/

        set H2-INSURANCE-CO-CD empty;

        H2-INSURANCE-CO-CD.INSRNC-CD = H2-2-MIP-PCO.INSU-CMPNY-CD;

        AIN90P9-ICO-Q1();

        AIN90P9-MOVE-CHINS(); /* Move child insurance info to outrec*/

      else /* Child has more than 1 insurance*/

        AIN90W3.WS-MORE-INSURANCE[WS-CHILD-INDEX] = "Y";

      end /* End not err*/
    end /* End not err*/
  end /* End While*/

  /* **** Get child inactive info (left household dt) ******/

  if (AIN90W1.WS-HH-LEFT-DT[WS-MBR-INDEX] != "9999-12-31")

    AIN90W3.WS-CH-INACTIVE[WS-CHILD-INDEX] = "Y";

  else

    AIN90W3.WS-CH-INACTIVE[WS-CHILD-INDEX] = "N";

  end

  /* ***** Get PREGNANCY details *******/

  set H2-PRGNCY-DTLS empty;
  AIN90W.WS-PREG-DT = " ";

  if (AIN90W1.WS-SEX-CD[WS-MBR-INDEX] == "F") /* perform if member is female*/

    H2-PRGNCY-DTLS.PRD-BEG-DT = AIN90W.WS-PREV-BEG-DT;
    H2-PRGNCY-DTLS.PRD-END-DT = AIN90W.WS-PREV-END-DT;
    H2-PRGNCY-DTLS.UPI = AIN90W1.WS-UPI[WS-MBR-INDEX];

    AIN90P9-PGD-Q1();

    AIN90W.WS-PREG-DT = H2-PRGNCY-DTLS.DUE-DT;

  end

  AIN90P9-FRMT-CHILD(); /* Format CHILD working storage*/

  AIN90P9-OTHER-INC(); /* Process other income info*/
end // end AIN90P9-CHILD

Function AIN90P9-DO-GRDIAN()
  /* -----------------------------------------------------------------------*/
  /* The following process determines who in the AIN90W1 Array is the*/
  /* Guardian.  Once the Guardian is identified based on the following*/
  /* hiearchy a "G" is moved to their member type field.The process then*/
  /* goes on to build the working storage areas for the guardian,*/
  /* employment and other income.*/
  /* -----------------------------------------------------------------------*/
  /*  */


  if ("N" in AIN90W1.WS-TANF-CRTKR-IND) /* non-needy caretaker (con)*/
    AIN90W1.WS-MEMBER-TYPE[sysVar.arrayIndex] = "G";
  else

    if ("K" in AIN90W1.WS-TANF-CRTKR-IND) /* non-needy caretaker (cok)*/
      AIN90W1.WS-MEMBER-TYPE[sysVar.arrayIndex] = "G";
    else

      if ("C" in AIN90W1.WS-TANF-CRTKR-IND) /* caretaker*/
        AIN90W1.WS-MEMBER-TYPE[sysVar.arrayIndex] = "G";
      else

        if ("1" in AIN90W1.WS-TANF-CRTKR-IND) /* caretaker (sh1)*/
          AIN90W1.WS-MEMBER-TYPE[sysVar.arrayIndex] = "G";
        else

          if ("2" in AIN90W1.WS-TANF-CRTKR-IND) /* caretaker (sh2)*/
            AIN90W1.WS-MEMBER-TYPE[sysVar.arrayIndex] = "G";
          else

            if ("01" in AIN90W1.WS-RELSH-CD &&  /* head of household*/
            AIN90W1.WS-AGE-OF-MEMBER[sysVar.arrayIndex] >= 180000)
              AIN90W1.WS-MEMBER-TYPE[sysVar.arrayIndex] = "G";
            else

              if ("04" in AIN90W1.WS-RELSH-CD) /* parent of applicant*/
                AIN90W1.WS-MEMBER-TYPE[sysVar.arrayIndex] = "G";
              else

                if ("19" in AIN90W1.WS-RELSH-CD) /* othr specfd adult rltve*/
                  AIN90W1.WS-MEMBER-TYPE[sysVar.arrayIndex] = "G";
                else

                  if ("30" in AIN90W1.WS-RELSH-CD) /* step parent*/
                    AIN90W1.WS-MEMBER-TYPE[sysVar.arrayIndex] = "G";
                  else

                    if ("08" in AIN90W1.WS-RELSH-CD &&  /* sibling*/
                    AIN90W1.WS-AGE-OF-MEMBER[sysVar.arrayIndex] >= 180000)
                      AIN90W1.WS-MEMBER-TYPE[sysVar.arrayIndex] = "G";
                    else

                      if ("18" in AIN90W1.WS-RELSH-CD &&  /* step bro/sis*/
                      AIN90W1.WS-AGE-OF-MEMBER[sysVar.arrayIndex] >= 180000)
                        AIN90W1.WS-MEMBER-TYPE[sysVar.arrayIndex] = "G";
                      else

                        if ("09" in AIN90W1.WS-RELSH-CD &&  /* 1st/2nd csn*/
                        AIN90W1.WS-AGE-OF-MEMBER[sysVar.arrayIndex] >= 180000)
                          AIN90W1.WS-MEMBER-TYPE[sysVar.arrayIndex] = "G";
                        else

                          if ("10" in AIN90W1.WS-RELSH-CD &&  /* other reltiv*/
                          AIN90W1.WS-AGE-OF-MEMBER[sysVar.arrayIndex] >= 180000)
                            AIN90W1.WS-MEMBER-TYPE[sysVar.arrayIndex] = "G";
                          else

                            if ("12" in AIN90W1.WS-RELSH-CD &&  /* friend*/
                            AIN90W1.WS-AGE-OF-MEMBER[sysVar.arrayIndex] >= 180000)
                              AIN90W1.WS-MEMBER-TYPE[sysVar.arrayIndex] = "G";
                            else

                              if ("21" in AIN90W1.WS-RELSH-CD && 
                               /* req mbr not hoh*/
                              AIN90W1.WS-AGE-OF-MEMBER[sysVar.arrayIndex] >= 180000)
                                AIN90W1.WS-MEMBER-TYPE[sysVar.arrayIndex] = "G";
                              else

                              end /* end IF '21'*/
                            end /* end IF '12'*/
                          end /* end IF '10'*/
                        end /* end IF '09'*/
                      end /* end IF '18'*/
                    end /* end IF '08'*/
                  end /* end IF '30'*/
                end /* end IF '19'*/
              end /* end IF '04'*/
            end /* end IF '01'*/
          end /* end IF '2'*/
        end /* end IF '1'*/
      end /* end IF 'c'*/
    end /* end IF 'k'*/
  end /* end IF 'n'*/

  /* WI 23901 -This IF was moved here due to if no one is over 18 in the*/
  /* household have the hoh be guardian even if under 18.*/
  if ("G" in AIN90W1.WS-MEMBER-TYPE) /* do nothing*/
  else
    if ("01" in AIN90W1.WS-RELSH-CD) /* head of household*/
      AIN90W1.WS-MEMBER-TYPE[sysVar.arrayIndex] = "G";
    end
  end

  /* Process guardian*/

  if ("G" in AIN90W1.WS-MEMBER-TYPE)
    AIN90W.WS-MBR-INDEX = sysVar.arrayIndex;
  end

  AIN90P9-GUARDIAN(); /* Build guardian record*/

  AIN90P9-EMPLOYMENT(); /* Build employment income records*/

  AIN90P9-OTHER-INC(); /* Build other income records*/
end // end AIN90P9-DO-GRDIAN

// Control Retrieve of Data
Function AIN90P9-GENASS-INC()
  set H2-GEN-ASST-INC-DT empty;
  /* Move retrieve key value to SQL row record*/
  H2-GEN-ASST-INC-DT.UPI = AIN90W1.WS-UPI[WS-MBR-INDEX];
  H2-GEN-ASST-INC-DT.PRD-BEG-DT = AIN90W.WS-PREV-BEG-DT;
  H2-GEN-ASST-INC-DT.PRD-END-DT = AIN90W.WS-PREV-END-DT;
  /*  */
  /* Perform process to retrieve data*/
  AIN90P9-GAI-Q1();
  /*  */
  /* If data retrieved successfully, continue*/
  while (sysVar.sqlData.sqlcode == 0)

    AIN90P9-GAI-QS1();

    if (sysVar.sqlData.sqlcode == 0)

      AIN90W.WS-FREQ = H2-GEN-ASST-INC-DT.GEN-ASIST-FREQ;
      AIN90P9-GET-FACTOR();

      AIN90W.WS-INC-AMOUNT = AIN90W.WS-GAGA-AMT       + (H2-GEN-ASST-INC-DT.GEN-ASIST-INC-AMT * AIN90W.WS-FACTOR);

      AIN90W.WS-TOTAL-CASE-INCOME = AIN90W.WS-TOTAL-CASE-INCOME +       AIN90W.WS-INC-AMOUNT;

      AIN90P9-GET-FREQ(); /* Formate NCU pay frequency*/
      AIN90W.WS-INC-TYPE-CD = 95;
      AIN90P9-FRMT-OINC(); /* Build other income file*/

    end
  end
end // end AIN90P9-GENASS-INC


// FINDS FACTOR OF A FREQUENCY
Function AIN90P9-GET-FACTOR()
  /* ************************************************************************/
  /* This process finds out the factor associated with a frequency. The*/
  /* input to this process is WS-FREQ and the output is WS-FACTOR*/
  /* ************************************************************************/

  if (AIN90W.WS-FREQ == "W")
    AIN90W.WS-FACTOR = AIN90W.WS-WEEKLY-FACTOR;
    return;
  end

  if (AIN90W.WS-FREQ == "B")
    AIN90W.WS-FACTOR = AIN90W.WS-BIWEEKLY-FACTOR;
    return;
  end

  if (AIN90W.WS-FREQ == "S")
    AIN90W.WS-FACTOR = 2;
    return;
  end

  if (AIN90W.WS-FREQ == "M")
    AIN90W.WS-FACTOR = 1;
    return;
  end

  if (AIN90W.WS-FREQ == "Q")
    AIN90W.WS-FACTOR = mathLib.round(1 / 3, -mathLib.decimals(AIN90W.WS-FACTOR));
    return;
  end

  if (AIN90W.WS-FREQ == "T")
    AIN90W.WS-FACTOR = mathLib.round(1 / 6, -mathLib.decimals(AIN90W.WS-FACTOR));
    return;
  end

  if (AIN90W.WS-FREQ == "A")
    AIN90W.WS-FACTOR = mathLib.round(1 / 12, -mathLib.decimals(AIN90W.WS-FACTOR));
    return;
  end
end // end AIN90P9-GET-FACTOR


// FINDS FACTOR OF A FREQUENCY
Function AIN90P9-GET-FREQ()
  /* ************************************************************************/
  /* This process finds out the factor associated with a frequency. The*/
  /* input to this process is WS-FREQ and the output is WS-FACTOR*/
  /* ************************************************************************/

  if (AIN90W.WS-FREQ == "W")
    AIN90W.WS-NCU-FREQ = 52;
    return;
  end

  if (AIN90W.WS-FREQ == "B")
    AIN90W.WS-NCU-FREQ = 26;
    return;
  end

  if (AIN90W.WS-FREQ == "S")
    AIN90W.WS-NCU-FREQ = 24;
    return;
  end

  if (AIN90W.WS-FREQ == "M")
    AIN90W.WS-NCU-FREQ = 12;
    return;
  end

  if (AIN90W.WS-FREQ == "Q")
    AIN90W.WS-NCU-FREQ = 4;
    return;
  end

  if (AIN90W.WS-FREQ == "T")
    AIN90W.WS-NCU-FREQ = 2;
    return;
  end

  if (AIN90W.WS-FREQ == "A")
    AIN90W.WS-NCU-FREQ = 1;
    return;
  end
end // end AIN90P9-GET-FREQ

// Control Retrieve of Data
Function AIN90P9-JOB-INC()
  set H2-JOB-INCM-DTLS empty;
  /* Move retrieve key value to SQL row record*/
  H2-JOB-INCM-DTLS.UPI = AIN90W1.WS-UPI[WS-MBR-INDEX];
  H2-JOB-INCM-DTLS.PRD-BEG-DT = AIN90W.WS-PREV-BEG-DT;
  H2-JOB-INCM-DTLS.PRD-END-DT = AIN90W.WS-PREV-END-DT;

  /* Perform process to retrieve data*/
  AIN90P9-JID-Q1();

  AIN90W.WS-VDBCONTROL-URC = sysVar.sqlData.sqlcode;
  /*  */
  /* If data retrieved successfully, continue*/

  while (AIN90W.WS-VDBCONTROL-URC == 0) /* while row exists*/

    AIN90P9-JID-QS1(); /* ******FIRST JOB SCAN************/

    AIN90W.WS-VDBCONTROL-URC = sysVar.sqlData.sqlcode;

    if (AIN90W.WS-VDBCONTROL-URC == 0)

      set H2-EMPLOYER empty;

      H2-EMPLOYER.EMPLR-ID = H2-JOB-INCM-DTLS.EMPLR-ID;

      AIN90P9-EMP-Q1(); /* ******RETRIEVE EMPLOYER INFO**********/

      /* *** Beg Comment ****/
      /* Read TWNPAY_DTLS table to find out wage information for*/
      /* the member*/
      /* *** End Comment ****/

      /* ********* Inner query starts ******************************/
      AIN90P9-PYD-INC();
      if (VDBCONTROL.URC > 104)
        exit program;
      end
  /* IF FACTOR is 'Y' (YES), retrieve the factor depending on the frequency*/
  /* else do not apply factor.*/

      if (H2-JOB-INCM-DTLS.OVRD-433-IND == "Y")

     /* *** Beg Comment ****/
     /* consider gross wages and tips and commission received as job income*/
     /* No factoring is to be done*/
     /* *** End Comment ****/

        AIN90W.WS-JINC-AMT = AIN90W.WS-GRS-WGS-AMT + AIN90W.WS-TIPS-COMSN-AMT;

        AIN90W.WS-PAY-FREQ = H2-JOB-INCM-DTLS.PAY-FREQ;

      else /* If H2-JOB-INCM-DTLS.OVRD-433-IND =*/
                                   /* 'Y'*/

  /* Depending on frequency, find factor from the system table. for this*/
  /* perform get-factor process. Input to this process is frequency*/
  /* (WS-FREQ) and the output is the corresponding factor (WS-FACTOR).*/

  /* Get factor for wages depending on frequency.*/

        AIN90W.WS-FREQ = H2-JOB-INCM-DTLS.PAY-FREQ;
        AIN90W.WS-PAY-FREQ = H2-JOB-INCM-DTLS.PAY-FREQ;
        AIN90P9-GET-FACTOR();

        AIN90W.WS-WGS-FACTOR = AIN90W.WS-FACTOR;

  /* GET FACTOR FOR TIPS/COMMISSION DEPENDING ON IT'S FREQUENCY.*/

        AIN90W.WS-FREQ = H2-JOB-INCM-DTLS.TIPS-COMSN-FREQ;
        AIN90W.WS-FACTOR = 0;
        AIN90P9-GET-FACTOR();
        AIN90W.WS-TIPS-FACTOR = AIN90W.WS-FACTOR;

  /* COMPUTE AVERAGE GROSS WAGE AMOUNT, HOURS WORKED AND TIPS/COMMISSION*/

  /* AMOUNT FOR THE MONTH.*/

        if (AIN90W.WS-CHECK-NMB != 0)
          AIN90W.WS-AVG-GRS-WGS-AMT = AIN90W.WS-GRS-WGS-AMT /           AIN90W.WS-CHECK-NMB;
          AIN90W.WS-AVG-TIPS-COMSN-AMT = AIN90W.WS-TIPS-COMSN-AMT /           AIN90W.WS-CHECK-NMB;
          AIN90W.WS-JINC-AMT = (AIN90W.WS-AVG-GRS-WGS-AMT           * AIN90W.WS-WGS-FACTOR) + (AIN90W.WS-AVG-TIPS-COMSN-AMT *           AIN90W.WS-TIPS-FACTOR);
        else
          AIN90W.WS-JINC-AMT = (AIN90W.WS-GRS-WGS-AMT           * AIN90W.WS-WGS-FACTOR) + (AIN90W.WS-TIPS-COMSN-AMT *           AIN90W.WS-TIPS-FACTOR);
        end /* If ws-check-nmb ne 0*/
      end /* If H2-JOB-INCM-DTLS.OVRD-433-IND = y*/

      AIN90W.WS-TOTAL-CASE-INCOME = AIN90W.WS-TOTAL-CASE-INCOME +       AIN90W.WS-JINC-AMT;

      AIN90W.WS-FREQ = AIN90W.WS-PAY-FREQ;

      AIN90P9-GET-FREQ(); /* Format NCU pay frequency*/
      AIN90P9-FRMT-EMP(); /* Format employment file*/

    end
  end /* End of while loop for*/
end // end AIN90P9-JOB-INC

Function AIN90P9-LS-INC-TYP()
  /* This process is to convert lump sum income types for NCU*/

  if ("CM" in H2-LSUM-RSRCE-DTLS.LUMP-SUM-RSRCE-TYP) /* Commission*/
    AIN90W.WS-INC-TYPE-CD = 80;
  else
    if ("CR" in H2-LSUM-RSRCE-DTLS.LUMP-SUM-RSRCE-TYP) /* Rebate/Cred/Refund*/
      AIN90W.WS-INC-TYPE-CD = 81;
    else
      if ("EB" in H2-LSUM-RSRCE-DTLS.LUMP-SUM-RSRCE-TYP) /* Emply Bonus*/
        AIN90W.WS-INC-TYPE-CD = 82;
      else
        if ("GW" in H2-LSUM-RSRCE-DTLS.LUMP-SUM-RSRCE-TYP) /* Gamble/Win*/
          AIN90W.WS-INC-TYPE-CD = 83;
        else
          if ("IH" in H2-LSUM-RSRCE-DTLS.LUMP-SUM-RSRCE-TYP) /* Inheritance*/
            AIN90W.WS-INC-TYPE-CD = 84;
          else
            if ("IN" in H2-LSUM-RSRCE-DTLS.LUMP-SUM-RSRCE-TYP) /* Incentive*/
              AIN90W.WS-INC-TYPE-CD = 85;
            else
              if ("IS" in H2-LSUM-RSRCE-DTLS.LUMP-SUM-RSRCE-TYP)
                                   /* Ins settle*/
                AIN90W.WS-INC-TYPE-CD = 86;
              else
                if ("OI" in H2-LSUM-RSRCE-DTLS.LUMP-SUM-RSRCE-TYP)
                                   /* Other LSUM*/
                  AIN90W.WS-INC-TYPE-CD = 87;
                else
                  if ("RP" in H2-LSUM-RSRCE-DTLS.LUMP-SUM-RSRCE-TYP)
                                   /* Retro bene/wage*/
                    AIN90W.WS-INC-TYPE-CD = 88;
                  else
                    if ("SA" in H2-LSUM-RSRCE-DTLS.LUMP-SUM-RSRCE-TYP)
                                   /* RSDI retro pymt*/
                      AIN90W.WS-INC-TYPE-CD = 89;
                    else
                      if ("SI" in H2-LSUM-RSRCE-DTLS.LUMP-SUM-RSRCE-TYP)
                                   /* SSI retro pymt*/
                        AIN90W.WS-INC-TYPE-CD = 90;
                      else
                        if ("SP" in H2-LSUM-RSRCE-DTLS.LUMP-SUM-RSRCE-TYP)
                                   /* Sale property*/
                          AIN90W.WS-INC-TYPE-CD = 91;
                        else
                        end
                      end
                    end
                  end
                end
              end
            end
          end
        end
      end
    end
  end
end // end AIN90P9-LS-INC-TYP

Function AIN90P9-LSUM-INC()
  set H2-LSUM-RSRCE-DTLS empty;
  /* *** Beg Comment ****/
  /* This process calculates the total lumpsum income for the member.*/
  /* Read TWNLSUM_RSRCE_DTLS table for member's lumpsum income rows which*/
  /* have receipt date falling in the benefit month.  The following*/
  /* lumpsum resources are considered as lumpsum earned income:*/
  /*  */
  /* EB - EMPLOYMENT BONUS*/
  /* IN - INCENTIVE*/
  /* RP - RETRO BENEFITS / WAGES*/
  /* CM - COMMISSIONS*/
  /* IS - INSURANCE SETTLEMENTS*/
  /* CR - CREDITS/REBATES/REFUNDS*/
  /* IO - OTHER LUMP SUM INCOME*/
  /* IH - Inheritence*/
  /* GW - GAMBLING/WINNIGS*/
  /* SA - RSDI RETRO PAYMENTS*/
  /* SI - SSI RETRO PAYMENTS*/
  /* SP - SALE OF PROPERTY*/
  /*  */
  /* Key setting*/
  H2-LSUM-RSRCE-DTLS.UPI = AIN90W1.WS-UPI[WS-MBR-INDEX];
  H2-LSUM-RSRCE-DTLS.PRD-BEG-DT = AIN90W.WS-PREV-BEG-DT;
  H2-LSUM-RSRCE-DTLS.PRD-END-DT = AIN90W.WS-PREV-END-DT;
  /*  */
  /* Perform process to retrieve data*/
  AIN90P9-LRD-Q1();
  /*  */
  while (sysVar.sqlData.sqlcode == 0)

    AIN90P9-LRD-QS1();

    if (sysVar.sqlData.sqlcode == 0)
      /* If data retrieved successfully, continue*/

      /* *** Beg Comment ****/
      /* Calculate the total LSUM income.*/
      /* *** End Comment ****/

      AIN90W.WS-INC-AMOUNT = H2-LSUM-RSRCE-DTLS.LUMP-SUM-RSRCE-AMT;

      AIN90W.WS-TOTAL-CASE-INCOME = AIN90W.WS-TOTAL-CASE-INCOME +   AIN90W.WS-INC-AMOUNT;

      AIN90W.WS-NCU-FREQ = 1;
      AIN90P9-LS-INC-TYP(); /* Format income type*/
      AIN90P9-FRMT-OINC(); /* Build other income file*/

    end
  end
end // end AIN90P9-LSUM-INC


// MAIN PROCESS
Function AIN90P9-MAIN()
  /* ----------------------------------------------------------------------*/
  /* This application is designed to retrieve all cases that have at*/
  /* least one member who has been marked as possibly eligible for Nevada*/
  /* Check Up. The member is marked by the TANF eligiblity process and is*/
  /* stored in table TWNELG_VER_MBR as NV_CHK_UP.*/
  /*  */
  /* Each member of the case is identified as one of the following:*/
  /*  */
  /* 1. Guardian    2. Child     3. Other Adult*/
  /*  */
  /* Information about each member is retrieved and placed into working*/
  /* storage W1.  Five working storage areas are populated and model the*/
  /* five output files required for Nevada Check Up.*/
  /*  */
  /* The five working storage areas are defined as follows:*/
  /*  */
  /* AIN90W2 - Other Adult information.*/
  /* AIN90W3 - Child information.*/
  /* AIN90W4 - Employment Information.*/
  /* AIN90W5 - Guardian Information.*/
  /* AIN90W6 - Other Income Information.*/
  /*  */
  /*  */
  /* The five output files are defined as follows:*/
  /*  */
  /* SR-AIN90-ADULT-OUT - Other Adult Information.*/
  /* SR-AIN90-CHILD-OUT - Child information.*/
  /* SR-AIN90-EMPLR-OUT - Employment Information.*/
  /* SR-AIN90-GUARD-OUT - Guardian information.*/
  /* SR-AIN90-OINCM-OUT - Other Income Information.*/
  /*  */
  /* After all working storage has been populated the application checks to*/
  /* see if the case is under the 200% poverty income limit.  If the case*/
  /* is found to be under the cap the application will move working storage*/
  /* to the SR-OUT records and write to the files.*/
  /*  */
  /* -----------------------------------------------------------------------*/
  /*  */
  AIN90P9-INITIALIZE(); /* Initialize everything*/

  AIN90P9-FRMT-DATES(); /* Format Dates*/

  /* Set Key*/
  H2-2-EVM-EVR.BNFT-MTH = AIN90W.WS-BNFT-MTH;
  H2-2-EVM-EVR.BNFT-YR = AIN90W.WS-BNFT-YR;


  /* WI 23901 - Change made to backbone cursor to check if child is elig*/
  /* in a future month. If so, do not process member.*/
  AIN90P9-EVM-EVR-Q1(); /* Set backbone cursor*/
                                   /* Get all cases with a member elig NCU*/

  while (H2-2-EVM-EVR not ioError)
    AIN90P9-EVM-EVR-S1(); /* Read the case*/
    if (H2-2-EVM-EVR not ioError)

      AIN90P9-RESET-CASE(); /* Reset Records and indexes for next*/
                                   /* case*/

      AIN90P9-DO-CASE(); /* Process each case*/

      AIN90P9-FED-LIMIT(); /* Get 200% Poverty amount*/

  /* Compare the total case income to the 200% Poverty limit if under the*/
  /* limit build the output files.*/

      if (AIN90W.WS-TOTAL-CASE-INCOME <= 
      AIN90W.WS-200-PERCENT-POVERTY-LEVEL)

        AIN90P9-BLD-FILES(); /* Write to the five output files*/

      end

    end /* End not err*/
  end /* End not err*/

  VGVar.handleSysLibraryErrors = 0;
  sysLib.commit();

  /* END OF PROGRAM*/
end // end AIN90P9-MAIN
