var titles = [];
var time_opened = 0;
$(document).ready(function() {
	$('.hover-help').hover(function(a) {
		titles = [];
		$('table[title]').each(function(index) {
			titles.push([ index, $(this).attr('title') ]);
		});
		$('table[title]').attr('title', '');
		$(this).css({"z-index":"1000"}).find('.hover-help-msg').show(100);
		start_secds();
	}, function(a) {
		$(this).css({"z-index":"auto"}).find('.hover-help-msg').hide();
		$('table[title]').each(function(index) {
			$(this).attr('title', titles[index][1]);
		});
		if(!$(this).parent().find('.hover-help-msg').is(":visible") && time_opened > 0)
		{
		//Was requested.
		var hidden_count = $(this).parent().find('.hit_count');
		var secds_accesd = $(this).parent().find('.secds_accesd');
		set_hit_count(hidden_count);
		set_secds(secds_accesd);
		}
	});
		
	$('.hover-help a').click(function(){
		$(this).parent().css({"z-index" : "1000"});
		$(this).parent().find('.hover-help-msg').toggle(100,function(){
			//Was requested.
			if(!$(this).is(":visible"))
			{
				var hidden_count = $(this).parent().find('.hit_count');
				var secds_accesd = $(this).parent().find('.secds_accesd');
				set_hit_count(hidden_count);
				set_secds(secds_accesd);
			}else{
				start_secds();
			}			
		});
	});
	
	$('.hover-help a').blur(function(){
		//Was requested.
		if($(this).parent().find('.hover-help-msg').is(":visible"))
		{
			var hidden_count = $(this).parent().find('.hit_count');
			var secds_accesd = $(this).parent().find('.secds_accesd');
			set_hit_count(hidden_count);
			set_secds(secds_accesd);
		}	
		$(this).parent().css({"z-index":"auto"}).find('.hover-help-msg').hide();
	});
	
	
});


function set_hit_count(input_field){
	var value = $(input_field).val();
	$(input_field).val(++value);
}

function set_secds(input_field){
	var value = $(input_field).val();
	var date = new Date();
	if(time_opened && time_opened > 0){
		var diff_mls = (date.getTime() - time_opened) / 1000;
		time_opened = 0;
		$(input_field).val(parseInt(value) + diff_mls);
	}
}

function start_secds(){
	var date = new Date();
	time_opened = date.getTime();
}

const modal = document.getElementById("abwacModal");
const timerDisplay = document.getElementById('timerDisplay');
const resetBtn = document.getElementById('resetBtn');
const endSessionBtn = document.getElementById('endSessionBtn');
const returnBtn = document.getElementById('returnBtn');

const fourthText=document.getElementById('text4');

var inactivityTime = 1680000; //28 mins 1680000
var attempts = 0;
var timeout;
let timeLeft = 120;
let countdownInterval;
var submitted=false;

function startCountdown() {
    countdownInterval = setInterval(() => {
        timeLeft--;
        timerDisplay.textContent = `${timerTextStart} ${timeLeft} ${timerTextEnd}`;
        if (timeLeft <= 0) {
            clearInterval(countdownInterval);
            if(submitted){
            	window.location.href ='/access/accessLogout?VALID_GET=Y&amp;MyAccessLogout=Y';
            }
            else{
            	window.location.href = '/access/jsp/framework/sessionTimeOut.jsp';
            }
        }
    }, 1000);
}

// Reset the timer
resetBtn.addEventListener('click', () => {
	modal.style.display = "none";
    clearInterval(countdownInterval);
    resetTimer();
});

// Redirect on button click
endSessionBtn.addEventListener('click', () => {
    window.location.href = '/access/jsp/framework/sessionTimeOut.jsp';
});

//Return to application on button click
returnBtn.addEventListener('click', () => {
	modal.style.display = "none";
});


// Show modal after inactivity
function showModal() {
  modal.style.display = 'block';
  attempts++;
  timeLeft = 120;
  if(attempts === 2){
	  fourthText.innerHTML=fourthTextSecondTime;
  }
  if(attempts === 3){
	  resetBtn.style.display = 'none';
	  endSessionBtn.style.display = 'none';
	  returnBtn.style.display = 'block';
	  firstText.textContent=firstTextThirdTime;
	  fourthText.textContent=fourthTextThirdTime;
  }
  clearInterval(countdownInterval);
  timerDisplay.textContent = `${timerTextStart} ${timeLeft} ${timerTextEnd}`;
  startCountdown();
}

// Reset inactivity timer
function resetTimer() {
  clearTimeout(timeout);
  timeout = setTimeout(showModal, inactivityTime);
}

// Track activity: reset the timer whenever user interacts with the page
function activityDetected() {
  resetTimer();
}

// Listen for user activity (mouse movements, clicks, keypresses, scrolling)
window.onload = resetTimer;
window.onmousemove = activityDetected;
window.onkeypress = activityDetected;
window.onscroll = activityDetected;
window.onclick = activityDetected;
