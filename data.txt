	public void getVcrButtonStatus() {
		if (voidSw != null && voidSw.booleanValue() == true) {
			disableVcrPrevious = "N";
			disableVcrLast = "N";
		} else {
			if (histNavInd == null) {
				histNavInd = "";
			} else {
				histNavInd = histNavInd.trim();
			}

			debug("histNavInd " + histNavInd);
			if (histNavInd.equals("")) {
				disableVcrPrevious = "Y";
				disableVcrLast = "Y";
			} else
				if (histNavInd.equalsIgnoreCase("S")) {
					disableVcrPrevious = "Y";
					disableVcrLast = "Y";
				} else
					if (histNavInd.equalsIgnoreCase("F")) {
						disableVcrPrevious = "Y";
						disableVcrLast = "N";
					} else
						if (histNavInd.equalsIgnoreCase("P")) {
							disableVcrPrevious = "N";
							disableVcrLast = "Y";
						} else
							if (histNavInd.equalsIgnoreCase("M")) {
								disableVcrPrevious = "N";
								disableVcrLast = "N";
							}
		}

		debug("histNavInd " + histNavInd);
		debug("lastStatus " + disableVcrLast);
		debug("firstStatus " + disableVcrPrevious);

	}
