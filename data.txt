package gov.illinois.ies.presentation.view.pageelements;

public class NameGroup implements IPageElement {
  public static String WIDTH = "0";
  
  public static String MANDATORY = "";
  
  public static String LABEL = "Name Group";
  
  public static String NAME = "nameGroup";
  
  public String disabled = "";
  
  public static String FIELD_NAME = "nameGroup";
  
  public String getWIDTH() {
    return WIDTH;
  }
  
  public String getMANDATORY() {
    return MANDATORY;
  }
  
  public void setWIDTH(String ignored) {}
  
  public void setMANDATORY(String ignored) {}
  
  public void setLABEL(String ignored) {}
  
  public String getLABEL() {
    return LABEL;
  }
  
  public void setDisabled(String newDisabled) {
    this.disabled = newDisabled;
  }
  
  public String getDisabled() {
    return this.disabled;
  }
  
  public void setFIELD_NAME(String ignored) {}
  
  public String getFIELD_NAME() {
    return FIELD_NAME;
  }
}
