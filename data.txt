// Overlay preserved joint filing answers onto taxIndvColl before validation
Map preservedJoint = (Map) request.get("PRESERVED_TAX_JOINTLY");
if (taxIndvColl != null && preservedJoint != null) {
    for (APP_TAX_RLT_Cargo t : taxIndvColl.getResults()) {
        String srcSeq = t.getSrc_indv_seq_num();
        if (srcSeq != null) {
            String v = (String) preservedJoint.get(srcSeq);
            if (v != null && v.trim().length() > 0) {
                t.setTax_jointly_filed_sw(v.trim());
            }
        }
    }
}
