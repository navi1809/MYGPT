  SELECT

    T4.OFC_CD
  , T2.UPI
  , T2.SSN
  , T2.DEATH_DT
  , case when T2.DEATH_DT != '9999-12-31'  -- This date has 9999-12-13 in it
         THEN T2.DEATH_DT end as T2Death
  , T1.MEDIC_BLNG_SQ_NMB AS "MEDICAID ID"
  , T1.MEDCR_CLM_NMB
  , T1.T_LST_NM AS "LAST NAME"
  , T1.T_FRST_NM AS "FIRST NAME"
  , T1.T_DEATH_DT AS "MEDICARE DEATH DATE"

FROM {?P_SCHEMA}.TWNBUYIN_MDCR_EXT T1
   , {?P_SCHEMA}.TWNPERSON T2
   , {?P_SCHEMA}.TWNFAM_CASE T3
    ,{?P_SCHEMA}. TWNFAM_CASELD_CASE T4

WHERE T1.T_DEATH_DT <> '0001-01-01'
  AND T1.B_DEATH_DT = '0001-01-01'
  AND T1.MEDIC_BLNG_SQ_NMB = T2.MEDIC_BLNG_SQ_NMB
  AND T1.T_SSN = T2.SSN
  AND T2.UPI = T3.HOH_UPI
  AND T2.DEATH_DT = '9999-12-31'
  AND T3.ICI = T4.ICI

ORDER BY T4.OFC_CD, T2.UPI
WITH UR;
