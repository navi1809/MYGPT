// OBTAIN FIRST
Function CIN01P9-READIP()
  /*  */
  /* NOMADS BEGIN*/
  /*  */
  CIN01W2.WS-REC-IND = "N";
  while (SR-QCIN01-REC not endOfFile)
    try
      get next SR-QCIN01-REC ;
    end
    if (SR-QCIN01-REC is ioError && SR-QCIN01-REC not endOfFile)
      VDBCONTROL.UMSQLCODE = 0;
      VDBCONTROL.UPROC-OBJ = "SR-QCIN01-REC";
      VDBCONTROL.UPROC-NM = "CIN01P9-READ-PARM";
      VDBCONTROL.UPROC-OPT = "SCAN";
      CIN01W2.WS-ERR-CD = sysVar.errorCode;
      VMESSAGE.UMSGINS[1] = CIN01W2.WS-ERR-LAST-FOUR;
      VMESSAGE.UMSGINS[2] = "CIN01A";
      VDBCONTROL.UEZESQRRM = " ";
      if (SR-QCIN01-REC is invalidFormat)
        VDBCONTROL.UEZESQRRM = "FILE FORMAT MISMATCH";
      end
      WCM00P2-TO-CEP01A();
    end

    if (SR-QCIN01-REC is endOfFile && CIN01W2.WS-REC-IND == "N")
      VDBCONTROL.UMSQLCODE = 0;
      VDBCONTROL.UPROC-OBJ = "SR-QCIN01-REC";
      VDBCONTROL.UPROC-NM = "CIN01P9-READ-PARM";
      VDBCONTROL.UPROC-OPT = "SCAN";
      VMESSAGE.UMSGINS[1] = " ";
      VMESSAGE.UMSGINS[2] = "CIN01A";
      VMESSAGE.UMSGCODE = "GEN0274";
      VDBCONTROL.UEZESQRRM = "NO DATA FOUND";
      VMESSAGE.UMSGCODE-JCL-ERR = 8;
      WCM00P2-TO-CEP01A();
    end

    if (sysVar.errorCode == "00000000")
      CIN01W2.WS-REC-IND = "Y";
      CIN01W.WS-DATA-REC = " ";
      CIN01W.WS-DATA-REC = SR-QCIN01-REC.SR-DATA-RECORD;
      if (CIN01W.WS-REC-ID == "E1")
        CIN01W1.WS-EMP-DOH = CIN01W.WS-EMP-DOB;
      else
        if (CIN01W.WS-REC-ID == "W4")
          CIN01P9-APPEND();
        end
      end
    end
  end
  CIN01P9-CLOSE-FILE();
  /*  */
  /* NOMADS END*/
  /*  */
end // end CIN01P9-READIP
