function submitSecDetails(screenId, actionUpdate) {
	$("#messageContainer").html("");
	if (!checkAllRequiredFields()) {
		return true;
	}
	var URL = '';
	var jsonVar = securityMenuConstructMap[screenId];
	if (actionUpdate == 'true') {
		URL = jsonVar.updateDetailsEndPoint;
	} else {
		URL = jsonVar.insertDetailsEndPoint;
	}
	var formID = jsonVar.formID;

	showWait();

	$.ajax({
		type : "POST",
		url : URL,
		data : $("#" + formID).serializeArray(),

		success : function(data, status, xhr) {

			var ct = xhr.getResponseHeader("content-type") || "";

			if (ct.indexOf('json') > -1) {
				updateScreenForJSONResponse(data, status, xhr,'messageContainer', '', false);
				$("#messageContainer").show();
			}else{
				var radioValue = $("input[name='selectRecord']:checked").val();
				updateScreenForJSONResponse(data, status, xhr,jsonVar.detailsDivID, '', false);
				if (data.length > 1000){
					var selectedValue = $("#selectedRecordValue").val();
					var selectedRadio = '';
					if(screenId == 'SOSD'){
						if(actionUpdate == 'true'){
							radioValue = undefined;
						}
						
					}
					if(radioValue!=undefined){
						selectedRadio = $('input:radio[name="selectRecord"][value="'
								+ radioValue + '"]');
					}else{
						selectedRadio = $('input:radio[name="selectRecord"][value="'
								+ selectedValue + '"]');
					}
					if (selectedRadio) {
						selectedRadio.attr('checked', true);
						displaySecSelectedRecordDetails(screenId);
						var message = $("#message").val();
						var sr = $("#selectRecord").val();
						$("#messageContainer").html(displayCustomValidation('blue', message));
						$("#messageContainer").show();
					}
				}
			}
		},
		error : function(xhr, ajaxOptions, thrownError) {
			var msg = [ "Technical Error : " + xhr.status + '-'
			            + xhr.statusText ];
			var errors = displayeMessage(msg, 'red');
			$('#messageContainer', findContainerForm(currentElement)).html(errors);
			$("#messageContainer").show();
			$.unblockUI();
		}
	});
}

function submitCaseTransfer(){
	$("#messageContainer").html("");
	if (!checkAllRequiredFields()) {
		return true;
	}
	var radioValue = $("input[name='searchRadio']:checked").val();
	$("#selectionType").val(radioValue);
	var urlPattern = "submitMultipleCaseTransfer";
	showWait();
	$.ajax({
		type : "POST",
		url : urlPattern,
		data : $("#casetransferHomeForm").serializeArray(),
		success : function(data, status, xhr) {
			var ct = xhr.getResponseHeader("content-type") || "";
			if (ct.indexOf('json') > -1) {
				var messageList= data.response.messages;
				if(messageList[0].includes('WOULD YOU LIKE TO CONTINUE WITH THE TRANSFER')){
					showTransferSimpleDialog(messageList[0],'submitMultipleCaseTransfer','casetransferHomeForm',radioValue);
				}else{
					updateScreenForJSONResponse(data, status, xhr,'messageContainer', '', false);
					$("#messageContainer").show();			
					if (data.response.status) {
						$('#casetransferHomeForm')[0].reset();
						if(radioValue==='Z'){
							$("#selectZipCode").attr("checked", true);
						}else if(radioValue==='C'){
							$("#caseload").attr("checked", true);
						}
						highlightMandatoryFieldsOnLoad();
					}
				}
			} else{
				updateScreenForJSONResponse(data, status, xhr,'casetransferHomeFormdiv', '', false);
				$("#casetransferHomeFormdiv").show();
			}
			$.unblockUI();
		},
		error : function(xhr, ajaxOptions, thrownError) {
			var msg = [ "Technical Error : " + xhr.status + '-'
			            + xhr.statusText ];
			var errors = displayeMessage(msg, 'red');
			$('#messageContainer', findContainerForm(currentElement)).html(errors);
			$("#messageContainer").show();
			$.unblockUI();
		}
	});
}
