public List<SnapCharacteristicRecord> getIncapDiablityIndicator(SnapCharacteristicInputParam inputParam)
			throws BatchBusinessException {
		try {
			String sqlQuery = getQueryWithSchemaName(QueryConstants.GET_INCAP_DSBLTY_IND);
			Object[] params = new Object[] { inputParam.getParam1(), inputParam.getParam2(), inputParam.getParam3() };
			final String value =
					jdbcTemplate.queryForObject(sqlQuery, params, String.class);
			if (StringConstants.CHAR_Y.equalsIgnoreCase(value)) {
				return true;
			}
		} catch (EmptyResultDataAccessException e) {
			return Collections.emptyList();
		} catch (Exception e) {
			if (e.getCause() instanceof SQLException) {
				SQLException sqlException = (SQLException) e.getCause();
				int sqlErrorCode = sqlException.getErrorCode();
				SnapCharacteristicInputParam.setSqlErrorFlag(true);
			} else {
				SnapCharacteristicInputParam.setSqlErrorFlag(false);
			}
		}
		return null;
	}


 public  String GET_INCAP_DSBLTY_IND = "SELECT 'Y' from TWNINCAP_DSBLTY T1 WHERE T1.UPI = ? AND T1.PRD_BEG_DT <= ? AND T1.PRD_END_DT >= ? AND T1.INCAP_DSBL_RSN_CD = 'D' AND T1.INCAP_DSBL_VERF_CD = 'Y'  ";
