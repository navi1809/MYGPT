 	function calculateAge(){
	var mdob=document.getElementById('monthgroupDateOfBirth').value;
	var ddob=document.getElementById('dategroupDateOfBirth').value;
	var ydob=document.getElementById('yeargroupDateOfBirth').value;
	const currentDate = new Date();
	const year = currentDate.getFullYear();
	const month = String(currentDate.getMonth()+1).padStart(2,'0');
	const day =   String(currentDate.getDate()).padStart(2,'0');
	if(ydob == null || ydob.trim() != ""){
	var age = year-ydob;
	const mdiff = month-mdob;
	const ddiff = day-ddob;
	if(mdiff < 0 || (mdiff === 0 && ddiff < 0)){
		age--;
	}
			if (age >= <%=sogiAgeValue%>){
				disableTextBoxes();
				addOnChangeEventListeners();
				clearGenderFieldsValues();
				document.getElementById('sec3Div').hidden = false ;
			}
			else{
				clearGenderFields();
				document.getElementById('sec3Div').hidden = true;
			}
	}
}
