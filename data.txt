WITH STATE_FISCAL (FISCAL_START_DATE, FISCAL_END_DATE) AS (
    SELECT
        CASE
            WHEN PROCESS_MM > 6 THEN DATE(PROCESS_CCYY || '-07-01')
            ELSE DATE(PRV_PROCESS_CCYY || '-07-01')
        END,
        CASE
            WHEN PROCESS_MM > 6 THEN DATE((PROCESS_CCYY + 1) || '-06-30')
            ELSE DATE(PROCESS_CCYY || '-06-30')
        END
    FROM {?SCHEMA}.TWNSYSDATE
),
CURRENTDATE (REPORTDATE, BEGINDATE, ENDDATE) AS (
    SELECT
        SYSTEM_DATE,
        DATE(PROCESS_CCYY || '-' || PROCESS_MM || '-01'),
        MTH_END_DT
    FROM {?SCHEMA}.TWNSYSDATE
),
DATE_36 (DT_36) AS (
    SELECT
        CASE
            WHEN (PROCESS_MM + 1) > 12 
                THEN CAST(PROCESS_CCYY - 3 AS VARCHAR(4)) || '-01-01'
            ELSE CAST(PROCESS_CCYY - 3 AS VARCHAR(4)) || '-' || 
                CASE
                    WHEN (PROCESS_MM + 1) <= 9 
                        THEN '0' || CAST(PROCESS_MM + 1 AS VARCHAR(1))
                    ELSE CAST(PROCESS_MM + 1 AS VARCHAR(2))
                END || '-01'
        END
    FROM {?SCHEMA}.TWNSYSDATE
),
SQ_1 (OFC_CD, OFC_NM, ICI) AS (
    SELECT DISTINCT T1.OFC_CD, T2.OFC_NM, T1.ICI
    FROM {?SCHEMA}.TWNFAM_CASELD_CASE T1
    JOIN {?SCHEMA}.TWNOFFICE T2 ON T1.OFC_CD = T2.OFC_CD
    JOIN {?SCHEMA}.TWNPRGM_CASE_HSTRY T3 ON T1.ICI = T3.ICI
    WHERE T1.OFC_CD BETWEEN 'AA' AND 'ZZ'
      AND T3.PRGM_CASE_TYPE = 'FS'
    ORDER BY T1.OFC_CD
),
SQ_2 (UPI, ICI, OFC_NM, OFC_CD) AS (
    SELECT DISTINCT T1.UPI, T1.ICI, S1.OFC_NM, S1.OFC_CD
    FROM {?SCHEMA}.TWNCASE_MEMBER_HST T1
    JOIN SQ_1 S1 ON T1.ICI = S1.ICI
),
LINE1_FISCAL(UPI, OFC_NM, LINE1_COUNT_FISCAL) AS (
    SELECT S2.UPI, S2.OFC_NM, COUNT(*) AS LINE1_COUNT_FISCAL
    FROM {?SCHEMA}.TWNPERSON_ELIG_HST T1
    JOIN SQ_2 S2 ON T1.UPI = S2.UPI
    JOIN STATE_FISCAL ST ON T1.PRD_BEG_DT >= ST.FISCAL_START_DATE
    JOIN CURRENTDATE CD ON T1.PRD_END_DT <= CD.ENDDATE
    WHERE T1.ABAWD_FG NOT IN ('X','N')
    GROUP BY S2.UPI, S2.OFC_NM
),
LINE1_MONTH(UPI, OFC_NM, LINE1_COUNT_MONTH) AS (
    SELECT S2.UPI, S2.OFC_NM, COUNT(*) AS LINE1_COUNT_MONTH
    FROM {?SCHEMA}.TWNPERSON_ELIG_HST T1
    JOIN SQ_2 S2 ON T1.UPI = S2.UPI
    JOIN CURRENTDATE CD ON T1.PRD_BEG_DT >= CD.BEGINDATE
    AND T1.PRD_END_DT <= CD.ENDDATE
    WHERE T1.ABAWD_FG NOT IN ('X','N')
    GROUP BY S2.UPI, S2.OFC_NM
),
COUNT_Y_MONTH (UPI, OFC_NM, ABAWD_FG, Y_MONTH_COUNT) AS (
    SELECT T1.UPI, T2.OFC_NM, T1.ABAWD_FG, COUNT(*) AS Y_MONTH_COUNT
    FROM {?SCHEMA}.TWNPERSON_ELIG_HST T1
    JOIN SQ_2 T2 ON T1.UPI = T2.UPI
    JOIN CURRENTDATE CD ON T1.PRD_BEG_DT >= CD.BEGINDATE
        AND T1.PRD_END_DT <= CD.ENDDATE     
    WHERE T1.ABAWD_FG = 'Y'
    GROUP BY T1.UPI, T2.OFC_NM, T1.ABAWD_FG
),
COMBINED_Y_MTH_36MTH_TOT AS(
SELECT
    UPI,
    OFC_NM,
    ABAWD_FG,
    SUM(Y_MONTH_COUNT) AS TOTAL_Y_MONTH_COUNT
FROM (
    SELECT 
        TRIM(UPPER(T1.UPI)) AS UPI,
        T2.OFC_NM,
        T1.ABAWD_FG, 
        COUNT(*) AS Y_MONTH_COUNT
    FROM 
        {?SCHEMA}.TWNPERSON_ELIG_HST T1
    JOIN 
        SQ_2 T2 ON TRIM(UPPER(T1.UPI)) = TRIM(UPPER(T2.UPI))
    JOIN 
        CURRENTDATE CD ON T1.PRD_BEG_DT >= CD.BEGINDATE
                      AND T1.PRD_END_DT <= CD.ENDDATE
    WHERE 
        T1.ABAWD_FG = 'Y'
    GROUP BY 
        TRIM(UPPER(T1.UPI)), T2.OFC_NM, T1.ABAWD_FG 
    UNION ALL
    SELECT 
        TRIM(UPPER(T1.UPI)) AS UPI,
        T2.OFC_NM,
        T1.ABAWD_FG, 
        COUNT(*) AS Y_MONTH_COUNT
    FROM 
        {?SCHEMA}.TWNPERSON_ELIG_HST T1
    JOIN 
        SQ_2 T2 ON TRIM(UPPER(T1.UPI)) = TRIM(UPPER(T2.UPI))
    JOIN 
        DATE_36 D36 ON T1.PRD_BEG_DT >= D36.DT_36
    JOIN 
        CURRENTDATE CD ON T1.PRD_END_DT <= CD.ENDDATE
                      AND T1.PRD_BEG_DT NOT BETWEEN CD.BEGINDATE AND CD.ENDDATE
    WHERE 
        T1.ABAWD_FG = 'Y'
        AND EXISTS (
            SELECT 1 
            FROM COUNT_Y_MONTH T3 
            WHERE T3.Y_MONTH_COUNT = 1 AND TRIM(UPPER(T3.UPI)) = TRIM(UPPER(T1.UPI))
        )
    GROUP BY 
        TRIM(UPPER(T1.UPI)), T2.OFC_NM, T1.ABAWD_FG
) COMBINED
GROUP BY
    UPI, OFC_NM, ABAWD_FG
ORDER BY
    UPI, OFC_NM, ABAWD_FG
),

        JOIN SQ_2 T2 ON
             T1.UPI = T2.UPI

    MAX(OFC_NM) AS OFC_NM,      
    MAX(ABAWD_FG) AS ABAWD_FG,  
    SUM(Y_FISCAL_COUNT) AS Y_FISCAL_COUNT
FROM (
    SELECT 
        T1.UPI,
        T2.OFC_NM,
        T1.ABAWD_FG,
        COUNT(*) AS Y_FISCAL_COUNT

    JOIN SQ_2 T2 ON T1.UPI = T2.UPI
    JOIN STATE_FISCAL ST ON T1.PRD_BEG_DT >= ST.FISCAL_START_DATE
    JOIN CURRENTDATE CD ON T1.PRD_END_DT <= CD.ENDDATE 
    WHERE T1.ABAWD_FG = 'Y'
    GROUP BY T1.UPI, T2.OFC_NM, T1.ABAWD_FG
    UNION ALL
    SELECT 
        T1.UPI,
        T2.OFC_NM,
        T1.ABAWD_FG,
        COUNT(*) AS Y_FISCAL_COUNT
    FROM {?SCHEMA}.TWNPERSON_ELIG_HST T1
    JOIN SQ_2 T2 ON T1.UPI = T2.UPI
    JOIN DATE_36 D36 ON T1.PRD_BEG_DT >= D36.DT_36
    JOIN CURRENTDATE CD ON T1.PRD_END_DT <= CD.ENDDATE
    JOIN STATE_FISCAL ST ON T1.PRD_BEG_DT NOT BETWEEN ST.FISCAL_START_DATE AND ST.FISCAL_END_DATE
    WHERE T1.ABAWD_FG = 'Y' 
      AND EXISTS (
          SELECT 1 
          FROM COUNT_Y_FISCAL T3 
          WHERE T3.Y_FISCAL_COUNT > 0 
            AND T3.UPI = T1.UPI
      )
    GROUP BY T1.UPI, T2.OFC_NM, T1.ABAWD_FG
) Q
GROUP BY UPI
