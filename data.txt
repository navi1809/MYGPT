CREATE TABLE UnifiedReport (
    id INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    report_type VARCHAR(50),   -- Identifies the report type (e.g., FNS_MATCH, SSI_BENEFITS)
    report_date DATE,          -- Report date
    record_id INT,             -- Groups fields into a single row
    debt_number VARCHAR(20),   -- Debt number
    ssn VARCHAR(11),           -- Social security number
    last_name VARCHAR(50),     -- Last name
    first_name VARCHAR(50),    -- First name
    amount DECIMAL(15,2),      -- Debt or transaction amount
    additional_info CLOB,      -- Stores extra fields as a text-based JSON-like structure
    created_at TIMESTAMP DEFAULT CURRENT TIMESTAMP
);
