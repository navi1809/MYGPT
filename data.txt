	function validateNotice(){
			var isManagedCase = '<%=isManagedCase%>';
			var actualNotice = '<%=noticeMethodCd%>';
			var notice = document.form1.noticeMethod.value;
			
			//if no managed case, do not allow to change -- need to do through ABE
			//ILIES-230696
 			if(isManagedCase == 'N' && (((notice=="E" && actualNotice!="E") || (notice=="B" && actualNotice!="B"))) && notice!="P"){
				alertMe('DC4447',document.form1.noticeMethod);
				return false;
			}
			else{
				//if paper notice, cannot change to any electronic form
				if(actualNotice=="P" && (notice == "B" || notice == "E" || notice=="Y" || notice=="X")){
					alertMe('DC4475',document.form1.noticeMethod);
					return false;
				}
			}
			
			if (notice=="E" || notice=="B" || notice=="X" || notice=="Y"){
				if (document.form1.eMail.value==""){
					alertMe('DC4445',document.form1.eMail);
					return false;
				}
			}
			return true;
	}
