-- ============================================================================
-- PRIMARY TABLE: TWN_MCE_INFO (No change)
-- ============================================================================
CREATE TABLE TWN_MCE_INFO (
    MCE_INFO_ID INT PRIMARY KEY IDENTITY(1,1),
    UPI BIGINT NOT NULL,
    ICI VARCHAR(10) NULL,
    SCR_TYPE CHAR(1) NOT NULL CHECK (SCR_TYPE IN ('1','2','3')),
    SCR_DATE DATE NOT NULL,
    QSTN_CD VARCHAR(20) NOT NULL,
    QSTN_VAL VARCHAR(50) NULL,
    CRTD_USER VARCHAR(30) NOT NULL,
    CRTD_DTTM DATETIME NOT NULL DEFAULT GETDATE(),
    UPD_USER VARCHAR(30) NULL,
    UPD_DTTM DATETIME NULL,
    
    CONSTRAINT UK_MCE_INFO_KEY UNIQUE (UPI, SCR_TYPE, SCR_DATE, QSTN_CD)
);
CREATE INDEX IX_MCE_INFO_UPI ON TWN_MCE_INFO(UPI);
CREATE INDEX IX_MCE_INFO_QSTN ON TWN_MCE_INFO(QSTN_CD);
GO

-- ============================================================================
-- CONSOLIDATED GROUP 1: Simple Verification Details
-- Replaces: FOSTER, AIAN, MEDFRL, TANF, SNAP, VETER, PREG, CHILD18, MEDCR
-- ============================================================================
CREATE TABLE TWN_MCE_DETAIL_VERIFY (
    DETAIL_VERIFY_ID INT PRIMARY KEY IDENTITY(1,1),
    UPI BIGINT NOT NULL,
    MCE_INFO_ID INT NOT NULL,
    QSTN_CD VARCHAR(20) NOT NULL,                  -- FRM_FOSTER, AIAN, MEDFRL, TANF, SNAP, VETER, PREG, CHILD_18, MEDCR
    APPRN_MO CHAR(1) NULL,                         -- Application/Renewal Month (S1 only)
    LOOKBK_MO CHAR(1) NULL,                        -- Lookback Month(s) (S1 only)
    VER_STS VARCHAR(1) NULL,                       -- N/P/Y/X
    LOSS_STS CHAR(1) NULL,                         -- Loss of Status (S1 only)
    CRTD_USER VARCHAR(30) NOT NULL,
    CRTD_DTTM DATETIME NOT NULL DEFAULT GETDATE(),
    UPD_USER VARCHAR(30) NULL,
    UPD_DTTM DATETIME NULL,
    
    CONSTRAINT FK_DETAIL_VERIFY_MCE FOREIGN KEY (MCE_INFO_ID) 
        REFERENCES TWN_MCE_INFO(MCE_INFO_ID) ON DELETE CASCADE,
    CONSTRAINT UK_DETAIL_VERIFY_KEY UNIQUE (MCE_INFO_ID, QSTN_CD)
);
CREATE INDEX IX_DETAIL_VERIFY_UPI ON TWN_MCE_DETAIL_VERIFY(UPI);
CREATE INDEX IX_DETAIL_VERIFY_QSTN ON TWN_MCE_DETAIL_VERIFY(QSTN_CD);
