$('#ssn1confirmSocialSecurity,#ssn2confirmSocialSecurity,#ssn3confirmSocialSecurity,#ssn1socialSecurity,#ssn2socialSecurity,#ssn3socialSecurity').keyup(function(){
	var emptyString = '';
	var currentLngCode = '<%=languageType%>';
	if(joinSSN().length === 9 && joinConfirmSSN().length === 9 && joinSSN() === joinConfirmSSN()){
		$('#ssn3confirmSocialSecurity').parent().find('.inline-message').removeClass().addClass('inline-message inline-match').text(getErrorMessage("GL101" + currentLngCode));
	}else if(joinSSN().length === 9 && joinConfirmSSN().length === 9){

	}
});

var IsNoName = function(){
 	var emptyString = '';
 	var isFnameEmpty = $('#firstName').val().replace(/^\s+|\s+$/g, emptyString) === emptyString;
 	var isMnameEmpty = $('#middleInitial').val().replace(/^\s+|\s+$/g, emptyString) === emptyString;
 	var isLnameEmpty = $('#lastName').val().replace(/^\s+|\s+$/g, emptyString) === emptyString;
 	var isSuffixEmpty = $('#suffixName option:selected').text().replace(/^\s+|\s+$/g, emptyString) === emptyString;
 	if(isFnameEmpty){
 		$('input[class*="selected"]').next(".label").text($('#personDefault').text());
 	}
 	if(isFnameEmpty && isMnameEmpty && isLnameEmpty && isSuffixEmpty){
 		$('#personDefault').show(); 			
 	}else{$('#personDefault').hide();}
 };
