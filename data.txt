			//get driver array
			int[] driverArray = (int[]) session.get(FwConstants.RMC_DRIVER);
			String currentPageID = (String) request.get(FwConstants.CURRENT_PAGE_ID);
			rmcAfterColl = (RMC_RQST_Collection) pageCollection.get("RMC_RQST_Collection");
			String isRMB = AppConstants.NO;
			if(session.get("IS_RMB")!=null)
				isRMB = session.get("IS_RMB").toString();
			
			rmcRqstCargo = rmcAfterColl.getCargo(0);
			rmcRqstBeforeCargo=(RMC_RQST_Cargo)session.get(AppConstants.RMC_RQST);	
			otherBO.validatePageContents(rmcRqstBeforeCargo,rmcRqstCargo,isRMB);	
						
			if(otherBO.hasMessages()){
				request.put(FwConstants.MESSAGE_LIST, otherBO.getMessageList());
				//put the first Name into request to avoid null pointer in JSP
				pageCollection.put("RMC_RQST_Collection", pageCollection.get("RMC_RQST_Collection"));	
				return;				
			}		 	
			
			
			
			
			RMC_RQST_Cargo rmcRqstAfterCargo =  null; 
			
				if(rmcRqstBeforeCargo==null)
					rmcRqstBeforeCargo=new RMC_RQST_Cargo();			
				String othChngEffDt = rmcRqstCargo.getOther_change_eff_dt();
				if (othChngEffDt != null && !othChngEffDt.trim().equals("") && !othChngEffDt.equals(AppConstants.HIGH_TIMESTAMP))
				{
					StringBuffer dateConverter  = new StringBuffer();
					String sAppDate = rmcRqstCargo.getOther_change_eff_dt();
					dateConverter.append( sAppDate.substring(6,10)).append("-").append(sAppDate.substring(0,2)).append("-").append(sAppDate.substring(3,5));
					rmcRqstCargo.setOther_change_eff_dt(dateConverter.toString());	
					//rmcRqstCargo.setOther_change_stat_sw(rmcRqstBeforeCargo.getOther_change_stat_sw());
				}
				if(isRMB!=null && isRMB.trim().length()>0 && isRMB.equalsIgnoreCase(AppConstants.YES)){
					rmcRqstAfterCargo = (RMC_RQST_Cargo) rmcRqstBeforeCargo.clone();
					rmcRqstAfterCargo.setOther_change_eff_dt(rmcRqstCargo.getOther_change_eff_dt());
					rmcRqstAfterCargo.setOther_change_stat_sw(rmcRqstCargo.getOther_change_stat_sw());
					rmcRqstAfterCargo.setOther_change_text(rmcRqstCargo.getOther_change_text());
					
					// changed by Yogendra
					rmcRqstAfterCargo.setOther_change_type_cd(rmcRqstCargo.getOther_change_type_cd());
				}else
				rmcRqstAfterCargo = jobBo.getCompleteCargo(rmcRqstBeforeCargo, rmcRqstCargo);				
				//Checking the Before Collection is Null or Not
				if(rmcRqstBeforeCargo != null ){
					//Compare cargoes to see what has changed
					rmcRqstAfterCargo = (RMC_RQST_Cargo)isChanged(rmcRqstBeforeCargo,rmcRqstAfterCargo);
				}
				if(rmcRqstBeforeCargo == null || rmcRqstAfterCargo.isDirty()){
					rmcRqstAfterCargo.setRowAction(FwConstants.ROWACTION_UPDATE);
					RMC_RQST_Collection rmcRqstcoll =new RMC_RQST_Collection();
					rmcRqstcoll.addCargo(rmcRqstAfterCargo);

					session.put(AppConstants.RMC_RQST,rmcRqstAfterCargo);
					session.put(FwConstants.RMC_DRIVER,driverArray);
				}
				
				//Add the page collection to before collection
				if(session.get(FwConstants.BEFORE_COLLECTION) != null){
					((Map)session.get(FwConstants.BEFORE_COLLECTION)).putAll(pageCollection);					
				}
				else{
					//Map beforeColl = new HashMap();
					beforeColl.putAll(pageCollection);
					session.put(FwConstants.BEFORE_COLLECTION, beforeColl);
				}
				txnBean.setPageCollection(pageCollection);
				
				driverManager.makeComplete(FwConstants.RMC_DRIVER, currentPageID, driverArray, (String)session.get(AppConstants.APP_NUMBER));
				log(ILog.INFO, "BenefitsEJBBean::storeAddUserDetail:End");

					request.put(FwConstants.NEXT_PAGE_ACTION,(String)request.get(FwConstants.CURRENT_PAGE_ID)+"Load");

			RMC_IN_PRFL_Cargo rmcInPrflSessCargo = null;
			String indvSeqNum;
			if (rmcResponseCustomColl != null && rmcResponseCustomColl.size() > 0) {
				for (int i = 0; i < rmcResponseCustomColl.size(); i++) {
					rmcResponseCustomCargo = rmcResponseCustomColl.getCargo(i);

					indvSeqNumber = rmcInPrflPageColl.getCargo(i).getIndv_seq_num();

					}

			Map beforeCollMap = new HashMap();
			int rmcRespBeforeCollSize = rmcResponseCustomColl == null ? 0 : rmcResponseCustomColl.size();
			for(int i=0; i<rmcRespBeforeCollSize; i++){
				RMC_RESPONSE_Custom_Cargo beforeCargo = (RMC_RESPONSE_Custom_Cargo)rmcResponseCustomColl.get(i);
				String beforeCargoCatType = beforeCargo.getCategoryType();
				String beforeCargoIndivSeqNum = beforeCargo.getIndvSeqNum();
				String beforeCargoResponse = beforeCargo.getResponse();
				String key = beforeCargoCatType + "_" + beforeCargoIndivSeqNum;
				String value = beforeCargoResponse;
				beforeCollMap.put(key,value);
