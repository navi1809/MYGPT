Function AME48P9-IPVDQ-INIT()
  AME48Q22-S22();

    AME48W.WS-IPVDQ-CD-IDX = 1;

    AME48Q22-S22-SCAN();
    while (sysVar.sqlData.sqlcode == 0)
      AME48W.WS-IPVDQ-DESC[WS-IPVDQ-CD-IDX] = H2-CODE-TBL.CODE-DESC;
      AME48W.WS-IPVDQ-CD[WS-IPVDQ-CD-IDX] = H2-CODE-TBL.CODE-TBL-CD;

      AME48W.WS-IPV-ARRAY-SIZE = AME48W.WS-IPV-ARRAY-SIZE + 1;

      AME48Q22-S22-SCAN();

    end

  end

  /* DETERMINE THE SIZE OF THE IPV LEGEND*/

  AME48W.WS-IPV-LEGND-CALC = AME48W.WS-IPV-ARRAY-SIZE / 2;

  if (AME48W.WS-IPV-LEGND-DEC != 0)
    AME48W.WS-IPV-LEGND-SIZE = AME48W.WS-IPV-LEGND-WHOLE + 1;
  else
    AME48W.WS-IPV-LEGND-SIZE = AME48W.WS-IPV-LEGND-WHOLE;
  end

  AME48W.WS-IPV-LEGND-SIZE = AME48W.WS-IPV-LEGND-SIZE + 3;

  /*  */
