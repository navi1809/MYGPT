// CHECK TWNPERSON DATA
Function AIN71P-CHK-PERSON()
  /* *********************************************************************/
  /* This process is to check if the person whose details are reported **/
  /* from bendex is known to nomads                                    **/
  /* *********************************************************************/
  /*  */
  /* Read the table TWNPERSON to check if the person is KNOWN*/
  /*  */
  /* Set the sql row record empty*/
  /*  */
  set H2-PERSON empty;
  /*  */
  /* Key setting process*/
  /*  */
  /* Move the retreive key value to the sql row record*/
  /*  */
  /* if AIN71WS.BX1IR-SSN = ' ' OR    /*  CHECK FOR VALID SSN*/
  /* AIN71WS.BX1IR-SSN NOT NUMERIC OR*/
  if (AIN71WS.BX1IR-CNTRL-UPI-CHA not numeric)
    AIN71P-MOVE();
    AIN71P-WELF-WRT();
    return;
  end
  H2-PERSON.UPI = AIN71WS.BX1IR-CNTRL-UPI;
  /*  */
  /* End key setting process*/
  /*  */
  /* Perform the process to read the table TWNPERSON*/
  /*  */
  AIN71P-PER-INQ();
  /*  */
  /* Check for fatal errors*/
  /*  */
  if (VDBCONTROL.URC > 104)
    WCM00P2-TO-CEP01A();
  end
  /*  */
  if (sysVar.sqlData.sqlcode == 100)
    AIN71WS2.BX2WS-TEMP-UPI = AIN71WS.BX1IR-CNTRL-UPI;
    AIN71WS2.BX2WS-TEMP-REST = AIN71WS.BX1IR-CNTRL-REST;
    AIN71P-MOVE();
    AIN71P-WELF-WRT();
    return;
  end /* END NO RECORD FOUND*/
  /*  */
  if (sysVar.sqlData.sqlcode == 0)
    AIN71WS2.BX2WS-NUM-UPI = H2-PERSON.UPI;
    AIN71WS2.BX2WS-TEMP-UPI = AIN71WS2.BX2WS-NUM-UPI;
    AIN71WS2.BX2WS-TEMP-REST = " ";
    AIN71P-MOVE();
    AIN71P-NOM-WRT();
  end /* END PROCESS RECORD FOUND*/
  /*  */
end // end AIN71P-CHK-PERSON

// PRE-PROCESS INITIALIZE
Function AIN71P-INIT()
  /* WILL INITIALIZE ALL INPUT, OUTPUT,  AND WORKING STORAGE AREAS*/
  /*  */
  VGVar.handleHardIOErrors = 1;
  /*  */
  set AIN71R-INPUT empty;
  set AIN71R-WEL-OUT empty;
  set AIN71R-NOM-OUT empty;
  set AIN71R-COUNT empty;
  set AIN71WS2 empty;
  /*  */
end // end AIN71P-INIT


// Main Process
Function AIN71P-MAIN()
  /* ********************************************************************/
  /* Main process                                 **/
  /* ********************************************************************/
  /*  */
  /* THE BENDEX FILE NEEDS TO BE PRE-PROCESSED IN ORDER TO*/
  /* WRITE THE BENDEX RECORDS INTO EITHER A WELFARE FILE OR A NOMADS FILE*/
  /* IF THE BENDEX SSN IS NOT ON THE TWNPERSON TABLE THE RECORD WILL BE*/
  /* WRITTEN TO THE WELFARE FILE. IF THE SSN IS ON THE TWNPERSON TABLE*/
  /* THE RECORD IS WRITTEN TO THE NOMADS FILE.*/
  /*  */
  /* Perform the process to initialise the records used*/
  /*  */
  AIN71P-INIT();
  /*  */
  /* READ THE FIRST RECORD FROM THE INPUT FILE*/
  /*  */
  AIN71P-READ();
  /*  */
  /* PROCESS THE FOUND RECORD*/
  /*  */
  while (sysVar.errorCode == "00000000")
    if (AIN71WS2.BX2READ-CNT == 1 &&  /* HEADER RECORD*/
    AIN71WS.BX1IR-CLAIM-SSN not numeric)
      AIN71WS.BX1OT-REC = AIN71WS.BX1IN-REC;
      AIN71P-WELF-WRT();
      AIN71WS.BX1IN-REC = " ";
      AIN71WS.BX1OT-REC = " ";
    else
      AIN71P-CHK-PERSON();
      AIN71WS.BX1IN-REC = " ";
      AIN71WS.BX1OT-REC = " ";
    end
  /*  */
  /* FETCH THE NEXT RECORD FROM THE INPUT FILE*/
  /*  */
    AIN71P-READ();
  end /* END WHILE RECORD FOUND*/
  /*  */
  /* PERFORM THE PROCESS TO WRITE THE CONTROL FILE RECORDS*/
  /*  */
  AIN71P-CNTL();
  /*  */
end // end AIN71P-MAIN


// MOVES INPUT TO WORKING STORAGE

  /* CONTROL DATA INTO THE WORKING STORAGE OUTPUT FORMAT.*/
  /*  */
  AIN71WS.BX1OT-REC = AIN71WS.BX1IN-REC;
     /*  */
  AIN71WS.BX1OT-CNTRL-UPI = AIN71WS2.BX2WS-TEMP-UPI;
  AIN71WS.BX1OT-CNTRL-REST = AIN71WS2.BX2WS-TEMP-REST;
     /*  */
end // end AIN71P-MOVE


// WRITES THE NOMADS FILE REC
Function AIN71P-NOM-WRT()
  /* WILL MOVE THE WORKING STORAGE OUTPUT RECORD TO THE NOMADS*/
  /* FILE AND WRITE THE RECORD.*/
  /* ADDS 1 TO THE NOMADS RECORDS WRITTEN COUNT.*/
  /*  */
  AIN71R-NOM-OUT.BX2NOM-OUT = AIN71WS.BX1OT-REC;
  try
    add AIN71R-NOM-OUT ;
  end
  /*  */
  if (sysVar.errorCode == "00000000")
    AIN71WS2.BX2N-WRITE-CNT = AIN71WS2.BX2N-WRITE-CNT + 1;
  end
  if (AIN71R-NOM-OUT is ioError && 
  AIN71R-NOM-OUT not endOfFile)
    VDBCONTROL.UPROC-NM = "AIN71R-NOM-WRT";
    VDBCONTROL.UPROC-OBJ = "AIN71R-NOM-OUT";
    VDBCONTROL.UPROC-OPT = "ADD";
    AIN71WS2.BX2WS-ERR-CD = sysVar.errorCode;
    VMESSAGE.UMSGINS[1] = AIN71WS2.BX2WS-ERR-LAST-FOUR;
    VMESSAGE.UMSGINS[2] = "AIN71A";
  /*  */
  /* MOVES THE CORRESPONDING ERROR MESSAGES*/
  /*  */
    if (AIN71R-NOM-OUT is fileNotAvailable)
      VDBCONTROL.UEZESQRRM = "FILE IS NOT AVAILABLE";
    else
      if (AIN71R-NOM-OUT is fileNotFound)
        VDBCONTROL.UEZESQRRM = "FILE NOT FOUND";
      else
        if (AIN71R-NOM-OUT is invalidFormat)
          VDBCONTROL.UEZESQRRM = "FILE FORMAT MISMATCH";
        end /* ** END CHECK FOR FORMAT MISMATCH*/
      end /* ** END CHECK FOR FILE NOT FOUND*/
    end /* ** END CHECK FOR FILE NOT AVAILABLE*/
    WCM00P2-TO-CEP01A();
  end
end // end AIN71P-NOM-WRT

// Read the table TWNPERSON
Function AIN71P-PER-Q1()
  /* *********************************************************************/
  /* This process is to fetch data from the table TWNPERSON            **/
  /* *********************************************************************/
  /*  */
  /* Move name of sql row record to common record*/
  /*  */
  VDBCOMMON.USQLREC = "H2-PERSON";
  VDBCONTROL.UPROC-OBJ = "H2-PERSON";
  VDBCONTROL.UPROC-NM = "AIN71P-PER-Q1";
  VDBCONTROL.UPROC-OPT = "INQUIRY";
  /*  */
  try
    get H2-PERSON
      with #sql{
        select UPI,
          SEX_CD,
          DOB,
          SSN
        from TWNPERSON T1
        WHERE
            UPI = :UPI
         --** INSERT ORDER BY CLAUSE HERE **
      }
      into UPI,
           SEX-CD,
           DOB,
           SSN ;
  end
  /* increment count of rows retrieved*/
  /*  */
end // end AIN71P-PER-Q1


// READ INPUT PROCESS
Function AIN71P-READ()
  /* WILL INCREMENT THE RECORDS READ COUNT AND MOVE THE INPUT RECORD*/
  /* TO THE WORKING STORAGE INPUT FIELDS.*/
  /*  */
  /* Perform the process to read the INPUT BENDEX FILE*/
  /*  */
  AIN71P-SCAN();
  if (sysVar.errorCode == "00000000")
    AIN71WS2.BX2READ-CNT = AIN71WS2.BX2READ-CNT + 1;
    AIN71WS.BX1IN-REC = AIN71R-INPUT.BX2BN-IN;
  end
  /*  */
  /*  */
end // end AIN71P-READ


// READ THE INPUT FILE
Function AIN71P-SCAN()
  try
    get next AIN71R-INPUT ;
  end
  /* READ THE INPUT BENDEX RECORD AND MOVE IT TO*/
  /* WORKING STORAGE.  INCREMENT THE RECORDS READ COUNT.*/
  /*  */
  if (AIN71R-INPUT is ioError && 
  AIN71R-INPUT not endOfFile)
    VDBCONTROL.UPROC-NM = "AIN71P-SCAN";
    VDBCONTROL.UPROC-OBJ = "AIN71R-INPUT";
    VDBCONTROL.UPROC-OPT = "SCAN";
    AIN71WS2.BX2WS-ERR-CD = sysVar.errorCode;
    VMESSAGE.UMSGINS[1] = AIN71WS2.BX2WS-ERR-LAST-FOUR;
    VMESSAGE.UMSGINS[2] = "AIN71A";
  /*  */
  /* the following conditions are added to move the corresponding*/
  /* error message*/
  /*  */
    if (AIN71R-INPUT is fileNotAvailable)
      VDBCONTROL.UEZESQRRM = "FILE IS NOT AVAILABLE";
    else
      if (AIN71R-INPUT is fileNotFound)
        VDBCONTROL.UEZESQRRM = "FILE NOT FOUND";
      else
        if (AIN71R-INPUT is invalidFormat)
          VDBCONTROL.UEZESQRRM = "FILE FORMAT MISMATCH";
        end /* ** end check for format mismatch*/
      end /* ** end check for file not found*/
    end /* ** end check for file not available*/
    WCM00P2-TO-CEP01A();
  end
  /*  */
end // end AIN71P-SCAN
