WITH DATES AS (
SELECT 
	SYSTEM_DATE,
	MTH_BEG_DT,
	MTH_END_DT
FROM W100DP1.TWNSYSDATE
),
UPI AS (
SELECT 
	T1.APPL_UPI, 
	T2.OFC_CD
FROM
	W100DP1.TWNFAM_CASE_HSTRY T1,
	W100DP1.TWNFAM_CASELD_CASE T2,
	DATES 
WHERE 
	T1.ICI = T2.ICI
	AND T2.PRGM_OFC_TYPE = 'A'
	AND T2.OFC_CD NOT IN ( 'DC', 'DR', 'DL', 'DH' )
	AND T1.PRD_BEG_DT <= MTH_BEG_DT
	AND T1.PRD_END_DT >= MTH_END_DT
	AND T1.CASE_STS IN ('O', 'P')
),
OFFICE AS(
SELECT 
	OFC_NM,
	OFC_CD
FROM 
	W100DP1.TWNOFFICE T1
),
CASE_MEMBER AS (
SELECT 
	T1.AID_CD_TYPE,
	T1.AID_CD,
	T1.PRGM,
	T2.ICI
FROM 
	W100DP1.TWNAID_CODE_CATG T1,
	W100DP1.TWNCASE_MEMBER_HST T2,
	DATES
WHERE 
	T2.PRD_BEG_DT <= MTH_BEG_DT
	AND T2.PRD_END_DT >= MTH_END_DT
	AND ((T1.PRGM = 'AF' AND T1.AID_CD = T2.AFDC_AID_CD AND T2.AFDC_ELIG_CD NOT IN ('3','9'))
	OR  (T1.PRGM = 'SG' AND T1.AID_CD = T2.SG_AID_CD AND T2.SG_ELIG_CD NOT IN ('3','9'))
	OR  (T1.PRGM = 'TC' AND T1.AID_CD = T2.TC_AID_CD AND T2.TC_ELIG_CD NOT IN ('3','9'))
	OR  (T1.PRGM = 'TL' AND T1.AID_CD = T2.TL_AID_CD AND T2.TL_ELIG_CD NOT IN ('3','9'))
	OR  (T1.PRGM = 'TN' AND T1.AID_CD = T2.TN_AID_CD AND T2.TN_ELIG_CD NOT IN ('3','9'))
	OR  (T1.PRGM = 'TP' AND T1.AID_CD = T2.TP_AID_CD AND T2.TP_ELIG_CD NOT IN ('3','9'))
	OR  (T1.PRGM = 'FS' AND T1.AID_CD = T2.FS_AID_CD AND T2.FS_ELIG_CD NOT IN ('3','9'))
	OR  (T1.PRGM = 'MA' AND T1.AID_CD = T2.MA_AID_CD AND T2.MA_ELIG_CD NOT IN ('3','9'))
	OR  (T1.PRGM = 'MC' AND T1.AID_CD = 'MCB' AND T2.MCB_ASST_IND = 'Y')
	OR  (T1.PRGM = 'RP' AND T1.AID_CD = DIGITS(T2.IVE_AID_CD) AND T2.IVE_ELIG_CD NOT IN ('3','9'))
	OR         (t1.PRGM = 'CC'))
),
FAM_CASE AS (
SELECT 
	T3.APPL_UPI,
	T3.ICI
FROM 
	W100DP1.TWNFAM_CASE_HSTRY T3,
	DATES
WHERE 
	T3.PRD_BEG_DT <= MTH_BEG_DT
	AND T3.PRD_END_DT >= MTH_END_DT
),
PRGM_CASE AS (
SELECT 
	T4.PRGM_CASE_STS,
	T4.RCRT_APL_DT,
	T4.ICI,
	T4.PRGM_CASE_TYPE,
	T4.AID_CD,
	T4.ELIG_CD
FROM 
	W100DP1.TWNPRGM_CASE_HSTRY T4,
	DATES
WHERE 
	T4.PRD_BEG_DT <= MTH_BEG_DT
	AND T4.PRD_END_DT >= MTH_END_DT
	AND T4.PRGM_CASE_STS IN ('P','O','D')
),
PRGM (AID_CD_TYPE, AID_CD, APPL_UPI, RCRT_APL_DT, PRGM_CASE_STS) AS (
SELECT DISTINCT
    CM.AID_CD_TYPE,
    CM.AID_CD,
    FC.APPL_UPI,
	PC.RCRT_APL_DT,
	PC.PRGM_CASE_STS
FROM 
	CASE_MEMBER CM JOIN 
	FAM_CASE FC ON FC.ICI = CM.ICI JOIN 
	PRGM_CASE PC ON PC.ICI = CM.ICI AND PC.PRGM_CASE_TYPE = CM.PRGM
WHERE 
	((CM.PRGM <> 'CC')
	OR (CM.PRGM = 'CC' AND CM.AID_CD = PC.AID_CD AND PC.ELIG_CD NOT IN ('3','9')))
	AND PC.PRGM_CASE_STS IN ('P','O')
UNION 
SELECT 
    MIN(CM.AID_CD_TYPE),
	MAX(CM.AID_CD),
	FC.APPL_UPI,
	MAX(PC.RCRT_APL_DT),
	MAX(PC.PRGM_CASE_STS)
FROM 
	CASE_MEMBER CM JOIN 
	FAM_CASE FC ON FC.ICI = CM.ICI JOIN 
	PRGM_CASE PC ON PC.ICI = CM.ICI AND PC.PRGM_CASE_TYPE = CM.PRGM
WHERE 
	((CM.PRGM <> 'CC')
	OR (CM.PRGM = 'CC' AND CM.AID_CD = PC.AID_CD AND PC.ELIG_CD NOT IN ('3','9')))
	AND CM.PRGM <> 'MC' 
	AND PC.PRGM_CASE_STS = 'D'
GROUP BY APPL_UPI
),
TYPE AS (
SELECT 
O.OFC_CD,
O.OFC_NM,
U.APPL_UPI,
MAX(CASE WHEN P.PRGM_CASE_STS='O' THEN 'Y' ELSE 'N' END) AS ONGOING,
MAX(CASE WHEN P.PRGM_CASE_STS='P' THEN 'Y' ELSE 'N' END) AS INTAKE,
MAX(CASE WHEN RCRT_APL_DT >= MTH_BEG_DT AND RCRT_APL_DT <= MTH_END_DT AND P.PRGM_CASE_STS='D' THEN 'Y' ELSE 'N' END) AS DENIED,
MAX(CASE WHEN RCRT_APL_DT >= MTH_BEG_DT AND RCRT_APL_DT <= MTH_END_DT AND AID_CD='CA' THEN 'Y' ELSE 'N' END) AS APP_CA_EXISTS,
MAX(CASE WHEN RCRT_APL_DT >= MTH_BEG_DT AND RCRT_APL_DT <= MTH_END_DT AND AID_CD='PS' THEN 'Y' ELSE 'N' END) AS APP_PS_EXISTS,
MAX(CASE WHEN RCRT_APL_DT >= MTH_BEG_DT AND RCRT_APL_DT <= MTH_END_DT AND AID_CD_TYPE=1 THEN 'Y' ELSE 'N' END) AS APP_TYPE_1,    
MAX(CASE WHEN RCRT_APL_DT >= MTH_BEG_DT AND RCRT_APL_DT <= MTH_END_DT AND AID_CD_TYPE=2 THEN 'Y' ELSE 'N' END) AS APP_TYPE_2,
MAX(CASE WHEN RCRT_APL_DT >= MTH_BEG_DT AND RCRT_APL_DT <= MTH_END_DT AND AID_CD_TYPE=5 THEN 'Y' ELSE 'N' END) AS APP_TYPE_5,
MAX(CASE WHEN RCRT_APL_DT >= MTH_BEG_DT AND RCRT_APL_DT <= MTH_END_DT AND AID_CD_TYPE=6 THEN 'Y' ELSE 'N' END) AS APP_TYPE_6,
MAX(CASE WHEN RCRT_APL_DT >= MTH_BEG_DT AND RCRT_APL_DT <= MTH_END_DT AND AID_CD_TYPE=7 THEN 'Y' ELSE 'N' END) AS APP_TYPE_7,
MAX(CASE WHEN RCRT_APL_DT >= MTH_BEG_DT AND RCRT_APL_DT <= MTH_END_DT AND AID_CD_TYPE=8 THEN 'Y' ELSE 'N' END) AS APP_TYPE_8,
MAX(CASE WHEN RCRT_APL_DT >= MTH_BEG_DT AND RCRT_APL_DT <= MTH_END_DT AND AID_CD_TYPE=9 THEN 'Y' ELSE 'N' END) AS APP_TYPE_9,
MAX(CASE WHEN RCRT_APL_DT >= MTH_BEG_DT AND RCRT_APL_DT <= MTH_END_DT AND AID_CD_TYPE=10 THEN 'Y' ELSE 'N' END) AS APP_TYPE_10,
MAX(CASE WHEN RCRT_APL_DT >= MTH_BEG_DT AND RCRT_APL_DT <= MTH_END_DT AND AID_CD_TYPE=11 THEN 'Y' ELSE 'N' END) AS APP_TYPE_11,
MAX(CASE WHEN RCRT_APL_DT >= MTH_BEG_DT AND RCRT_APL_DT <= MTH_END_DT AND AID_CD_TYPE=12 THEN 'Y' ELSE 'N' END) AS APP_TYPE_12,
MAX(CASE WHEN RCRT_APL_DT >= MTH_BEG_DT AND RCRT_APL_DT <= MTH_END_DT AND AID_CD_TYPE=13 THEN 'Y' ELSE 'N' END) AS APP_TYPE_13,
MAX(CASE WHEN RCRT_APL_DT >= MTH_BEG_DT AND RCRT_APL_DT <= MTH_END_DT AND AID_CD_TYPE=14 THEN 'Y' ELSE 'N' END) AS APP_TYPE_14,
MAX(CASE WHEN RCRT_APL_DT >= MTH_BEG_DT AND RCRT_APL_DT <= MTH_END_DT AND AID_CD_TYPE=15 THEN 'Y' ELSE 'N' END) AS APP_TYPE_15,
MAX(CASE WHEN RCRT_APL_DT >= MTH_BEG_DT AND RCRT_APL_DT <= MTH_END_DT AND AID_CD_TYPE=18 THEN 'Y' ELSE 'N' END) AS APP_TYPE_18,
MAX(CASE WHEN RCRT_APL_DT >= MTH_BEG_DT AND RCRT_APL_DT <= MTH_END_DT AND AID_CD_TYPE BETWEEN 7 AND 13 THEN 'Y' ELSE 'N' END) AS APP_TYPE_MAABD,
MAX(CASE WHEN RCRT_APL_DT >= MTH_BEG_DT AND RCRT_APL_DT <= MTH_END_DT AND AID_CD_TYPE BETWEEN 7 AND 11 THEN 'Y' ELSE 'N' END) AS APP_TYPE_MA_NI,
MAX(CASE WHEN P.PRGM_CASE_STS <> 'D' AND AID_CD='CA' THEN 'Y' ELSE 'N' END) AS CA_EXISTS,
MAX(CASE WHEN P.PRGM_CASE_STS <> 'D' AND AID_CD='PS' THEN 'Y' ELSE 'N' END) AS PS_EXISTS,
MAX(CASE WHEN P.PRGM_CASE_STS <> 'D' AND AID_CD_TYPE=1 THEN 'Y' ELSE 'N' END) AS TYPE_1,    
MAX(CASE WHEN P.PRGM_CASE_STS <> 'D' AND AID_CD_TYPE=2 THEN 'Y' ELSE 'N' END) AS TYPE_2,
MAX(CASE WHEN P.PRGM_CASE_STS <> 'D' AND AID_CD_TYPE=5 THEN 'Y' ELSE 'N' END) AS TYPE_5,
MAX(CASE WHEN P.PRGM_CASE_STS <> 'D' AND AID_CD_TYPE=6 THEN 'Y' ELSE 'N' END) AS TYPE_6,
MAX(CASE WHEN P.PRGM_CASE_STS <> 'D' AND AID_CD_TYPE=7 THEN 'Y' ELSE 'N' END) AS TYPE_7,
MAX(CASE WHEN P.PRGM_CASE_STS <> 'D' AND AID_CD_TYPE=8 THEN 'Y' ELSE 'N' END) AS TYPE_8,
MAX(CASE WHEN P.PRGM_CASE_STS <> 'D' AND AID_CD_TYPE=9 THEN 'Y' ELSE 'N' END) AS TYPE_9,
MAX(CASE WHEN P.PRGM_CASE_STS <> 'D' AND AID_CD_TYPE=10 THEN 'Y' ELSE 'N' END) AS TYPE_10,
MAX(CASE WHEN P.PRGM_CASE_STS <> 'D' AND AID_CD_TYPE=11 THEN 'Y' ELSE 'N' END) AS TYPE_11,
MAX(CASE WHEN P.PRGM_CASE_STS <> 'D' AND AID_CD_TYPE=12 THEN 'Y' ELSE 'N' END) AS TYPE_12,
MAX(CASE WHEN P.PRGM_CASE_STS <> 'D' AND AID_CD_TYPE=13 THEN 'Y' ELSE 'N' END) AS TYPE_13,
MAX(CASE WHEN P.PRGM_CASE_STS <> 'D' AND AID_CD_TYPE=14 THEN 'Y' ELSE 'N' END) AS TYPE_14,
MAX(CASE WHEN P.PRGM_CASE_STS <> 'D' AND AID_CD_TYPE=15 THEN 'Y' ELSE 'N' END) AS TYPE_15,
MAX(CASE WHEN P.PRGM_CASE_STS <> 'D' AND AID_CD_TYPE=18 THEN 'Y' ELSE 'N' END) AS TYPE_18,
MAX(CASE WHEN P.PRGM_CASE_STS <> 'D' AND AID_CD_TYPE BETWEEN 7 AND 13 THEN 'Y' ELSE 'N' END) AS TYPE_MAABD,
MAX(CASE WHEN P.PRGM_CASE_STS <> 'D' AND AID_CD_TYPE BETWEEN 7 AND 11 THEN 'Y' ELSE 'N' END) AS TYPE_MA_NI
FROM 
 UPI U JOIN 
 PRGM P ON U.APPL_UPI=P.APPL_UPI JOIN
 OFFICE O ON O.OFC_CD=U.OFC_CD CROSS JOIN 
 DATES
GROUP BY O.OFC_CD, OFC_NM, U.APPL_UPI),
MANI AS (
SELECT 
APPL_UPI,
TYPE_10,
CASE WHEN TYPE_MA_NI='Y' AND TYPE_10='Y' THEN 'N' ELSE TYPE_9 END AS TYPE_9,
CASE 
	WHEN TYPE_MA_NI='Y' AND TYPE_10='Y' THEN 'N' 
	WHEN TYPE_MA_NI='Y' AND TYPE_9='Y' THEN 'N'
	ELSE TYPE_7 
END AS TYPE_7,
CASE 
	WHEN TYPE_MA_NI='Y' AND TYPE_10='Y' THEN 'N' 
	WHEN TYPE_MA_NI='Y' AND TYPE_9='Y' THEN 'N'
	WHEN TYPE_MA_NI='Y' AND TYPE_7='Y' THEN 'N'
	ELSE TYPE_8

CASE
	WHEN TYPE_MA_NI='Y' AND TYPE_10='Y' THEN 'N' 
	WHEN TYPE_MA_NI='Y' AND TYPE_9='Y' THEN 'N'
	WHEN TYPE_MA_NI='Y' AND TYPE_7='Y' THEN 'N'
	WHEN TYPE_MA_NI='Y' AND TYPE_8='Y' THEN 'N'
	ELSE TYPE_11 
END AS TYPE_11,
APP_TYPE_10,
CASE WHEN APP_TYPE_MA_NI='Y' AND APP_TYPE_10='Y' THEN 'N' ELSE APP_TYPE_9 END AS APP_TYPE_9,
CASE 
	WHEN APP_TYPE_MA_NI='Y' AND APP_TYPE_10='Y' THEN 'N' 
	WHEN APP_TYPE_MA_NI='Y' AND APP_TYPE_9='Y' THEN 'N'
	ELSE APP_TYPE_7 
END AS APP_TYPE_7,
CASE 
	WHEN APP_TYPE_MA_NI='Y' AND APP_TYPE_10='Y' THEN 'N' 
	WHEN APP_TYPE_MA_NI='Y' AND APP_TYPE_9='Y' THEN 'N'
	WHEN APP_TYPE_MA_NI='Y' AND APP_TYPE_7='Y' THEN 'N'
	ELSE APP_TYPE_8 
END AS APP_TYPE_8,
CASE
	WHEN APP_TYPE_MA_NI='Y' AND APP_TYPE_10='Y' THEN 'N' 
	WHEN APP_TYPE_MA_NI='Y' AND APP_TYPE_9='Y' THEN 'N'
	WHEN APP_TYPE_MA_NI='Y' AND APP_TYPE_7='Y' THEN 'N'
	WHEN APP_TYPE_MA_NI='Y' AND APP_TYPE_8='Y' THEN 'N'
	ELSE APP_TYPE_11 
END AS APP_TYPE_11
FROM TYPE 
),
FINAL AS (

OFC_CD,
OFC_NM,
--APP_MTH
CASE WHEN APP_TYPE_1='Y' AND APP_PS_EXISTS='N' AND APP_TYPE_5='N' AND APP_TYPE_6='N' AND APP_TYPE_2='N' AND APP_TYPE_MAABD='N' THEN 1 END AS TANF_CASH,

CASE WHEN APP_TYPE_1='Y' AND APP_PS_EXISTS='N' AND APP_TYPE_5='N' AND APP_TYPE_6='N' AND APP_TYPE_2='Y' AND APP_TYPE_MAABD='Y' THEN 1 END AS TANF_CASH_MED_MA,
