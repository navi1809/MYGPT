WITH DATES AS (
    SELECT
        SYSTEM_DATE,

        DATE(
            RTRIM(CHAR(PROCESS_CCYY)) || '-' ||
            CASE
                WHEN PROCESS_MM < 10 THEN '0' || RTRIM(CHAR(PROCESS_MM))
                ELSE RTRIM(CHAR(PROCESS_MM))
            END || '-01'
        ) AS CURR_MTH_BEG,

        MTH_END_DT AS CURR_MTH_END,

        ADD_MONTHS(
            DATE(
                RTRIM(CHAR(PROCESS_CCYY)) || '-' ||
                CASE
                    WHEN PROCESS_MM < 10 THEN '0' || RTRIM(CHAR(PROCESS_MM))
                    ELSE RTRIM(CHAR(PROCESS_MM))
                END || '-01'
            ),
            -1
        ) AS PREV_MTH_BEG,

        LAST_DAY(
            ADD_MONTHS(
                DATE(
                    RTRIM(CHAR(PROCESS_CCYY)) || '-' ||
                    CASE
                        WHEN PROCESS_MM < 10 THEN '0' || RTRIM(CHAR(PROCESS_MM))
                        ELSE RTRIM(CHAR(PROCESS_MM))
                    END || '-01'
                ),
                -1
            )
        ) AS PREV_MTH_END,

        CASE
            WHEN PROCESS_MM > 6
                THEN DATE(RTRIM(CHAR(PROCESS_CCYY)) || '-07-01')
            ELSE DATE(RTRIM(CHAR(PRV_PROCESS_CCYY)) || '-07-01')
        END AS FISCAL_START_DATE,

        CASE
            WHEN PROCESS_MM > 6
                THEN DATE(RTRIM(CHAR(PROCESS_CCYY + 1)) || '-06-30')
            ELSE DATE(RTRIM(CHAR(PROCESS_CCYY)) || '-06-30')
        END AS FISCAL_END_DATE,

        CASE
            WHEN (PROCESS_MM + 1) > 12
                THEN DATE(RTRIM(CHAR(PROCESS_CCYY - 3)) || '-01-01')
            ELSE DATE(
                RTRIM(CHAR(PROCESS_CCYY - 3)) || '-' ||
                CASE
                    WHEN (PROCESS_MM + 1) < 10 THEN '0' || RTRIM(CHAR(PROCESS_MM + 1))
                    ELSE RTRIM(CHAR(PROCESS_MM + 1))
                END || '-01'
            )
        END AS DT_36,

        YEAR(SYSTEM_DATE) AS CURR_YEAR,
        MONTH(SYSTEM_DATE) AS CURR_MONTH
    FROM {?SCHEMA}.TWNSYSDATE
),

CASE_UPI_OFFICE AS (
    SELECT DISTINCT
        CM.UPI,
        FC.ICI,
        FC.OFC_CD,
        O.OFC_NM
    FROM {?SCHEMA}.TWNFAM_CASELD_CASE FC
    JOIN {?SCHEMA}.TWNOFFICE O
      ON FC.OFC_CD = O.OFC_CD
    JOIN {?SCHEMA}.TWNPRGM_CASE_HSTRY PH
      ON FC.ICI = PH.ICI
     AND PH.PRGM_CASE_TYPE = 'FS'
    JOIN {?SCHEMA}.TWNCASE_MEMBER_HST CM
      ON FC.ICI = CM.ICI
    WHERE FC.OFC_CD BETWEEN 'AA' AND 'ZZ'
),

BASE_POP AS (
    SELECT DISTINCT
        UPI,
        OFC_CD,
        OFC_NM
    FROM CASE_UPI_OFFICE
),

PERSON_36 AS (
    SELECT
        B.UPI,
        B.OFC_CD,
        B.OFC_NM,
        P.PRD_BEG_DT,
        P.PRD_END_DT,
        P.ABAWD_FG,
        CASE
            WHEN P.PRD_BEG_DT >= D.CURR_MTH_BEG
             AND P.PRD_END_DT <= D.CURR_MTH_END THEN 1 ELSE 0
        END AS IN_CURR_MTH,
        CASE
            WHEN P.PRD_BEG_DT >= D.PREV_MTH_BEG
             AND P.PRD_END_DT <= D.PREV_MTH_END THEN 1 ELSE 0
        END AS IN_PREV_MTH,
        CASE
            WHEN P.PRD_BEG_DT >= D.FISCAL_START_DATE
             AND P.PRD_END_DT <= D.CURR_MTH_END THEN 1 ELSE 0
        END AS IN_FISCAL,
        CASE
            WHEN P.PRD_BEG_DT >= D.DT_36
             AND P.PRD_END_DT <= D.CURR_MTH_END THEN 1 ELSE 0
        END AS IN_36M
    FROM {?SCHEMA}.TWNPERSON_ELIG_HST P
    JOIN BASE_POP B
      ON P.UPI = B.UPI
    CROSS JOIN DATES D
    WHERE P.PRD_BEG_DT >= D.DT_36
      AND P.PRD_END_DT <= D.CURR_MTH_END
),

PERSON_METRICS AS (
    SELECT
        UPI,
        OFC_CD,
        OFC_NM,

        SUM(CASE WHEN ABAWD_FG NOT IN ('X','N') AND IN_FISCAL = 1 THEN 1 ELSE 0 END) AS LINE1_COUNT_FISCAL,
        SUM(CASE WHEN ABAWD_FG NOT IN ('X','N') AND IN_CURR_MTH = 1 THEN 1 ELSE 0 END) AS LINE1_COUNT_MONTH,

        SUM(CASE WHEN ABAWD_FG = 'Y' AND IN_CURR_MTH = 1 THEN 1 ELSE 0 END) AS Y_MONTH_COUNT,
        SUM(CASE WHEN ABAWD_FG = 'Y' AND IN_FISCAL = 1 THEN 1 ELSE 0 END) AS Y_FISCAL_COUNT,
        SUM(CASE WHEN ABAWD_FG = 'Y' AND IN_36M = 1 AND IN_CURR_MTH = 0 THEN 1 ELSE 0 END) AS Y_PRIOR_36_EXCL_MTH,
        SUM(CASE WHEN ABAWD_FG = 'Y' AND IN_36M = 1 AND IN_FISCAL = 0 THEN 1 ELSE 0 END) AS Y_PRIOR_36_EXCL_FIS,

        SUM(CASE WHEN ABAWD_FG = 'R' AND IN_CURR_MTH = 1 THEN 1 ELSE 0 END) AS R_MONTH_COUNT,
        SUM(CASE WHEN ABAWD_FG = 'R' AND IN_PREV_MTH = 1 THEN 1 ELSE 0 END) AS R_PREV_MONTH_COUNT,
        SUM(CASE WHEN ABAWD_FG = 'R' AND IN_FISCAL = 1 THEN 1 ELSE 0 END) AS R_FISCAL_COUNT,
        SUM(CASE WHEN ABAWD_FG = 'R' AND IN_36M = 1 AND IN_CURR_MTH = 0 THEN 1 ELSE 0 END) AS R_PRIOR_36_EXCL_MTH,
        SUM(CASE WHEN ABAWD_FG = 'R' AND IN_36M = 1 AND IN_FISCAL = 0 THEN 1 ELSE 0 END) AS R_PRIOR_36_EXCL_FIS,

        SUM(CASE WHEN ABAWD_FG = 'Z' AND IN_FISCAL = 1 THEN 1 ELSE 0 END) AS Z_FISCAL_COUNT,
        SUM(CASE WHEN ABAWD_FG = 'O' AND IN_FISCAL = 1 THEN 1 ELSE 0 END) AS O_FISCAL_COUNT,
        SUM(CASE WHEN ABAWD_FG = 'Z' AND IN_CURR_MTH = 1 THEN 1 ELSE 0 END) AS Z_MONTH_COUNT,
        SUM(CASE WHEN ABAWD_FG = 'O' AND IN_CURR_MTH = 1 THEN 1 ELSE 0 END) AS O_MONTH_COUNT,

        SUM(CASE WHEN ABAWD_FG = 'R' AND IN_FISCAL = 1 THEN 1 ELSE 0 END) AS WS_ABWD_CNTR_FIS
    FROM PERSON_36
    GROUP BY
        UPI,
        OFC_CD,
        OFC_NM
),

PERSON_FLAGS AS (
    SELECT
        UPI,
        OFC_CD,
        OFC_NM,
        LINE1_COUNT_FISCAL,
        LINE1_COUNT_MONTH,

        CASE
            WHEN Y_MONTH_COUNT = 1
                THEN Y_MONTH_COUNT + Y_PRIOR_36_EXCL_MTH
            ELSE Y_MONTH_COUNT
        END AS TOTAL_Y_MONTH_COUNT,

        CASE
            WHEN Y_FISCAL_COUNT > 0
                THEN Y_FISCAL_COUNT + Y_PRIOR_36_EXCL_FIS
            ELSE Y_FISCAL_COUNT
        END AS TOTAL_Y_FISCAL_COUNT,

        CASE
            WHEN R_MONTH_COUNT = 1
                THEN R_MONTH_COUNT + R_PRIOR_36_EXCL_MTH
            ELSE R_MONTH_COUNT
        END AS TOTAL_R_MONTH_COUNT,

        CASE
            WHEN R_FISCAL_COUNT > 0
                THEN R_FISCAL_COUNT + R_PRIOR_36_EXCL_FIS
            ELSE R_FISCAL_COUNT
        END AS TOTAL_R_FISCAL_COUNT,

        R_PREV_MONTH_COUNT,
        WS_ABWD_CNTR_FIS,

        Z_FISCAL_COUNT,
        O_FISCAL_COUNT,
        Z_MONTH_COUNT,
        O_MONTH_COUNT
    FROM PERSON_METRICS
),

MEMBER_FISCAL_MTH AS (
    SELECT
        C.UPI,
        C.OFC_CD,
        C.OFC_NM,
        M.PRD_BEG_DT,
        M.PRD_END_DT,
        M.ICI,
        M.ET_EXMPT_RSN_CD,
        M.ET_PRTCP_CD,
        M.FS_ASST_IND,
        CASE
            WHEN M.PRD_BEG_DT >= D.FISCAL_START_DATE
             AND M.PRD_END_DT <= D.FISCAL_END_DATE THEN 1 ELSE 0
        END AS IN_FISCAL,
        CASE
            WHEN M.PRD_BEG_DT >= D.CURR_MTH_BEG
             AND M.PRD_END_DT <= D.CURR_MTH_END THEN 1 ELSE 0
        END AS IN_CURR_MTH
    FROM {?SCHEMA}.TWNCASE_MEMBER_HST M
    JOIN CASE_UPI_OFFICE C
      ON M.ICI = C.ICI
     AND M.UPI = C.UPI
    CROSS JOIN DATES D
    WHERE M.PRD_BEG_DT >= D.FISCAL_START_DATE
      AND M.PRD_END_DT <= D.FISCAL_END_DATE
),

W_MEMBER_MATCH AS (
    SELECT
        M.UPI,
        M.OFC_CD,
        M.OFC_NM,
        M.ET_EXMPT_RSN_CD,
        M.ET_PRTCP_CD,
        M.FS_ASST_IND,
        M.IN_FISCAL,
        M.IN_CURR_MTH
    FROM MEMBER_FISCAL_MTH M
    JOIN PERSON_36 P
      ON M.UPI = P.UPI
     AND M.PRD_BEG_DT = P.PRD_BEG_DT
     AND M.PRD_END_DT = P.PRD_END_DT
     AND P.ABAWD_FG = 'W'
),

W_MEMBER_METRICS AS (
    SELECT
        UPI,
        OFC_CD,
        OFC_NM,

        SUM(CASE
                WHEN IN_FISCAL = 1
                 AND ET_EXMPT_RSN_CD NOT IN ('27','28')
                 AND FS_ASST_IND = 'Y' THEN 1 ELSE 0
            END) AS LINE3A1_FYR_COUNTS,

        SUM(CASE
                WHEN IN_CURR_MTH = 1
                 AND ET_EXMPT_RSN_CD NOT IN ('27','28')
                 AND FS_ASST_IND = 'Y' THEN 1 ELSE 0
            END) AS LINE3A1_MTH_COUNTS,

        SUM(CASE
                WHEN IN_FISCAL = 1
                 AND ET_EXMPT_RSN_CD = '27'
                 AND FS_ASST_IND = 'Y' THEN 1 ELSE 0
            END) AS LINE_3A2_FIS_CNTS,

        SUM(CASE
                WHEN IN_FISCAL = 1
                 AND ET_EXMPT_RSN_CD = '28'
                 AND FS_ASST_IND = 'Y' THEN 1 ELSE 0
            END) AS LINE_3A3_FIS_CNTS,

        SUM(CASE
                WHEN IN_CURR_MTH = 1
                 AND ET_EXMPT_RSN_CD = '27'
                 AND FS_ASST_IND = 'Y' THEN 1 ELSE 0
            END) AS LINE_3A2_MTH_CNTS,

        SUM(CASE
                WHEN IN_CURR_MTH = 1
                 AND ET_EXMPT_RSN_CD = '28'
                 AND FS_ASST_IND = 'Y' THEN 1 ELSE 0
            END) AS LINE_3A3_MTH_CNTS,

        SUM(CASE
                WHEN IN_FISCAL = 1
                 AND ET_PRTCP_CD = 'E'
                 AND ET_EXMPT_RSN_CD = '10'
                 AND FS_ASST_IND = 'Y' THEN 1 ELSE 0
            END) AS LINE_4A_FIS_CNTS,

        SUM(CASE
                WHEN IN_FISCAL = 1
                 AND ET_PRTCP_CD = 'E'
                 AND ET_EXMPT_RSN_CD IN ('25','26')
                 AND FS_ASST_IND = 'Y' THEN 1 ELSE 0
            END) AS LINE_4B_FIS_CNTS,

        SUM(CASE
                WHEN IN_FISCAL = 1
                 AND ET_PRTCP_CD = 'E'
                 AND ET_EXMPT_RSN_CD = '06'
                 AND FS_ASST_IND = 'Y' THEN 1 ELSE 0
            END) AS LINE_4C_FIS_CNTS,

        SUM(CASE
                WHEN IN_FISCAL = 1
                 AND ET_PRTCP_CD = 'E'
                 AND ET_EXMPT_RSN_CD = '18'
                 AND FS_ASST_IND = 'Y' THEN 1 ELSE 0
            END) AS LINE_4D_FIS_CNTS,

        SUM(CASE
                WHEN IN_CURR_MTH = 1
                 AND ET_PRTCP_CD = 'E'
                 AND ET_EXMPT_RSN_CD = '10'
                 AND FS_ASST_IND = 'Y' THEN 1 ELSE 0
            END) AS LINE_4A_MTH_CNTS,

        SUM(CASE
                WHEN IN_CURR_MTH = 1
                 AND ET_PRTCP_CD = 'E'
                 AND ET_EXMPT_RSN_CD IN ('25','26')
                 AND FS_ASST_IND = 'Y' THEN 1 ELSE 0
            END) AS LINE_4B_MTH_CNTS,

        SUM(CASE
                WHEN IN_CURR_MTH = 1
                 AND ET_PRTCP_CD = 'E'
                 AND ET_EXMPT_RSN_CD = '06'
                 AND FS_ASST_IND = 'Y' THEN 1 ELSE 0
            END) AS LINE_4C_MTH_CNTS,

        SUM(CASE
                WHEN IN_CURR_MTH = 1
                 AND ET_PRTCP_CD = 'E'
                 AND ET_EXMPT_RSN_CD = '18'
                 AND FS_ASST_IND = 'Y' THEN 1 ELSE 0
            END) AS LINE_4D_MTH_CNTS,

        SUM(CASE
                WHEN IN_FISCAL = 1
                 AND ET_PRTCP_CD = 'E'
                 AND ET_EXMPT_RSN_CD NOT IN ('06','10','18','25','26')
                 AND FS_ASST_IND = 'Y' THEN 1 ELSE 0
            END) AS LINE_4E_FIS_CNTS,

        SUM(CASE
                WHEN IN_CURR_MTH = 1
                 AND ET_PRTCP_CD = 'E'
                 AND ET_EXMPT_RSN_CD NOT IN ('06','10','18','25','26')
                 AND FS_ASST_IND = 'Y' THEN 1 ELSE 0
            END) AS LINE_4E_MTH_CNTS
    FROM W_MEMBER_MATCH
    GROUP BY
        UPI,
        OFC_CD,
        OFC_NM
),

ELIG_VER_MAX AS (
    SELECT
        ICI,
        PRGM_CASE_TYPE,
        BNFT_YR,
        BNFT_MTH,
        MAX(VER_NMB) AS VER_NMB
    FROM {?SCHEMA}.TWNELIG_VERSION
    WHERE PRGM_CASE_TYPE = 'FS'
      AND LST_POSTD_VER_IND = 'Y'
    GROUP BY
        ICI,
        PRGM_CASE_TYPE,
        BNFT_YR,
        BNFT_MTH
),

TSRSLT_BASE AS (
    SELECT
        C.UPI,
        C.OFC_CD,
        C.OFC_NM,
        T.BNFT_YR,
        T.BNFT_MTH,
        T.MBR_RSLT_CD,
        T.ABAWD_TST_RSCD,
        CASE
            WHEN (
                (T.BNFT_YR = YEAR(D.FISCAL_START_DATE) AND T.BNFT_MTH BETWEEN '07' AND '12')
                OR
                (T.BNFT_YR = YEAR(D.FISCAL_END_DATE) AND T.BNFT_MTH BETWEEN '01' AND '06')
            ) THEN 1 ELSE 0
        END AS IN_FISCAL,
        CASE
            WHEN T.BNFT_YR = YEAR(D.SYSTEM_DATE)
             AND T.BNFT_MTH =
                 CASE
                     WHEN MONTH(D.SYSTEM_DATE) < 10 THEN '0' || RTRIM(CHAR(MONTH(D.SYSTEM_DATE)))
                     ELSE RTRIM(CHAR(MONTH(D.SYSTEM_DATE)))
                 END
            THEN 1 ELSE 0
        END AS IN_CURR_MTH
    FROM {?SCHEMA}.TWNELIG_MBR_TSRSLT T
    JOIN CASE_UPI_OFFICE C
      ON C.ICI = T.ICI
     AND C.UPI = T.UPI
    JOIN ELIG_VER_MAX V
      ON T.ICI = V.ICI
     AND T.PRGM_CASE_TYPE = V.PRGM_CASE_TYPE
     AND T.BNFT_YR = V.BNFT_YR
     AND T.BNFT_MTH = V.BNFT_MTH
     AND T.VER_NMB = V.VER_NMB
    CROSS JOIN DATES D
    WHERE T.PRGM_CASE_TYPE = 'FS'
      AND T.MBR_RSLT_CD = 'I'
      AND T.RSRCE_TST_RSCD     <> 'F'
      AND T.SSN_TEST_RSLT_CD   <> 'F'
      AND T.SFU_TST_RSCD       <> 'F'
      AND T.SSN_OVRD_RSLT_CD   <> 'F'
      AND T.GRP_LVG_TST_RSCD   <> 'F'
      AND T.STDNT_TST_RSCD     <> 'F'
      AND T.FS_ATTND_TST_RSCD  <> 'F'
      AND T.BRDR_TST_RSCD      <> 'F'
      AND T.TEMP_ABSN_TST_RSCD <> 'F'
      AND T.STRK_TST_RSCD      <> 'F'
      AND T.TEMP_ABSN_OVRS_CD  <> 'F'
      AND T.LMPSM_INC_TSTRS_CD <> 'F'
      AND T.ALN_RES_TST_RSCD   <> 'F'
      AND T.CHD_AGE_TST_RSCD   <> 'F'
      AND T.HM_PRSNT_TST_RSCD  <> 'F'
      AND T.DUPL_AST_TST_RSCD  <> 'F'
      AND T.IPV_TEST_RSCD      <> 'F'
      AND T.WORK_PRTP_TST_RSCD <> 'F'
      AND T.CTZNP_TST_RSCD     <> 'F'
      AND T.FELON_TST_RSCD     <> 'F'
      AND T.DRGCNVT_TST_RSCD   <> 'F'
      AND T.ABAWD_TST_RSCD     =  'F'
),

TSRSLT_METRICS AS (
    SELECT
        T.UPI,
        T.OFC_CD,
        T.OFC_NM,
        SUM(CASE WHEN T.IN_FISCAL = 1 THEN 1 ELSE 0 END) AS WS_1ST_3YS_EXHST_CNT_FIS,
        SUM(CASE WHEN T.IN_CURR_MTH = 1 THEN 1 ELSE 0 END) AS WS_1ST_3YS_EXHST_CNT_MTH
    FROM TSRSLT_BASE T
    JOIN PERSON_FLAGS P
      ON P.UPI = T.UPI
     AND P.OFC_CD = T.OFC_CD
     AND P.OFC_NM = T.OFC_NM
    WHERE P.TOTAL_Y_FISCAL_COUNT >= 3
    GROUP BY
        T.UPI,
        T.OFC_CD,
        T.OFC_NM
),

UPI_FACTS AS (
    SELECT
        B.UPI,
        B.OFC_CD,
        B.OFC_NM,

        COALESCE(P.LINE1_COUNT_FISCAL, 0) AS LINE1_COUNT_FISCAL,
        COALESCE(P.LINE1_COUNT_MONTH, 0) AS LINE1_COUNT_MONTH,
        COALESCE(P.TOTAL_Y_MONTH_COUNT, 0) AS TOTAL_Y_MONTH_COUNT,
        COALESCE(P.TOTAL_Y_FISCAL_COUNT, 0) AS TOTAL_Y_FISCAL_COUNT,
        COALESCE(P.TOTAL_R_MONTH_COUNT, 0) AS TOTAL_R_MONTH_COUNT,
        COALESCE(P.TOTAL_R_FISCAL_COUNT, 0) AS TOTAL_R_FISCAL_COUNT,
        COALESCE(P.R_PREV_MONTH_COUNT, 0) AS R_PREV_MONTH_COUNT,
        COALESCE(P.WS_ABWD_CNTR_FIS, 0) AS WS_ABWD_CNTR_FIS,
        COALESCE(P.Z_FISCAL_COUNT, 0) AS Z_FISCAL_COUNT,
        COALESCE(P.O_FISCAL_COUNT, 0) AS O_FISCAL_COUNT,
        COALESCE(P.Z_MONTH_COUNT, 0) AS Z_MONTH_COUNT,
        COALESCE(P.O_MONTH_COUNT, 0) AS O_MONTH_COUNT,

        COALESCE(T.WS_1ST_3YS_EXHST_CNT_FIS, 0) AS WS_1ST_3YS_EXHST_CNT_FIS,
        COALESCE(T.WS_1ST_3YS_EXHST_CNT_MTH, 0) AS WS_1ST_3YS_EXHST_CNT_MTH,

        COALESCE(W.LINE3A1_FYR_COUNTS, 0) AS LINE3A1_FYR_COUNTS,
        COALESCE(W.LINE3A1_MTH_COUNTS, 0) AS LINE3A1_MTH_COUNTS,
        COALESCE(W.LINE_3A2_FIS_CNTS, 0) AS LINE_3A2_FIS_CNTS,
        COALESCE(W.LINE_3A3_FIS_CNTS, 0) AS LINE_3A3_FIS_CNTS,
        COALESCE(W.LINE_3A2_MTH_CNTS, 0) AS LINE_3A2_MTH_CNTS,
        COALESCE(W.LINE_3A3_MTH_CNTS, 0) AS LINE_3A3_MTH_CNTS,
        COALESCE(W.LINE_4A_FIS_CNTS, 0) AS LINE_4A_FIS_CNTS,
        COALESCE(W.LINE_4B_FIS_CNTS, 0) AS LINE_4B_FIS_CNTS,
        COALESCE(W.LINE_4C_FIS_CNTS, 0) AS LINE_4C_FIS_CNTS,
        COALESCE(W.LINE_4D_FIS_CNTS, 0) AS LINE_4D_FIS_CNTS,
        COALESCE(W.LINE_4A_MTH_CNTS, 0) AS LINE_4A_MTH_CNTS,
        COALESCE(W.LINE_4B_MTH_CNTS, 0) AS LINE_4B_MTH_CNTS,
        COALESCE(W.LINE_4C_MTH_CNTS, 0) AS LINE_4C_MTH_CNTS,
        COALESCE(W.LINE_4D_MTH_CNTS, 0) AS LINE_4D_MTH_CNTS,
        COALESCE(W.LINE_4E_FIS_CNTS, 0) AS LINE_4E_FIS_CNTS,
        COALESCE(W.LINE_4E_MTH_CNTS, 0) AS LINE_4E_MTH_CNTS
    FROM BASE_POP B
    LEFT JOIN PERSON_FLAGS P
      ON B.UPI = P.UPI
     AND B.OFC_CD = P.OFC_CD
     AND B.OFC_NM = P.OFC_NM
    LEFT JOIN TSRSLT_METRICS T
      ON B.UPI = T.UPI
     AND B.OFC_CD = T.OFC_CD
     AND B.OFC_NM = T.OFC_NM
    LEFT JOIN W_MEMBER_METRICS W
      ON B.UPI = W.UPI
     AND B.OFC_CD = W.OFC_CD
     AND B.OFC_NM = W.OFC_NM
),

FINAL_QUERY AS (
    SELECT
        OFC_NM,
        OFC_CD,

        SUM(CASE WHEN LINE1_COUNT_FISCAL = 1 THEN 1 ELSE 0 END) AS LINE1_FIS,
        SUM(CASE WHEN LINE1_COUNT_MONTH  = 1 THEN 1 ELSE 0 END) AS LINE1_MTH,

        SUM(CASE WHEN TOTAL_Y_MONTH_COUNT  = 1 THEN 1 ELSE 0 END) AS WS_MTH_Y_CNT_1O,
        SUM(CASE WHEN TOTAL_Y_MONTH_COUNT  = 2 THEN 1 ELSE 0 END) AS WS_MTH_Y_CNT_20,
        SUM(CASE WHEN TOTAL_Y_MONTH_COUNT  = 3 THEN 1 ELSE 0 END) AS WS_MTH_Y_CNT_30,

        SUM(CASE WHEN TOTAL_Y_FISCAL_COUNT = 1 THEN 1 ELSE 0 END) AS WS_FYR_Y_CNT_1O,
        SUM(CASE WHEN TOTAL_Y_FISCAL_COUNT = 2 THEN 1 ELSE 0 END) AS WS_FYR_Y_CNT_20,
        SUM(CASE WHEN TOTAL_Y_FISCAL_COUNT = 3 THEN 1 ELSE 0 END) AS WS_FYR_Y_CNT_30,

        SUM(CASE WHEN WS_1ST_3YS_EXHST_CNT_FIS >= 1 THEN 1 ELSE 0 END) AS LINE5_FIS,
        SUM(CASE WHEN WS_1ST_3YS_EXHST_CNT_MTH = 1 THEN 1 ELSE 0 END) AS LINE5_MTH,

        SUM(CASE WHEN TOTAL_R_MONTH_COUNT = 1 THEN 1 ELSE 0 END) AS WS_MTH_R_CNT_1O,
        SUM(CASE WHEN TOTAL_R_MONTH_COUNT = 2 THEN 1 ELSE 0 END) AS WS_MTH_R_CNT_20,
        SUM(CASE WHEN TOTAL_R_MONTH_COUNT = 3 THEN 1 ELSE 0 END) AS WS_MTH_R_CNT_30,

        COUNT(DISTINCT CASE
            WHEN R_PREV_MONTH_COUNT > 0 AND TOTAL_R_MONTH_COUNT <> 1
            THEN UPI END) AS LINE6_MTH,

        SUM(CASE WHEN TOTAL_R_FISCAL_COUNT = 1 THEN 1 ELSE 0 END) AS WS_FYR_R_CNT_1O,
        SUM(CASE WHEN TOTAL_R_FISCAL_COUNT = 2 THEN 1 ELSE 0 END) AS WS_FYR_R_CNT_20,
        COUNT(DISTINCT CASE WHEN TOTAL_R_FISCAL_COUNT = 3 THEN UPI END) AS WS_FYR_R_CNT_30,

        COUNT(DISTINCT CASE WHEN WS_ABWD_CNTR_FIS > 1 THEN UPI END) AS LINE6_FIS,

        SUM(CASE WHEN LINE3A1_FYR_COUNTS > 0 THEN 1 ELSE 0 END) AS WS_3A1_FIS,
        SUM(CASE WHEN LINE3A1_MTH_COUNTS = 1 THEN 1 ELSE 0 END) AS WS_3A1_MTH,

        SUM(CASE WHEN LINE_3A2_FIS_CNTS > 0 THEN 1 ELSE 0 END) AS WS_3A2_FIS,
        SUM(CASE WHEN LINE_3A3_FIS_CNTS > 0 THEN 1 ELSE 0 END) AS WS_3A3_FIS,

        SUM(CASE WHEN LINE_3A2_MTH_CNTS = 1 THEN 1 ELSE 0 END) AS WS_3A2_MTH,
        SUM(CASE WHEN LINE_3A3_MTH_CNTS = 1 THEN 1 ELSE 0 END) AS WS_3A3_MTH,

        SUM(CASE WHEN Z_FISCAL_COUNT > 0 THEN 1 ELSE 0 END) AS WS_3B_FIS,
        SUM(CASE WHEN O_FISCAL_COUNT > 0 THEN 1 ELSE 0 END) AS WS_3C_FIS,

        SUM(CASE WHEN Z_MONTH_COUNT = 1 THEN 1 ELSE 0 END) AS WS_3B_MTH,
        SUM(CASE WHEN O_MONTH_COUNT = 1 THEN 1 ELSE 0 END) AS WS_3C_MTH,

        SUM(CASE WHEN LINE_4A_FIS_CNTS > 0 THEN 1 ELSE 0 END) AS WS_4A_FIS,
        SUM(CASE WHEN LINE_4B_FIS_CNTS > 0 THEN 1 ELSE 0 END) AS WS_4B_FIS,
        SUM(CASE WHEN LINE_4C_FIS_CNTS > 0 THEN 1 ELSE 0 END) AS WS_4C_FIS,
        SUM(CASE WHEN LINE_4D_FIS_CNTS > 0 THEN 1 ELSE 0 END) AS WS_4D_FIS,

        SUM(CASE WHEN LINE_4A_MTH_CNTS = 1 THEN 1 ELSE 0 END) AS WS_4A_MTH,
        SUM(CASE WHEN LINE_4B_MTH_CNTS = 1 THEN 1 ELSE 0 END) AS WS_4B_MTH,
        SUM(CASE WHEN LINE_4C_MTH_CNTS = 1 THEN 1 ELSE 0 END) AS WS_4C_MTH,
        SUM(CASE WHEN LINE_4D_MTH_CNTS = 1 THEN 1 ELSE 0 END) AS WS_4D_MTH,

        SUM(CASE WHEN LINE_4E_FIS_CNTS > 0 THEN 1 ELSE 0 END) AS WS_4E_FIS,
        SUM(CASE WHEN LINE_4E_MTH_CNTS = 1 THEN 1 ELSE 0 END) AS WS_4E_MTH,

        0 AS SORT
    FROM UPI_FACTS
    GROUP BY
        OFC_NM,
        OFC_CD

    UNION ALL

    SELECT
        'NEVADA STATEWIDE' AS OFC_NM,
        'NS' AS OFC_CD,

        SUM(CASE WHEN LINE1_COUNT_FISCAL = 1 THEN 1 ELSE 0 END) AS LINE1_FIS,
        SUM(CASE WHEN LINE1_COUNT_MONTH  = 1 THEN 1 ELSE 0 END) AS LINE1_MTH,

        SUM(CASE WHEN TOTAL_Y_MONTH_COUNT  = 1 THEN 1 ELSE 0 END) AS WS_MTH_Y_CNT_1O,
        SUM(CASE WHEN TOTAL_Y_MONTH_COUNT  = 2 THEN 1 ELSE 0 END) AS WS_MTH_Y_CNT_20,
        SUM(CASE WHEN TOTAL_Y_MONTH_COUNT  = 3 THEN 1 ELSE 0 END) AS WS_MTH_Y_CNT_30,

        SUM(CASE WHEN TOTAL_Y_FISCAL_COUNT = 1 THEN 1 ELSE 0 END) AS WS_FYR_Y_CNT_1O,
        SUM(CASE WHEN TOTAL_Y_FISCAL_COUNT = 2 THEN 1 ELSE 0 END) AS WS_FYR_Y_CNT_20,
        SUM(CASE WHEN TOTAL_Y_FISCAL_COUNT = 3 THEN 1 ELSE 0 END) AS WS_FYR_Y_CNT_30,

        SUM(CASE WHEN WS_1ST_3YS_EXHST_CNT_FIS >= 1 THEN 1 ELSE 0 END) AS LINE5_FIS,
        SUM(CASE WHEN WS_1ST_3YS_EXHST_CNT_MTH = 1 THEN 1 ELSE 0 END) AS LINE5_MTH,

        SUM(CASE WHEN TOTAL_R_MONTH_COUNT = 1 THEN 1 ELSE 0 END) AS WS_MTH_R_CNT_1O,
        SUM(CASE WHEN TOTAL_R_MONTH_COUNT = 2 THEN 1 ELSE 0 END) AS WS_MTH_R_CNT_20,
        SUM(CASE WHEN TOTAL_R_MONTH_COUNT = 3 THEN 1 ELSE 0 END) AS WS_MTH_R_CNT_30,

        COUNT(DISTINCT CASE
            WHEN R_PREV_MONTH_COUNT > 0 AND TOTAL_R_MONTH_COUNT <> 1
            THEN UPI END) AS LINE6_MTH,

        SUM(CASE WHEN TOTAL_R_FISCAL_COUNT = 1 THEN 1 ELSE 0 END) AS WS_FYR_R_CNT_1O,
        SUM(CASE WHEN TOTAL_R_FISCAL_COUNT = 2 THEN 1 ELSE 0 END) AS WS_FYR_R_CNT_20,
        COUNT(DISTINCT CASE WHEN TOTAL_R_FISCAL_COUNT = 3 THEN UPI END) AS WS_FYR_R_CNT_30,

        COUNT(DISTINCT CASE WHEN WS_ABWD_CNTR_FIS > 1 THEN UPI END) AS LINE6_FIS,

        SUM(CASE WHEN LINE3A1_FYR_COUNTS > 0 THEN 1 ELSE 0 END) AS WS_3A1_FIS,
        SUM(CASE WHEN LINE3A1_MTH_COUNTS = 1 THEN 1 ELSE 0 END) AS WS_3A1_MTH,

        SUM(CASE WHEN LINE_3A2_FIS_CNTS > 0 THEN 1 ELSE 0 END) AS WS_3A2_FIS,
        SUM(CASE WHEN LINE_3A3_FIS_CNTS > 0 THEN 1 ELSE 0 END) AS WS_3A3_FIS,

        SUM(CASE WHEN LINE_3A2_MTH_CNTS = 1 THEN 1 ELSE 0 END) AS WS_3A2_MTH,
        SUM(CASE WHEN LINE_3A3_MTH_CNTS = 1 THEN 1 ELSE 0 END) AS WS_3A3_MTH,

        SUM(CASE WHEN Z_FISCAL_COUNT > 0 THEN 1 ELSE 0 END) AS WS_3B_FIS,
        SUM(CASE WHEN O_FISCAL_COUNT > 0 THEN 1 ELSE 0 END) AS WS_3C_FIS,

        SUM(CASE WHEN Z_MONTH_COUNT = 1 THEN 1 ELSE 0 END) AS WS_3B_MTH,
        SUM(CASE WHEN O_MONTH_COUNT = 1 THEN 1 ELSE 0 END) AS WS_3C_MTH,

        SUM(CASE WHEN LINE_4A_FIS_CNTS > 0 THEN 1 ELSE 0 END) AS WS_4A_FIS,
        SUM(CASE WHEN LINE_4B_FIS_CNTS > 0 THEN 1 ELSE 0 END) AS WS_4B_FIS,
        SUM(CASE WHEN LINE_4C_FIS_CNTS > 0 THEN 1 ELSE 0 END) AS WS_4C_FIS,
        SUM(CASE WHEN LINE_4D_FIS_CNTS > 0 THEN 1 ELSE 0 END) AS WS_4D_FIS,

        SUM(CASE WHEN LINE_4A_MTH_CNTS = 1 THEN 1 ELSE 0 END) AS WS_4A_MTH,
        SUM(CASE WHEN LINE_4B_MTH_CNTS = 1 THEN 1 ELSE 0 END) AS WS_4B_MTH,
        SUM(CASE WHEN LINE_4C_MTH_CNTS = 1 THEN 1 ELSE 0 END) AS WS_4C_MTH,
        SUM(CASE WHEN LINE_4D_MTH_CNTS = 1 THEN 1 ELSE 0 END) AS WS_4D_MTH,

        SUM(CASE WHEN LINE_4E_FIS_CNTS > 0 THEN 1 ELSE 0 END) AS WS_4E_FIS,
        SUM(CASE WHEN LINE_4E_MTH_CNTS = 1 THEN 1 ELSE 0 END) AS WS_4E_MTH,

        1 AS SORT
    FROM UPI_FACTS
)

SELECT
    OFC_NM,
    OFC_CD,
    LINE1_FIS,
    LINE1_MTH,
    WS_MTH_Y_CNT_1O,
    WS_MTH_Y_CNT_20,
    WS_MTH_Y_CNT_30,
    WS_FYR_Y_CNT_1O,
    WS_FYR_Y_CNT_20,
    WS_FYR_Y_CNT_30,
    LINE5_FIS,
    LINE5_MTH,
    WS_MTH_R_CNT_1O,
    WS_MTH_R_CNT_20,
    WS_MTH_R_CNT_30,
    LINE6_MTH,
    WS_FYR_R_CNT_1O,
    WS_FYR_R_CNT_20,
    WS_FYR_R_CNT_30,
    LINE6_FIS,
    WS_3A1_FIS,
    WS_3A1_MTH,
    WS_3A2_FIS,
    WS_3A3_FIS,
    WS_3A2_MTH,
    WS_3A3_MTH,
    WS_3B_FIS,
    WS_3C_FIS,
    WS_3B_MTH,
    WS_3C_MTH,
    WS_4A_FIS,
    WS_4B_FIS,
    WS_4C_FIS,
    WS_4D_FIS,
    WS_4A_MTH,
    WS_4B_MTH,
    WS_4C_MTH,
    WS_4D_MTH,
    WS_4E_FIS,
    WS_4E_MTH,
    SUM(
        LINE1_FIS + LINE1_MTH + WS_MTH_Y_CNT_1O + WS_MTH_Y_CNT_20 + WS_MTH_Y_CNT_30 +
        WS_FYR_Y_CNT_1O + WS_FYR_Y_CNT_20 + WS_FYR_Y_CNT_30 + LINE5_FIS + LINE5_MTH +
        WS_MTH_R_CNT_1O + WS_MTH_R_CNT_20 + WS_MTH_R_CNT_30 + LINE6_MTH + WS_FYR_R_CNT_1O +
        WS_FYR_R_CNT_20 + WS_FYR_R_CNT_30 + LINE6_FIS + WS_3A1_FIS + WS_3A1_MTH +
        WS_3A2_FIS + WS_3A3_FIS + WS_3A2_MTH + WS_3A3_MTH + WS_3B_FIS + WS_3C_FIS +
        WS_3B_MTH + WS_3C_MTH + WS_4A_FIS + WS_4B_FIS + WS_4C_FIS + WS_4D_FIS +
        WS_4A_MTH + WS_4B_MTH + WS_4C_MTH + WS_4D_MTH + WS_4E_FIS + WS_4E_MTH
    ) OVER() AS NO_DATA,
    D.SYSTEM_DATE AS SYS_DATE
FROM FINAL_QUERY F
CROSS JOIN DATES D
ORDER BY SORT, OFC_CD
FOR FETCH ONLY WITH UR;
