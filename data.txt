 for (PersonData person : people) {
            JAXBElement<? extends ComplexObjectType> memberElement;

            if ("HOH".equalsIgnoreCase(person.relationship)) {
                PrimaryTaxFilerType primary = new PrimaryTaxFilerType();
                primary.setTINIdentification(createTIN(person));
                primary.setRoleOfPersonReference(createRef(person.personId));
                memberElement = new JAXBElement<>(new QName("http://hix.cms.gov/0.1/hix-ee", "PrimaryTaxFiler"), PrimaryTaxFilerType.class, primary);
            } else if ("Spouse".equalsIgnoreCase(person.relationship)) {
                SpouseTaxFilerType spouse = new SpouseTaxFilerType();
                spouse.setTINIdentification(createTIN(person));
                spouse.setRoleOfPersonReference(createRef(person.personId));
                memberElement = new JAXBElement<>(new QName("http://hix.cms.gov/0.1/hix-ee", "SpouseTaxFiler"), SpouseTaxFilerType.class, spouse);
            } else if (person.isDependent()) {
                TaxDependentType dependent = new TaxDependentType();
                dependent.setTINIdentification(createTIN(person));
                dependent.setRoleOfPersonReference(createRef(person.personId));
                memberElement = new JAXBElement<>(new QName("http://hix.cms.gov/0.1/hix-ee", "TaxDependent"), TaxDependentType.class, dependent);
            } else {
                continue;
            }

            household.getPrimaryTaxFilerOrSpouseTaxFilerOrTaxDependent().add(memberElement);
        }

        // 4. Add household back to tax return
        taxReturn.setTaxHousehold(household);
